85 static term externalterm_to_term_internal(const void *external_term, size_t size, Context *ctx,
86  ExternalTermOpts opts, size_t *bytes_read, bool copy)
87 {
88  const uint8_t *external_term_buf = (const uint8_t *) external_term;
89 
90  if (UNLIKELY(external_term_buf[0] != EXTERNAL_TERM_TAG)) {
91  return term_invalid_term();
92  }
93 
94  if (size == 0) {
95  return term_invalid_term();
96  }
97 
98  size_t eterm_size;
99  int heap_usage = calculate_heap_usage(external_term_buf + 1, size - 1, &eterm_size, copy);
100  if (heap_usage == INVALID_TERM_SIZE) {
101  return term_invalid_term();
102  }
103 
104  term result;
105  if (opts & ExternalTermToHeapFragment) {
106  // We need to allocate fragments as reading external terms from modules
107  // is not accounted for by the compiler when it emits test_heap opcodes
108  Heap heap;
109  if (UNLIKELY(memory_init_heap(&heap, heap_usage) != MEMORY_GC_OK)) {
110  return term_invalid_term();
111  }
112  result = parse_external_terms(external_term_buf + 1, &eterm_size, copy, &heap, ctx->global);
113  memory_heap_append_heap(&ctx->heap, &heap);
114  } else {
115  if (UNLIKELY(memory_ensure_free(ctx, heap_usage) != MEMORY_GC_OK)) {
116  fprintf(stderr, "Unable to ensure %zu free words in heap\n", eterm_size);
117  return term_invalid_term();
118  }
119  result = parse_external_terms(external_term_buf + 1, &eterm_size, copy, &ctx->heap, ctx->global);
120  }
121  *bytes_read = eterm_size + 1;
122  return result;
123 }

131 enum ExternalTermResult externalterm_from_binary(Context *ctx, term *dst, term binary, size_t *bytes_read)
132 {
133  if (!term_is_binary(binary)) {
134  return EXTERNAL_TERM_BAD_ARG;
135  }
136  //
137  // Copy the binary data to a buffer (in case of GC)
138  //
139  size_t len = term_binary_size(binary);
140  const uint8_t *data = (const uint8_t *) term_binary_data(binary);
141  uint8_t *buf = malloc(len);
142  if (IS_NULL_PTR(buf)) {
143  fprintf(stderr, "Unable to allocate %zu bytes for binary buffer.\n", len);
144  return EXTERNAL_TERM_MALLOC;
145  }
146  memcpy(buf, data, len);
147  //
148  // convert
149  //
150  term t = externalterm_to_term_internal(buf, len, ctx, false, bytes_read, true);
151  free(buf);
152  if (term_is_invalid_term(t)) {
153  return EXTERNAL_TERM_BAD_ARG;
154  } else {
155  *dst = t;
156  return EXTERNAL_TERM_OK;
157  }
158 }

160 static int externalterm_from_term(uint8_t **buf, size_t *len, term t, GlobalContext *glb)
161 {
162  *len = compute_external_size(t, glb) + 1;
163  *buf = malloc(*len);
164  if (IS_NULL_PTR(*buf)) {
165  fprintf(stderr, "Unable to allocate %zu bytes for externalized term.\n", *len);
166  AVM_ABORT();
167  }
168  size_t k = serialize_term(*buf + 1, t, glb);
169  *buf[0] = EXTERNAL_TERM_TAG;
170  return k + 1;
171 }

173 term externalterm_to_binary(Context *ctx, term t)
174 {
175  //
176  // convert
177  //
178  uint8_t *buf;
179  size_t len;
180  externalterm_from_term(&buf, &len, t, ctx->global);
181  //
182  // Ensure enough free space in heap for binary
183  //
184  int size_in_terms = term_binary_heap_size(len);
185  if (UNLIKELY(memory_ensure_free(ctx, size_in_terms) != MEMORY_GC_OK)) {
186  fprintf(stderr, "Unable to ensure %i free words in heap\n", size_in_terms);
187  return term_invalid_term();
188  }
189  //
190  // create and return the binary
191  //
192  term binary = term_from_literal_binary((void *) buf, len, &ctx->heap, ctx->global);
193  free(buf);
194  return binary;
195 }

212 static void write_bytes(uint8_t *buf, avm_uint64_t val)
213 {
214  uint8_t i = 0;
215  while (val != 0) {
216  uint8_t byte = val & 0xFF;
217  buf[i] = byte;
218  val = val >> 8;
219  ++i;
220  }
221 }

223 static int serialize_term(uint8_t *buf, term t, GlobalContext *glb)
224 {
225  if (term_is_uint8(t)) {
226  if (!IS_NULL_PTR(buf)) {
227  buf[0] = SMALL_INTEGER_EXT;
228  buf[1] = term_to_uint8(t);
229  }
230  return 2;
231 
232  } else if (term_is_any_integer(t)) {
233 
234  avm_int64_t val = term_maybe_unbox_int64(t);
235  if (val >= INT32_MIN && val <= INT32_MAX) {
236  if (buf != NULL) {
237  buf[0] = INTEGER_EXT;
238  WRITE_32_UNALIGNED(buf + 1, (int32_t) val);
239  }
240  return INTEGER_EXT_SIZE;
241  } else {
242  avm_uint64_t unsigned_val = SIGNED_INT_TO_UNSIGNED(val, avm_uint64_t);
243  uint8_t num_bytes = get_num_bytes(unsigned_val);
244  if (buf != NULL) {
245  buf[0] = SMALL_BIG_EXT;
246  buf[1] = num_bytes;
247  buf[2] = val < 0 ? 0x01 : 0x00;
248  write_bytes(buf + 3, unsigned_val);
249  }
250  return SMALL_BIG_EXT_BASE_SIZE + num_bytes;
251  }
252 
253  } else if (term_is_float(t)) {
254  if (!IS_NULL_PTR(buf)) {
255  avm_float_t val = term_to_float(t);
256  buf[0] = NEW_FLOAT_EXT;
257  union {
258  uint64_t intvalue;
259  double doublevalue;
260  } v;
261  v.doublevalue = val;
262  WRITE_64_UNALIGNED(buf + 1, v.intvalue);
263  }
264  return NEW_FLOAT_EXT_SIZE;
265 
266  } else if (term_is_atom(t)) {
267  AtomString atom_string = globalcontext_atomstring_from_term(glb, t);
268  size_t atom_len = atom_string_len(atom_string);
269  if (!IS_NULL_PTR(buf)) {
270  buf[0] = ATOM_EXT;
271  WRITE_16_UNALIGNED(buf + 1, atom_len);
272  int8_t *atom_data = (int8_t *) atom_string_data(atom_string);
273  for (size_t i = 3; i < atom_len + 3; ++i) {
274  buf[i] = (int8_t) atom_data[i - 3];
275  }
276  }
277  return 3 + atom_len;
278 
279  } else if (term_is_tuple(t)) {
280  size_t arity = term_get_tuple_arity(t);
281  if (arity > 255) {
282  fprintf(stderr, "Tuple arity greater than 255: %zu\n", arity);
283  AVM_ABORT();
284  }
285  if (!IS_NULL_PTR(buf)) {
286  buf[0] = SMALL_TUPLE_EXT;
287  buf[1] = (int8_t) arity;
288  }
289  size_t k = 2;
290  for (size_t i = 0; i < arity; ++i) {
291  term e = term_get_tuple_element(t, i);
292  k += serialize_term(IS_NULL_PTR(buf) ? NULL : buf + k, e, glb);
293  }
294  return k;
295 
296  } else if (term_is_nil(t)) {
297  if (!IS_NULL_PTR(buf)) {
298  buf[0] = NIL_EXT;
299  }
300  return 1;
301 
302  } else if (term_is_string(t)) {
303  if (!IS_NULL_PTR(buf)) {
304  buf[0] = STRING_EXT;
305  }
306  size_t len = 0;
307  size_t k = 3;
308  term i = t;
309  while (!term_is_nil(i)) {
310  term e = term_get_list_head(i);
311  if (!IS_NULL_PTR(buf)) {
312  *(buf + k) = term_to_uint8(e);
313  }
314  ++k;
315  i = term_get_list_tail(i);
316  ++len;
317  }
318  if (!IS_NULL_PTR(buf)) {
319  WRITE_16_UNALIGNED(buf + 1, len);
320  }
321  return k;
322 
323  } else if (term_is_list(t)) {
324  if (!IS_NULL_PTR(buf)) {
325  buf[0] = LIST_EXT;
326  }
327  size_t len = 0;
328  size_t k = 5;
329  term i = t;
330  while (term_is_nonempty_list(i)) {
331  term e = term_get_list_head(i);
332  k += serialize_term(IS_NULL_PTR(buf) ? NULL : buf + k, e, glb);
333  i = term_get_list_tail(i);
334  ++len;
335  }
336  k += serialize_term(IS_NULL_PTR(buf) ? NULL : buf + k, i, glb);
337  if (!IS_NULL_PTR(buf)) {
338  WRITE_32_UNALIGNED(buf + 1, len);
339  }
340  return k;
341 
342  } else if (term_is_binary(t)) {
343  if (!IS_NULL_PTR(buf)) {
344  buf[0] = BINARY_EXT;
345  }
346  size_t len = term_binary_size(t);
347  if (!IS_NULL_PTR(buf)) {
348  const uint8_t *data = (const uint8_t *) term_binary_data(t);
349  WRITE_32_UNALIGNED(buf + 1, len);
350  memcpy(buf + 5, data, len);
351  }
352  return 5 + len;
353 
354  } else if (term_is_map(t)) {
355  size_t size = term_get_map_size(t);
356  if (!IS_NULL_PTR(buf)) {
357  buf[0] = MAP_EXT;
358  WRITE_32_UNALIGNED(buf + 1, size);
359  }
360  size_t k = 5;
361  for (size_t i = 0; i < size; ++i) {
362  term key = term_get_map_key(t, i);
363  k += serialize_term(IS_NULL_PTR(buf) ? NULL : buf + k, key, glb);
364  term value = term_get_map_value(t, i);
365  k += serialize_term(IS_NULL_PTR(buf) ? NULL : buf + k, value, glb);
366  }
367  return k;
368  } else if (term_is_function(t)) {
369  if (!IS_NULL_PTR(buf)) {
370  buf[0] = EXPORT_EXT;
371  }
372  size_t k = 1;
373  const term *boxed_value = term_to_const_term_ptr(t);
374  for (size_t i = 1; i <= 3; ++i) {
375  term mfa = boxed_value[i];
376  k += serialize_term(IS_NULL_PTR(buf) ? NULL : buf + k, mfa, glb);
377  }
378  return k;
379  } else {
380  fprintf(stderr, "Unknown external term type: %" TERM_U_FMT "\n", t);
381  AVM_ABORT();
382  }
383 }

385 static avm_uint64_t read_bytes(const uint8_t *buf, uint8_t num_bytes)
386 {
387  avm_uint64_t value = 0;
388  for (uint8_t i = 0; i < num_bytes; ++i) {
389  value |= (((avm_uint64_t) buf[i]) << (i * 8));
390  }
391  return value;
392 }

394 static term parse_external_terms(const uint8_t *external_term_buf, size_t *eterm_size, bool copy, Heap *heap, GlobalContext *glb)
395 {
396  switch (external_term_buf[0]) {
397  case NEW_FLOAT_EXT: {
398  union {
399  uint64_t intvalue;
400  double doublevalue;
401  } v;
402  v.intvalue = READ_64_UNALIGNED(external_term_buf + 1);
403 
404  *eterm_size = NEW_FLOAT_EXT_SIZE;
405  return term_from_float(v.doublevalue, heap);
406  }
407 
408  case SMALL_INTEGER_EXT: {
409  *eterm_size = 2;
410  return term_from_int11(external_term_buf[1]);
411  }
412 
413  case INTEGER_EXT: {
414  int32_t value = READ_32_UNALIGNED(external_term_buf + 1);
415 
416  *eterm_size = 5;
417 
418  return term_make_maybe_boxed_int64(value, heap);
419  }
420 
421  case SMALL_BIG_EXT: {
422  uint8_t num_bytes = external_term_buf[1];
423  uint8_t sign = external_term_buf[2];
424  avm_uint64_t unsigned_value = read_bytes(external_term_buf + 3, num_bytes);
425  // NB due to call to calculate_heap_usage, there is no loss of precision:
426  // 1. 0 <= unsigned_value <= INT64_MAX if sign is 0
427  // 2. 0 <= unsigned_value <= INT64_MAX + 1 if sign is not 0
428  avm_int64_t value = 0;
429  if (sign != 0x00) {
430  value = -((avm_int64_t) unsigned_value);
431  } else {
432  value = (avm_int64_t) unsigned_value;
433  }
434  *eterm_size = SMALL_BIG_EXT_BASE_SIZE + num_bytes;
435  return term_make_maybe_boxed_int64(value, heap);
436  }
437 
438  case ATOM_EXT: {
439  uint16_t atom_len = READ_16_UNALIGNED(external_term_buf + 1);
440 
441  int global_atom_id = globalcontext_insert_atom_maybe_copy(glb, (AtomString) (external_term_buf + 2), copy);
442 
443  *eterm_size = 3 + atom_len;
444  return term_from_atom_index(global_atom_id);
445  }
446 
447  case SMALL_TUPLE_EXT: {
448  uint8_t arity = external_term_buf[1];
449  term tuple = term_alloc_tuple(arity, heap);
450 
451  int buf_pos = 2;
452 
453  for (int i = 0; i < arity; i++) {
454  size_t element_size;
455  term put_value = parse_external_terms(external_term_buf + buf_pos, &element_size, copy, heap, glb);
456  if (UNLIKELY(term_is_invalid_term(put_value))) {
457  return put_value;
458  }
459  term_put_tuple_element(tuple, i, put_value);
460 
461  buf_pos += element_size;
462  }
463 
464  *eterm_size = buf_pos;
465  return tuple;
466  }
467 
468  case NIL_EXT: {
469  *eterm_size = 1;
470  return term_nil();
471  }
472 
473  case STRING_EXT: {
474  uint16_t string_size = READ_16_UNALIGNED(external_term_buf + 1);
475  *eterm_size = 3 + string_size;
476  return term_from_string((uint8_t *) external_term_buf + 3, string_size, heap);
477  }
478 
479  case LIST_EXT: {
480  uint32_t list_len = READ_32_UNALIGNED(external_term_buf + 1);
481 
482  term list_begin = term_nil();
483  term *prev_term = NULL;
484 
485  int buf_pos = 5;
486 
487  for (unsigned int i = 0; i < list_len; i++) {
488  size_t item_size;
489  term head = parse_external_terms(external_term_buf + buf_pos, &item_size, copy, heap, glb);
490  if (UNLIKELY(term_is_invalid_term(head))) {
491  return head;
492  }
493  term *new_list_item = term_list_alloc(heap);
494 
495  if (prev_term) {
496  prev_term[0] = term_list_from_list_ptr(new_list_item);
497  } else {
498  list_begin = term_list_from_list_ptr(new_list_item);
499  }
500 
501  prev_term = new_list_item;
502  new_list_item[1] = head;
503 
504  buf_pos += item_size;
505  }
506 
507  if (prev_term) {
508  size_t tail_size;
509  term tail = parse_external_terms(external_term_buf + buf_pos, &tail_size, copy, heap, glb);
510  if (UNLIKELY(term_is_invalid_term(tail))) {
511  return tail;
512  }
513  prev_term[0] = tail;
514  buf_pos += tail_size;
515  }
516 
517  *eterm_size = buf_pos;
518  return list_begin;
519  }
520 
521  case BINARY_EXT: {
522  uint32_t binary_size = READ_32_UNALIGNED(external_term_buf + 1);
523  *eterm_size = 5 + binary_size;
524  if (copy) {
525  return term_from_literal_binary((uint8_t *) external_term_buf + 5, binary_size, heap, glb);
526  } else {
527  return term_from_const_binary((uint8_t *) external_term_buf + 5, binary_size, heap, glb);
528  }
529  }
530 
531  case EXPORT_EXT: {
532  size_t buf_pos = 1;
533  size_t element_size;
534 
535  term m = parse_external_terms(external_term_buf + buf_pos, &element_size, copy, heap, glb);
536  if (UNLIKELY(term_is_invalid_term(m))) {
537  return m;
538  }
539  buf_pos += element_size;
540 
541  term f = parse_external_terms(external_term_buf + buf_pos, &element_size, copy, heap, glb);
542  if (UNLIKELY(term_is_invalid_term(f))) {
543  return f;
544  }
545  buf_pos += element_size;
546 
547  term a = parse_external_terms(external_term_buf + buf_pos, &element_size, copy, heap, glb);
548  if (UNLIKELY(term_is_invalid_term(a))) {
549  return a;
550  }
551  buf_pos += element_size;
552 
553  *eterm_size = buf_pos;
554  return term_make_function_reference(m, f, a, heap);
555  }
556 
557  case MAP_EXT: {
558  uint32_t size = READ_32_UNALIGNED(external_term_buf + 1);
559  term map = term_alloc_map(size, heap);
560  size_t buf_pos = 5;
561  for (uint32_t i = 0; i < size; ++i) {
562  size_t key_size;
563  term key = parse_external_terms(external_term_buf + buf_pos, &key_size, copy, heap, glb);
564  if (UNLIKELY(term_is_invalid_term(key))) {
565  return key;
566  }
567  buf_pos += key_size;
568 
569  size_t value_size;
570  term value = parse_external_terms(external_term_buf + buf_pos, &value_size, copy, heap, glb);
571  if (UNLIKELY(term_is_invalid_term(value))) {
572  return value;
573  }
574  buf_pos += value_size;
575 
576  term_set_map_assoc(map, i, key, value);
577  }
578  *eterm_size = buf_pos;
579  return map;
580  }
581 
582  case SMALL_ATOM_UTF8_EXT: {
583  uint8_t atom_len = *(external_term_buf + 1);
584 
585  // AtomString first byte is the atom length
586  int global_atom_id = globalcontext_insert_atom_maybe_copy(glb, (AtomString) (external_term_buf + 1), copy);
587 
588  *eterm_size = 2 + atom_len;
589  return term_from_atom_index(global_atom_id);
590  }
591 
592  default:
593  return term_invalid_term();
594  }
595 }

597 static int calculate_heap_usage(const uint8_t *external_term_buf, size_t remaining, size_t *eterm_size, bool copy)
598 {
599  if (UNLIKELY(remaining < 1)) {
600  return INVALID_TERM_SIZE;
601  }
602  switch (external_term_buf[0]) {
603  case NEW_FLOAT_EXT: {
604  if (UNLIKELY(remaining < NEW_FLOAT_EXT_SIZE)) {
605  return INVALID_TERM_SIZE;
606  }
607  *eterm_size = NEW_FLOAT_EXT_SIZE;
608  return FLOAT_SIZE;
609  }
610 
611  case SMALL_INTEGER_EXT: {
612  if (UNLIKELY(remaining < SMALL_INTEGER_EXT_SIZE)) {
613  return INVALID_TERM_SIZE;
614  }
615  *eterm_size = SMALL_INTEGER_EXT_SIZE;
616  return 0;
617  }
618 
619  case INTEGER_EXT: {
620  if (UNLIKELY(remaining < INTEGER_EXT_SIZE)) {
621  return INVALID_TERM_SIZE;
622  }
623  int32_t value = READ_32_UNALIGNED(external_term_buf + 1);
624  *eterm_size = INTEGER_EXT_SIZE;
625  return term_boxed_integer_size(value);
626  }
627 
628  case SMALL_BIG_EXT: {
629  uint8_t num_bytes = external_term_buf[1];

637  if (UNLIKELY((sign == 0 && unsigned_value > INT64_MAX) || (sign != 0 && unsigned_value > (((avm_uint64_t) INT64_MAX) + 1)))) {
638  return INVALID_TERM_SIZE;
639  }

641  // on 1 term while 2^27 and 2^59 respectively (32/64 bits) cannot.
642  avm_int64_t value = 0;
643  if (sign != 0x00) {
644  value = -((avm_int64_t) unsigned_value);
