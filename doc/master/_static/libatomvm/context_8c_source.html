40 Context *context_new(GlobalContext *glb)
41 {
42  Context *ctx = malloc(sizeof(Context));
43  if (IS_NULL_PTR(ctx)) {
44  fprintf(stderr, "Failed to allocate memory: %s:%i.\n", __FILE__, __LINE__);
45  return NULL;
46  }
47  ctx->cp = 0;
48 
49  ctx->heap_start = (term *) calloc(DEFAULT_STACK_SIZE, sizeof(term));
50  if (IS_NULL_PTR(ctx->heap_start)) {
51  fprintf(stderr, "Failed to allocate memory: %s:%i.\n", __FILE__, __LINE__);
52  free(ctx);
53  return NULL;
54  }
55  ctx->stack_base = ctx->heap_start + DEFAULT_STACK_SIZE;
56  ctx->e = ctx->stack_base;
57  ctx->heap_ptr = ctx->heap_start;
58 
59  context_clean_registers(ctx, 0);
60 
61  ctx->fr = NULL;
62 
63  ctx->min_heap_size = 0;
64  ctx->max_heap_size = 0;
65  ctx->has_min_heap_size = 0;
66  ctx->has_max_heap_size = 0;
67 
68  list_append(&glb->ready_processes, &ctx->processes_list_head);
69 
70  list_init(&ctx->mailbox);
71  list_init(&ctx->save_queue);
72  list_init(&ctx->dictionary);
73 
74  ctx->global = glb;
75 
76  ctx->process_id = globalcontext_get_new_process_id(glb);
77  list_append(&glb->processes_table, &ctx->processes_table_head);
78 
79  ctx->native_handler = NULL;
80 
81  ctx->saved_ip = NULL;
82  ctx->jump_to_on_restore = NULL;
83 
84  ctx->leader = 0;
85 
86  timer_wheel_item_init(&ctx->timer_wheel_head, NULL, 0);
87 
88  list_init(&ctx->monitors_head);
89 
90  ctx->trap_exit = false;
91  #ifdef ENABLE_ADVANCED_TRACE
92  ctx->trace_calls = 0;
93  ctx->trace_call_args = 0;
94  ctx->trace_returns = 0;
95  ctx->trace_send = 0;
96  ctx->trace_receive = 0;
97  #endif
98 
99  list_init(&ctx->heap_fragments);
100  ctx->heap_fragments_size = 0;
101 
102  ctx->flags = 0;
103 
104  ctx->platform_data = NULL;
105 
106  ctx->group_leader = term_from_local_process_id(INVALID_PROCESS_ID);
107 
108  ctx->bs = term_invalid_term();
109  ctx->bs_offset = 0;
110 
111  ctx->exit_reason = NORMAL_ATOM;
112  ctx->mso_list = term_nil();
113 
114  return ctx;
115 }

159 static void context_monitors_handle_terminate(Context *ctx)
160 {
161  struct ListHead *item;
162  struct ListHead *tmp;
163  MUTABLE_LIST_FOR_EACH (item, tmp, &ctx->monitors_head) {
164  struct Monitor *monitor = GET_LIST_ENTRY(item, struct Monitor, monitor_list_head);
165  int local_process_id = term_to_local_process_id(monitor->monitor_pid);
166  Context *target = globalcontext_get_process(ctx->global, local_process_id);
167  if (IS_NULL_PTR(target)) {
168  // TODO: we should scan for existing monitors when a context is destroyed
169  // otherwise memory might be wasted for long living processes
170  free(monitor);
171  continue;
172  }
173 
174  if (monitor->linked && (ctx->exit_reason != NORMAL_ATOM || target->trap_exit)) {
175  if (target->trap_exit) {
176  if (UNLIKELY(memory_ensure_free(ctx, TUPLE_SIZE(3)) != MEMORY_GC_OK)) {
177  //TODO: handle out of memory here
178  fprintf(stderr, "Cannot handle out of memory.\n");
179  AVM_ABORT();
180  }
181 
182  // TODO: move it out of heap
183  term info_tuple = term_alloc_tuple(3, ctx);
184  term_put_tuple_element(info_tuple, 0, EXIT_ATOM);
185  term_put_tuple_element(info_tuple, 1, term_from_local_process_id(ctx->process_id));
186  term_put_tuple_element(info_tuple, 2, ctx->exit_reason);
187 
188  mailbox_send(target, info_tuple);
189  } else {
190  size_t estimated_mem_usage = memory_estimate_usage(ctx->exit_reason);
191  if (UNLIKELY(memory_ensure_free(target, estimated_mem_usage) != MEMORY_GC_OK)) {
192  //TODO: handle out of memory here
193  fprintf(stderr, "Cannot handle out of memory.\n");
194  AVM_ABORT();
195  }
196  target->exit_reason = memory_copy_term_tree(&target->heap_ptr, ctx->exit_reason, &target->mso_list);
197 
198  // TODO: this cannot work on multicore systems
199  // target context should be marked as killed and terminated during next scheduling
200  scheduler_terminate(target);
201  }
202  } else if (!monitor->linked) {
203  int required_terms = REF_SIZE + TUPLE_SIZE(5);
204  if (UNLIKELY(memory_ensure_free(ctx, required_terms) != MEMORY_GC_OK)) {
205  //TODO: handle out of memory here
206  fprintf(stderr, "Cannot handle out of memory.\n");
207  AVM_ABORT();
208  }
209 
210  // TODO: move it out of heap
211  term ref = term_from_ref_ticks(monitor->ref_ticks, ctx);
212 
213  term info_tuple = term_alloc_tuple(5, ctx);
214  term_put_tuple_element(info_tuple, 0, DOWN_ATOM);
215  term_put_tuple_element(info_tuple, 1, ref);
216  term_put_tuple_element(info_tuple, 2, PROCESS_ATOM);
217  term_put_tuple_element(info_tuple, 3, term_from_local_process_id(ctx->process_id));
218  term_put_tuple_element(info_tuple, 4, ctx->exit_reason);
219 
220  mailbox_send(target, info_tuple);
221  }
222  free(monitor);
223  }
224 }

226 uint64_t context_monitor(Context *ctx, term monitor_pid, bool linked)
227 {
228  uint64_t ref_ticks = globalcontext_get_ref_ticks(ctx->global);
229 
230  struct Monitor *monitor = malloc(sizeof(struct Monitor));
231  if (IS_NULL_PTR(monitor)) {
232  return 0;
233  }
234  monitor->monitor_pid = monitor_pid;
235  monitor->ref_ticks = ref_ticks;
236  monitor->linked = linked;
237  list_append(&ctx->monitors_head, &monitor->monitor_list_head);
238 
239  return ref_ticks;
240 }

242 void context_demonitor(Context *ctx, term monitor_pid, bool linked)
243 {
244  struct ListHead *item;
245  LIST_FOR_EACH (item, &ctx->monitors_head) {
246  struct Monitor *monitor = GET_LIST_ENTRY(item, struct Monitor, monitor_list_head);
247  if ((monitor->monitor_pid == monitor_pid) && (monitor->linked == linked)) {
248  list_remove(&monitor->monitor_list_head);
249  free(monitor);
250  return;
251  }
252  }
