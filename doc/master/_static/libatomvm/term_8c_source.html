112 int term_funprint(PrinterFun *fun, term t, const GlobalContext *global)
113 {
114  if (term_is_atom(t)) {
115  int atom_index = term_to_atom_index(t);
116  AtomString atom_string = (AtomString) valueshashtable_get_value(
117  global->atoms_ids_table, atom_index, (unsigned long) NULL);
118  return fun->print(fun, "%.*s", (int) atom_string_len(atom_string),
119  (char *) atom_string_data(atom_string));
120 
121  } else if (term_is_integer(t)) {
122  avm_int_t iv = term_to_int(t);
123  return fun->print(fun, AVM_INT_FMT, iv);
124 
125  } else if (term_is_nil(t)) {
126  return fun->print(fun, "[]");
127 
128  } else if (term_is_nonempty_list(t)) {
129  int is_printable = 1;
130  term list_item = t;
131  while (term_is_nonempty_list(list_item)) {
132  term head = term_get_list_head(list_item);
133  is_printable = is_printable && term_is_uint8(head) && isprint(term_to_uint8(head));
134  list_item = term_get_list_tail(list_item);
135  }
136  // improper lists are not printable
137  if (!term_is_nil(list_item)) {
138  is_printable = 0;
139  }
140 
141  if (is_printable) {
142  int ok;
143  char *printable = interop_list_to_string(t, &ok);
144  if (LIKELY(ok)) {
145  int ret = fun->print(fun, "\"%s\"", printable);
146  free(printable);
147  return ret;
148  } else {
149  return fun->print(fun, "???");
150  }
151 
152  } else {
153  int ret = fun->print(fun, "[");
154  if (UNLIKELY(ret < 0)) {
155  return ret;
156  }
157  int display_separator = 0;
158  while (term_is_nonempty_list(t)) {
159  if (display_separator) {
160  ret += fun->print(fun, ",");
161  } else {
162  display_separator = 1;
163  }
164 
165  int printed = term_funprint(fun, term_get_list_head(t), global);
166  if (UNLIKELY(printed < 0)) {
167  return printed;
168  }
169  ret += printed;
170  t = term_get_list_tail(t);
171  }
172  if (!term_is_nil(t)) {
173  int printed = fun->print(fun, "|");
174  if (UNLIKELY(printed < 0)) {
175  return printed;
176  }
177  ret += printed;
178 
179  printed = term_funprint(fun, t, global);
180  if (UNLIKELY(printed < 0)) {
181  return printed;
182  }
183  ret += printed;
184  }
185  int printed = fun->print(fun, "]");
186  if (UNLIKELY(printed < 0)) {
187  return printed;
188  }
189  ret += printed;
190  return ret;
191  }
192  } else if (term_is_pid(t)) {
193  return fun->print(fun, "<0.%" PRIu32 ".0>", term_to_local_process_id(t));
194 
195  } else if (term_is_function(t)) {
196  const term *boxed_value = term_to_const_term_ptr(t);
197  Module *fun_module = (Module *) boxed_value[1];
198  uint32_t fun_index = boxed_value[2];
199  const char *format =
200  #ifdef __clang__
201  "#Fun<erl_eval.%lu.%llu>";
202  #else
203  "#Fun<erl_eval.%lu.%llu>";
204  #endif
205  return fun->print(fun, format, fun_index, (unsigned long) fun_module);
206 
207  } else if (term_is_tuple(t)) {
208  int ret = fun->print(fun, "{");
209  if (UNLIKELY(ret < 0)) {
210  return ret;
211  }
212 
213  int tuple_size = term_get_tuple_arity(t);
214  for (int i = 0; i < tuple_size; i++) {
215  if (i != 0) {
216  int printed = fun->print(fun, ",");
217  if (UNLIKELY(printed < 0)) {
218  return printed;
219  }
220  ret += printed;
221  }
222  int printed = term_funprint(fun, term_get_tuple_element(t, i), global);
223  if (UNLIKELY(printed < 0)) {
224  return printed;
225  }
226  ret += printed;
227  }
228 
229  int printed = fun->print(fun, "}");
230  if (UNLIKELY(printed < 0)) {
231  return printed;
232  }
233  ret += printed;
234  return ret;
235 
236  } else if (term_is_map(t)) {
237  int ret = fun->print(fun, "#{");
238  if (UNLIKELY(ret < 0)) {
239  return ret;
240  }
241 
242  int map_size = term_get_map_size(t);
243  for (int i = 0; i < map_size; i++) {
244  if (i != 0) {
245  int printed = fun->print(fun, ",");
246  if (UNLIKELY(printed < 0)) {
247  return printed;
248  }
249  ret += printed;
250  }
251  int printed = term_funprint(fun, term_get_map_key(t, i), global);
252  if (UNLIKELY(printed < 0)) {
253  return printed;
254  }
255  ret += printed;
256 
257  printed = fun->print(fun, "=>");
258  if (UNLIKELY(printed < 0)) {
259  return printed;
260  }
261  ret += printed;
262 
263  printed = term_funprint(fun, term_get_map_value(t, i), global);
264  if (UNLIKELY(printed < 0)) {
265  return printed;
266  }
267  ret += printed;
268  }
269 
270  int printed = fun->print(fun, "}");
271  if (UNLIKELY(printed < 0)) {
272  return printed;
273  }
274  ret += printed;
275  return ret;
276 
277  } else if (term_is_binary(t)) {
278  int len = term_binary_size(t);
279  const unsigned char *binary_data = (const unsigned char *) term_binary_data(t);
280 
281  int is_printable = 1;
282  for (int i = 0; i < len; i++) {
283  if (!isprint(binary_data[i])) {
284  is_printable = 0;
285  break;
286  }
287  }
288 
289  int ret = fun->print(fun, "<<");
290  if (UNLIKELY(ret < 0)) {
291  return ret;
292  }
293 
294  if (is_printable) {
295  int printed = fun->print(fun, "\"%.*s\"", len, binary_data);
296  if (UNLIKELY(printed < 0)) {
297  return printed;
298  }
299  ret += printed;
300 
301  } else {
302  int display_separator = 0;
303  for (int i = 0; i < len; i++) {
304  if (display_separator) {
305  int printed = fun->print(fun, ",");
306  if (UNLIKELY(printed < 0)) {
307  return printed;
308  }
309  ret += printed;
310  } else {
311  display_separator = 1;
312  }
313 
314  uint8_t c = (uint8_t) binary_data[i];
315  int printed = fun->print(fun, "%i", (int) c);
316  if (UNLIKELY(printed < 0)) {
317  return printed;
318  }
319  ret += printed;
320  }
321  }
322  int printed = fun->print(fun, ">>");
323  if (UNLIKELY(printed < 0)) {
324  return printed;
325  }
326  ret += printed;
327  return ret;
328 
329  } else if (term_is_reference(t)) {
330  const char *format =
331 #ifdef __clang__
332  "#Ref<0.0.0.%llu>";
333 #else
334  "#Ref<0.0.0.%lu>";
335 #endif
336  return fun->print(fun, format, term_to_ref_ticks(t));
337 
338  } else if (term_is_boxed_integer(t)) {
339  int size = term_boxed_size(t);
340  switch (size) {
341  case 1:
342  return fun->print(fun, AVM_INT_FMT, term_unbox_int(t));
343 
344 #if BOXED_TERMS_REQUIRED_FOR_INT64 == 2
345  case 2:
346  return fun->print(fun, AVM_INT64_FMT, term_unbox_int64(t));
347 #endif
348  default:
349  AVM_ABORT();
350  }
351 
352  } else if (term_is_float(t)) {
353  avm_float_t f = term_to_float(t);
354  return fun->print(fun, AVM_FLOAT_FMT, f);
355 
356  } else {
357  return fun->print(fun, "Unknown term type: %" TERM_U_FMT, t);
358  }
359 }

418 TermCompareResult term_compare(term t, term other, TermCompareOpts opts, GlobalContext *global)
419 {
420  struct TempStack temp_stack;
421  if (UNLIKELY(temp_stack_init(&temp_stack) != TempStackOk)) {
422  return TermCompareMemoryAllocFail;
423  }
424 
425  if (UNLIKELY(temp_stack_push(&temp_stack, t) != TempStackOk)) {
426  return TermCompareMemoryAllocFail;
427  }
428  if (UNLIKELY(temp_stack_push(&temp_stack, other) != TempStackOk)) {
429  return TermCompareMemoryAllocFail;
430  }
431 
432  TermCompareResult result = TermEquals;
433  int map_key_nesting = 0;
434 
435  while (!temp_stack_is_empty(&temp_stack)) {
436  if (t == other) {
437  CMP_POP_AND_CONTINUE();
438 
439  } else if (term_is_integer(t) && term_is_integer(other)) {
440  avm_int_t t_int = term_to_int(t);
441  avm_int_t other_int = term_to_int(other);
442  //They cannot be equal
443  result = (t_int > other_int) ? TermGreaterThan : TermLessThan;
444  break;
445 
446  } else if (term_is_reference(t) && term_is_reference(other)) {
447  int64_t t_ticks = term_to_ref_ticks(t);
448  int64_t other_ticks = term_to_ref_ticks(other);
449  if (t_ticks == other_ticks) {
450  CMP_POP_AND_CONTINUE();
451  } else {
452  result = (t_ticks > other_ticks) ? TermGreaterThan : TermLessThan;
453  break;
454  }
455 
456  } else if (term_is_nonempty_list(t) && term_is_nonempty_list(other)) {
457  term t_tail = term_get_list_tail(t);
458  term other_tail = term_get_list_tail(other);
459  // invalid term is used as a term lower than any other
460  // so "a" < "ab" -> true can be implemented.
461  if (term_is_nil(t_tail)) {
462  t_tail = term_invalid_term();
463  }
464  if (term_is_nil(other_tail)) {
465  other_tail = term_invalid_term();
466  }
467  if (UNLIKELY(temp_stack_push(&temp_stack, t_tail) != TempStackOk)) {
468  return TermCompareMemoryAllocFail;
469  }
470  if (UNLIKELY(temp_stack_push(&temp_stack, other_tail) != TempStackOk)) {
471  return TermCompareMemoryAllocFail;
472  }
473  t = term_get_list_head(t);
474  other = term_get_list_head(other);
475 
476  } else if (term_is_tuple(t) && term_is_tuple(other)) {
477  int tuple_size = term_get_tuple_arity(t);
478  int other_tuple_size = term_get_tuple_arity(other);
479 
480  if (tuple_size != other_tuple_size) {
481  result = (tuple_size > other_tuple_size) ? TermGreaterThan : TermLessThan;
482  break;
483  }
484 
485  if (tuple_size > 0) {
486  for (int i = tuple_size - 1; i >= 1; i--) {
487  if (UNLIKELY(temp_stack_push(&temp_stack, term_get_tuple_element(t, i))
488  != TempStackOk)) {
489  return TermCompareMemoryAllocFail;
490  }
491  if (UNLIKELY(temp_stack_push(&temp_stack, term_get_tuple_element(other, i))
492  != TempStackOk)) {
493  return TermCompareMemoryAllocFail;
494  }
495  }
496  t = term_get_tuple_element(t, 0);
497  other = term_get_tuple_element(other, 0);
498 
499  } else {
500  CMP_POP_AND_CONTINUE();
501  }
502 
503  } else if (term_is_binary(t) && term_is_binary(other)) {
504  int t_size = term_binary_size(t);
505  int other_size = term_binary_size(other);
506 
507  const char *t_data = term_binary_data(t);
508  const char *other_data = term_binary_data(other);
509 
510  int cmp_size = (t_size > other_size) ? other_size : t_size;
511 
512  int memcmp_result = memcmp(t_data, other_data, cmp_size);
513  if (memcmp_result == 0) {
514  if (t_size == other_size) {
515  CMP_POP_AND_CONTINUE();
516  } else {
517  result = (t_size > other_size) ? TermGreaterThan : TermLessThan;
518  break;
519  }
520  } else {
521  result = (memcmp_result > 0) ? TermGreaterThan : TermLessThan;
522  break;
523  }
524 
525  } else if (term_is_map(t) && term_is_map(other)) {
526  int t_size = term_get_map_size(t);
527  int other_size = term_get_map_size(other);
528 
529  if (t_size != other_size) {
530  result = (t_size > other_size) ? TermGreaterThan : TermLessThan;
531  break;
532  }
533  if (t_size > 0) {
534  for (int i = t_size - 1; i >= 1; i--) {
535  if (UNLIKELY(temp_stack_push(&temp_stack, term_get_map_value(t, i))
536  != TempStackOk)) {
537  return TermCompareMemoryAllocFail;
538  }
539  if (UNLIKELY(temp_stack_push(&temp_stack, term_get_map_value(other, i))
540  != TempStackOk)) {
541  return TermCompareMemoryAllocFail;
542  }
543  if (UNLIKELY(
544  temp_stack_push(&temp_stack, END_MAP_KEY) != TempStackOk)) {
545  return TermCompareMemoryAllocFail;
546  }
547  if (UNLIKELY(
548  temp_stack_push(&temp_stack, term_get_map_key(t, i)) != TempStackOk)) {
549  return TermCompareMemoryAllocFail;
550  }
551  if (UNLIKELY(temp_stack_push(&temp_stack, term_get_map_key(other, i))
552  != TempStackOk)) {
553  return TermCompareMemoryAllocFail;
554  }
555  if (UNLIKELY(
556  temp_stack_push(&temp_stack, BEGIN_MAP_KEY) != TempStackOk)) {
557  return TermCompareMemoryAllocFail;
558  }
559  }
560  if (UNLIKELY(temp_stack_push(&temp_stack, term_get_map_value(t, 0)) != TempStackOk)) {
561  return TermCompareMemoryAllocFail;
562  }
563  if (UNLIKELY(temp_stack_push(&temp_stack, term_get_map_value(other, 0)) != TempStackOk)) {
564  return TermCompareMemoryAllocFail;
565  }
566  map_key_nesting++;
567  if (UNLIKELY(temp_stack_push(&temp_stack, END_MAP_KEY) != TempStackOk)) {
568  return TermCompareMemoryAllocFail;
569  }
570  t = term_get_map_key(t, 0);
571  other = term_get_map_key(other, 0);
572 
573  } else {
574  CMP_POP_AND_CONTINUE();
575  }
576 
577  } else if (term_is_any_integer(t) && term_is_any_integer(other)) {
578  avm_int64_t t_int = term_maybe_unbox_int64(t);
579  avm_int64_t other_int = term_maybe_unbox_int64(other);
580  if (t_int == other_int) {
581  CMP_POP_AND_CONTINUE();
582  } else {
583  result = (t_int > other_int) ? TermGreaterThan : TermLessThan;
584  break;
585  }
586 
587  } else if (term_is_float(t) && term_is_float(other)) {
588  avm_float_t t_float = term_to_float(t);
589  avm_float_t other_float = term_to_float(other);
590  if (t_float == other_float) {
591  CMP_POP_AND_CONTINUE();
592  } else {
593  result = (t_float > other_float) ? TermGreaterThan : TermLessThan;
594  break;
595  }
596 
597  } else if (term_is_number(t) && term_is_number(other)
598  && ((opts & TermCompareExact) != TermCompareExact) && (map_key_nesting == 0)) {
599  avm_float_t t_float = term_conv_to_float(t);
600  avm_float_t other_float = term_conv_to_float(other);
601  if (t_float == other_float) {
602  CMP_POP_AND_CONTINUE();
603  } else {
604  result = (t_float > other_float) ? TermGreaterThan : TermLessThan;
605  break;
606  }
607 
608  } else if (term_is_atom(t) && term_is_atom(other)) {
609  int t_atom_index = term_to_atom_index(t);
610  AtomString t_atom_string = (AtomString) valueshashtable_get_value(global->atoms_ids_table,
611  t_atom_index, (unsigned long) NULL);
612 
613  int t_atom_len = atom_string_len(t_atom_string);
614  const char *t_atom_data = (const char *) atom_string_data(t_atom_string);
615 
616  int other_atom_index = term_to_atom_index(other);
617  AtomString other_atom_string = (AtomString) valueshashtable_get_value(global->atoms_ids_table,
618  other_atom_index, (unsigned long) NULL);
619 
620  int other_atom_len = atom_string_len(other_atom_string);
621  const char *other_atom_data = (const char *) atom_string_data(other_atom_string);
622 
623  int cmp_size = (t_atom_len > other_atom_len) ? other_atom_len : t_atom_len;
624 
625  int memcmp_result = memcmp(t_atom_data, other_atom_data, cmp_size);
626  if (memcmp_result == 0) {
627  result = (t_atom_len > other_atom_len) ? TermGreaterThan : TermLessThan;
628  break;
629  } else {
630  result = memcmp_result > 0 ? TermGreaterThan : TermLessThan;
631  break;
632  }
633 
634  } else if (term_is_pid(t) && term_is_pid(other)) {
635  //TODO: handle ports
636  result = (t > other) ? TermGreaterThan : TermLessThan;
637  break;
638 
639  } else {
640  result = (term_type_to_index(t) > term_type_to_index(other)) ? TermGreaterThan : TermLessThan;
641  break;
642  }
643  }
644 
645  temp_stack_destroy(&temp_stack);
646 
647  return result;
648 }

650 term term_alloc_refc_binary(size_t size, bool is_const, Heap *heap, GlobalContext *glb)
651 {
652  term *boxed_value = memory_heap_alloc(heap, TERM_BOXED_REFC_BINARY_SIZE);
653  boxed_value[0] = ((TERM_BOXED_REFC_BINARY_SIZE - 1) << 6) | TERM_BOXED_REFC_BINARY;
654  boxed_value[1] = (term) size;
655  boxed_value[2] = (term) is_const ? RefcBinaryIsConst : RefcNoFlags;
656  term ret = ((term) boxed_value) | TERM_BOXED_VALUE_TAG;
657  if (is_const) {
658  boxed_value[3] = (term) NULL;
659  // TODO Consider making const refc binaries 4 words instead of 6
660  boxed_value[4] = term_nil(); // mso_list is not used
661  boxed_value[5] = term_nil(); // for const binaries
662  } else {
