288 static inline const struct ExportedFunction *module_resolve_function(Module *mod, int import_table_index, GlobalContext *glb)
289 {
290  SMP_MODULE_LOCK(mod);
291  // We cannot optimistically read the unresolved function call.
292  // While reads of imported_funcs can happen outside the lock, read of the func
293  // pointer itself cannot, as UnresolvedFunctionCall pointers are freed.
294  const struct ExportedFunction *func = mod->imported_funcs[import_table_index];
295  if (func->type != UnresolvedFunctionCall) {
296  SMP_MODULE_UNLOCK(mod);
297  return func;
298  }
299  struct UnresolvedFunctionCall *unresolved = EXPORTED_FUNCTION_TO_UNRESOLVED_FUNCTION_CALL(func);
300  const struct ExportedFunction *result = module_resolve_function0(mod, import_table_index, unresolved, glb);
301  SMP_MODULE_UNLOCK(mod);
302  return result;
303 }
