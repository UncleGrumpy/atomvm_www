131 static inline bool bitstring_extract_integer(term src_bin, size_t offset, avm_int_t n,
132  enum BitstringFlags bs_flags, union maybe_unsigned_int64 *dst)
133 {
134  unsigned long capacity = term_binary_size(src_bin);
135  if (8 * capacity - offset < (unsigned long) n) {
136  return false;
137  }
138 
139  if ((offset & 0x7) == 0) {
140  int byte_offset = offset >> 3;
141  const uint8_t *src = (const uint8_t *) term_binary_data(src_bin) + byte_offset;
142 
143  switch (n) {
144  case 8: {
145  union maybe_unsigned_int8 i8;
146 
147  i8.u = *src;
148  if (bs_flags & SignedInteger) {
149  dst->s = i8.s;
150  } else {
151  dst->u = i8.u;
152  }
153  return true;
154  }
155 
156  case 16: {
157  union maybe_unsigned_int16 i16;
158 
159  if (bs_flags & LittleEndianIntegerMask) {
160  i16.u = READ_16LE_UNALIGNED(src);
161  } else {
162  i16.u = READ_16_UNALIGNED(src);
163  }
164  if (bs_flags & SignedInteger) {
165  dst->s = i16.s;
166  } else {
167  dst->u = i16.u;
168  }
169  return true;
170  }
171 
172  case 32: {
173  union maybe_unsigned_int32 i32;
174  if (bs_flags & LittleEndianIntegerMask) {
175  i32.u = READ_32LE_UNALIGNED(src);
176  } else {
177  i32.u = READ_32_UNALIGNED(src);
178  }
179  if (bs_flags & SignedInteger) {
180  dst->s = i32.s;
181  } else {
182  dst->u = i32.u;
183  }
184  return true;
185  }
186 
187  case 64: {
188  union maybe_unsigned_int64 i64;
189 
190  if (bs_flags & LittleEndianIntegerMask) {
191  i64.u = READ_64LE_UNALIGNED(src);
192  } else {
193  i64.u = READ_64_UNALIGNED(src);
194  }
195  if (bs_flags & SignedInteger) {
196  if (UNLIKELY(i64.u & ((uint64_t) 1 << 63))) {
197  return false;
198  }
199  dst->s = i64.s;
200  } else {
201  dst->u = i64.u;
202  }
203  return true;
204  }
205 
206  default:
207  return bitstring_extract_any_integer(src, 0, n, bs_flags, dst);
208  }
209  } else {
210  int byte_offset = offset / 8;
211  const uint8_t *src = (const uint8_t *) term_binary_data(src_bin) + byte_offset;
212  return bitstring_extract_any_integer(src, offset - (byte_offset * 8), n, bs_flags, dst);
213  }
214 }

216 static inline bool bitstring_insert_integer(term dst_bin, size_t offset, avm_int64_t value, size_t n, enum BitstringFlags bs_flags)
217 {
218  unsigned long capacity = term_binary_size(dst_bin);
219  if (8 * capacity - offset < (unsigned long) n) {
220  return false;
221  }
222 
223  if ((offset & 0x7) == 0) {
224  int byte_offset = offset >> 3;
225  uint8_t *dst = (uint8_t *) term_binary_data(dst_bin) + byte_offset;
226 
227  switch (n) {
228  case 8: {
229  *dst = value & 0xFF;
230  return true;
231  }
232 
233  case 16: {
234  if (bs_flags & LittleEndianIntegerMask) {
235  if (bs_flags & SignedInteger) {
236  int16_t signed_value = (int16_t) value;
237  WRITE_16LE_UNALIGNED(int16_t, dst, signed_value);
238  } else {
239  uint16_t unsigned_value = (uint16_t) value;
240  WRITE_16LE_UNALIGNED(uint16_t, dst, unsigned_value);
241  }
242  } else {
243  if (bs_flags & SignedInteger) {
244  int16_t signed_value = (int16_t) value;
245  WRITE_16_UNALIGNED(dst, signed_value);
246  } else {
247  uint16_t unsigned_value = (uint16_t) value;
248  WRITE_16_UNALIGNED(dst, unsigned_value);
249  }
250  }
251  return true;
252  }
253 
254  case 32: {
255  if (bs_flags & LittleEndianIntegerMask) {
256  if (bs_flags & SignedInteger) {
257  int32_t signed_value = (int32_t) value;
258  WRITE_32LE_UNALIGNED(int32_t, dst, signed_value);
259  } else {
260  uint32_t unsigned_value = (uint32_t) value;
261  WRITE_32LE_UNALIGNED(uint32_t, dst, unsigned_value);
262  }
263  } else {
264  if (bs_flags & SignedInteger) {
265  int32_t signed_value = (int32_t) value;
266  WRITE_32_UNALIGNED(dst, signed_value);
267  } else {
268  uint32_t unsigned_value = (uint32_t) value;
269  WRITE_32_UNALIGNED(dst, unsigned_value);
270  }
271  }
272  return true;
273  }
274 
275  case 64: {
276  if (bs_flags & LittleEndianIntegerMask) {
277  if (bs_flags & SignedInteger) {
278  avm_int64_t signed_value = value;
279  WRITE_64LE_UNALIGNED(avm_int64_t, dst, signed_value);
280  } else {
281  uint64_t unsigned_value = (uint64_t) value;
282  WRITE_64LE_UNALIGNED(avm_int64_t, dst, unsigned_value);
283  }
284  } else {
285  if (bs_flags & SignedInteger) {
286  int64_t signed_value = value;
287  WRITE_64_UNALIGNED(dst, signed_value);
288  } else {
289  uint64_t unsigned_value = (uint64_t) value;
290  WRITE_64_UNALIGNED(dst, unsigned_value);
291  }
292  }
293  return true;
294  }
295 
296  default:
297  return bitstring_insert_any_integer(dst, offset - (byte_offset * 8), value, n, bs_flags);
298  }
299  }
300 
301  return bitstring_insert_any_integer((uint8_t *) term_binary_data(dst_bin), offset, value, n, bs_flags);
