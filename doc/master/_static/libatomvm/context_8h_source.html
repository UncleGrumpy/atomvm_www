243 static inline term *context_stack_base(const Context *ctx)
244 {
245  // Find which fragment the stack belongs to.
246  if (ctx->e >= ctx->heap.heap_start && ctx->e <= ctx->heap.heap_end) {
247  return ctx->heap.heap_end;
248  }
249  HeapFragment *fragment = ctx->heap.root->next;
250  while (fragment) {
251  if (ctx->e >= fragment->storage && ctx->e <= fragment->heap_end) {
252  return fragment->heap_end;
253  }
254  fragment = fragment->next;
255  }
256  fprintf(stderr, "Context heap is corrupted, cannot find fragment with stack pointer.\n");
257  AVM_ABORT();
258 }
