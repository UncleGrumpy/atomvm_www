66 static term eai_errno_to_term(int err, GlobalContext *glb)
67 {
68  switch (err) {
69  case EAI_AGAIN:
70  return globalcontext_make_atom(glb, ATOM_STR("\x8", "eaiagain"));
71  case EAI_BADFLAGS:
72  return globalcontext_make_atom(glb, ATOM_STR("\xB", "eaibadflags"));
73  case EAI_FAIL:
74  return globalcontext_make_atom(glb, ATOM_STR("\x7", "eaifail"));
75  case EAI_FAMILY:
76  return globalcontext_make_atom(glb, ATOM_STR("\x9", "eaifamily"));
77  case EAI_MEMORY:
78  return globalcontext_make_atom(glb, ATOM_STR("\x9", "eaimemory"));
79  case EAI_NONAME:
80  return globalcontext_make_atom(glb, ATOM_STR("\x9", "eainoname"));
81  case EAI_SERVICE:
82  return globalcontext_make_atom(glb, ATOM_STR("\xA", "eaiservice"));
83  case EAI_SOCKTYPE:
84  return globalcontext_make_atom(glb, ATOM_STR("\xB", "eaisocktype"));
85 #ifdef HAVE_EXTENDED_EAI_ERRNO
86  case EAI_BADHINTS:
87  return globalcontext_make_atom(glb, ATOM_STR("\xB", "eaibadhints"));
88  case EAI_OVERFLOW:
89  return globalcontext_make_atom(glb, ATOM_STR("\xB", "eaioverflow"));
90  case EAI_PROTOCOL:
91  return globalcontext_make_atom(glb, ATOM_STR("\xB", "eaiprotocol"));
92  case EAI_SYSTEM:
93  return globalcontext_make_atom(glb, ATOM_STR("\x9", "eaisystem"));
94 #endif
95  }
96  return term_from_int(err);
97 }

115 static term make_getaddrinfo_result(term *keys, int ai_protocol, int ai_socktype, term inner_addr, GlobalContext *global, Heap *heap)
116 {
117  term result_map;
118  if (term_is_invalid_term(*keys)) {
119  result_map = term_alloc_map(5, heap);
120  } else {
121  result_map = term_alloc_map_maybe_shared(5, *keys, heap);
122  }
123 
124  // in the current implementation, this will always be `inet`
125  term family_atom = globalcontext_make_atom(global, ATOM_STR("\x6", "family"));
126  term family = globalcontext_make_atom(global, ATOM_STR("\x4", "inet"));
127  term_set_map_assoc(result_map, 0, family_atom, family);
128 
129  term protocol_atom = globalcontext_make_atom(global, ATOM_STR("\x8", "protocol"));
130  term protocol = interop_atom_term_select_atom(protocol_table, ai_protocol, global);
131  term_set_map_assoc(result_map, 1, protocol_atom, term_is_invalid_term(protocol) ? UNDEFINED_ATOM : protocol);
132 
133  term type_atom = globalcontext_make_atom(global, ATOM_STR("\x4", "type"));
134  term type = interop_atom_term_select_atom(type_table, ai_socktype, global);
135  term_set_map_assoc(result_map, 2, type_atom, term_is_invalid_term(type) ? UNDEFINED_ATOM : type);
136 
137  // embed the inner_addr, but reference it from both address and addr
138  // for compatibility with OTP
139  term address_atom = globalcontext_make_atom(global, ATOM_STR("\x7", "address"));
140  term_set_map_assoc(result_map, 3, address_atom, inner_addr);
141 
142  term addr_atom = globalcontext_make_atom(global, ATOM_STR("\x4", "addr"));
143  term_set_map_assoc(result_map, 4, addr_atom, inner_addr);
144 
145  if (term_is_invalid_term(*keys)) {
146  *keys = term_get_map_keys(result_map);
147  }
148 
149  return result_map;
150 }

156 static term nif_net_getaddrinfo(Context *ctx, int argc, term argv[])
157 {
158  TRACE("nif_net_getaddrinfo\n");
159  UNUSED(argc);
160 
161  GlobalContext *global = ctx->global;
162 
163  term host = argv[0];
164  term service = argv[1];
165 
166  if (host == UNDEFINED_ATOM && service == UNDEFINED_ATOM) {
167  TRACE("Host and Service params may not both be undefined");
168  RAISE_ERROR(BADARG_ATOM);
169  }
170 
171  char *host_str = NULL;
172  if (host != UNDEFINED_ATOM) {
173  int ok;
174  host_str = interop_term_to_string(host, &ok);
175  if (!ok) {
176  RAISE_ERROR(BADARG_ATOM);
177  }
178  TRACE("Host: %s\n", host_str);
179  }
180 
181  char *service_str = NULL;
182  if (service != UNDEFINED_ATOM) {
183  int ok;
184  service_str = interop_term_to_string(service, &ok);
185  if (!ok) {
186  free(host_str);
187  RAISE_ERROR(BADARG_ATOM);
188  }
189  TRACE("Service: %s\n", service_str);
190  }
191 
192  avm_uint_t port = 0;
193 #ifdef HAVE_SERVBYNAME
194  if (!IS_NULL_PTR(service_str)) {
195  struct servent *svt = getservbyname(service_str, NULL);
196  if (!IS_NULL_PTR(svt)) {
197  port = ntohs(svt->s_port);
198  }
199  }
200 #endif
201  TRACE("port: %zu\n", port);
202 
203  // for now, we are only supporting IPv4 addresses
204  struct addrinfo hints;
205  memset(&hints, 0, sizeof(hints));
206  hints.ai_family = PF_INET;
207 
208  struct addrinfo *host_info;
209  int err = getaddrinfo(host_str, service_str, &hints, &host_info);
210 
211  // some implementations do not support service filters
212  if (err == EAI_SERVICE) {
213  fprintf(stderr, "WARNING: EAI_SERVICE unsupported on this platform.\n");
214  err = getaddrinfo(host_str, NULL, &hints, &host_info);
215  }
216 
217  free(host_str);
218  free(service_str);
219 
220  if (err != 0 && err != EAI_SERVICE) {
221  if (UNLIKELY(memory_ensure_free(ctx, TUPLE_SIZE(2)) != MEMORY_GC_OK)) {
222  RAISE_ERROR(OUT_OF_MEMORY_ATOM);
223  }
224  return make_error_tuple(eai_errno_to_term(err, global), ctx);
225  }
226  TRACE("getaddrinfo succeeded\n");
227 
228  size_t num_addrinfos = 0;
229  for (struct addrinfo *p = host_info; p != NULL; p = p->ai_next) {
230  num_addrinfos++;
231  }
232  TRACE("num_addrinfos: %zu\n", num_addrinfos);
233 
234  if (num_addrinfos == 0) {
235  return term_nil();
236  }
237 
238  // {ok, [#{
239  // family => Family :: atom()
240  // protocol => Protocol :: atom()
241  // type -> Type :: atom()
242  // address, addr =>
243  // #{
244  // addr => Address :: {0..255, 0..255, 0..255, 0..255},
245  // port => 0..65535,
246  // family => inet
247  // }
248  // }]}
249  // Note. We might over-allocate for some more esoteric calls
250 
251  // Determine the number of entries, if we have ai_protocol or ai_socktype as unspec, return two
252  size_t nb_results = 0;
253  size_t requested_size = TUPLE_SIZE(2); // {ok, _}
254  for (struct addrinfo *p = host_info; p != NULL; p = p->ai_next) {
255  // Each list item is:
256  // 1 CONS
257  // 1 IPv4 address
258  // 1 map with 5 items (family, protocol, type, address, addr)
259  // 1 map with 3 items (addr, port, family)
260  requested_size += CONS_SIZE + INET_ADDR4_TUPLE_SIZE;
261  // First result: regular maps
262  // Subsequent results: shared maps
263  if (nb_results) {
264  requested_size += TERM_MAP_SHARED_SIZE(5) + TERM_MAP_SHARED_SIZE(3);
265  } else {
266  requested_size += TERM_MAP_SIZE(5) + TERM_MAP_SIZE(3);
267  }
268  nb_results++;
269  // If protocol or socktype are unspecified (what esp-idf returns), add
270  // another entry so we'll have tcp and udp
271  if (p->ai_protocol == 0 || p->ai_socktype == 0) {
272  nb_results++;
273  // We only need cons and shared maps here as the IP address will be shared
274  requested_size += CONS_SIZE + TERM_MAP_SHARED_SIZE(5) + TERM_MAP_SHARED_SIZE(3);
275  }
276  }
277  if (UNLIKELY(memory_ensure_free(ctx, requested_size) != MEMORY_GC_OK)) {
278  RAISE_ERROR(OUT_OF_MEMORY_ATOM);
279  }
280 
281  term infos = term_nil();
282  term result_keys = term_invalid_term();
283  term addrinfo_keys = term_invalid_term();
284  term family_atom = globalcontext_make_atom(global, ATOM_STR("\x6", "family"));
285  term inet_atom = globalcontext_make_atom(global, ATOM_STR("\x4", "inet"));
286  for (struct addrinfo *p = host_info; p != NULL; p = p->ai_next) {
287  term addrinfo_map;
288  if (term_is_invalid_term(addrinfo_keys)) {
289  addrinfo_map = term_alloc_map(3, &ctx->heap);
290  } else {
291  addrinfo_map = term_alloc_map_maybe_shared(3, addrinfo_keys, &ctx->heap);
292  }
293  // The inner addr contains a family, port, and addr
294  term addr_atom = globalcontext_make_atom(global, ATOM_STR("\x4", "addr"));
295  term_set_map_assoc(addrinfo_map, 0, family_atom, inet_atom);
296  term_set_map_assoc(addrinfo_map, 1, PORT_ATOM, term_from_int(port));
297  term address = inet_make_addr4(ntohl(((struct sockaddr_in *) p->ai_addr)->sin_addr.s_addr), &ctx->heap);
298  term_set_map_assoc(addrinfo_map, 2, addr_atom, address);
299 
300  if (term_is_invalid_term(addrinfo_keys)) {
301  addrinfo_keys = term_get_map_keys(addrinfo_map);
302  }
303 
304  if (p->ai_protocol != 0 && p->ai_socktype != 0) {
305  term result_map = make_getaddrinfo_result(&result_keys, p->ai_protocol, p->ai_socktype, addrinfo_map, ctx->global, &ctx->heap);
306  infos = term_list_prepend(result_map, infos, &ctx->heap);
307  } else {
308  term tcp_map = make_getaddrinfo_result(&result_keys, IPPROTO_TCP, SOCK_STREAM, addrinfo_map, ctx->global, &ctx->heap);
309  infos = term_list_prepend(tcp_map, infos, &ctx->heap);
310  term udp_map = make_getaddrinfo_result(&result_keys, IPPROTO_UDP, SOCK_DGRAM, addrinfo_map, ctx->global, &ctx->heap);
311  infos = term_list_prepend(udp_map, infos, &ctx->heap);
312  }
313  }
314  freeaddrinfo(host_info);
315 
316  term ret = term_alloc_tuple(2, &ctx->heap);
317  term_put_tuple_element(ret, 0, OK_ATOM);
318  term_put_tuple_element(ret, 1, infos);
319 
320 #ifdef ENABLE_TRACE
321  fprintf(stdout, "host info: ");
322  term_display(stdout, ret, ctx);
323  fprintf(stdout, "\n");
324 #endif
325 
326  return ret;
327 }
