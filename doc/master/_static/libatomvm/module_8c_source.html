72 static enum ModuleLoadResult module_populate_atoms_table(Module *this_module, uint8_t *table_data, GlobalContext *glb)
73 {
74  int atoms_count = READ_32_ALIGNED(table_data + 8);
75  const char *current_atom = (const char *) table_data + 12;
76 
77  this_module->local_atoms_to_global_table = calloc(atoms_count + 1, sizeof(int));
78  if (IS_NULL_PTR(this_module->local_atoms_to_global_table)) {
79  fprintf(stderr, "Cannot allocate memory while loading module (line: %i).\n", __LINE__);
80  return MODULE_ERROR_FAILED_ALLOCATION;
81  }
82 
83  const char *atom = NULL;
84  for (int i = 1; i <= atoms_count; i++) {
85  // atom 0 is NON TERM for historical reasons
86  int atom_len = *current_atom;
87  atom = current_atom;
88 
89  int global_atom_id = globalcontext_insert_atom(glb, (AtomString) atom);
90  if (UNLIKELY(global_atom_id < 0)) {
91  fprintf(stderr, "Cannot allocate memory while loading module (line: %i).\n", __LINE__);
92  return MODULE_ERROR_FAILED_ALLOCATION;
93  }
94 
95  this_module->local_atoms_to_global_table[i] = global_atom_id;
96 
97  current_atom += atom_len + 1;
98  }
99 
100  return MODULE_LOAD_OK;
101 }

103 static enum ModuleLoadResult module_build_imported_functions_table(Module *this_module, uint8_t *table_data, GlobalContext *glb)
104 {
105  int functions_count = READ_32_ALIGNED(table_data + 8);
106 
107  this_module->imported_funcs = calloc(functions_count, sizeof(struct ExportedFunction *));
108  if (IS_NULL_PTR(this_module->imported_funcs)) {
109  fprintf(stderr, "Cannot allocate memory while loading module (line: %i).\n", __LINE__);
110  return MODULE_ERROR_FAILED_ALLOCATION;
111  }
112 
113  for (int i = 0; i < functions_count; i++) {
114  int local_module_atom_index = READ_32_ALIGNED(table_data + i * 12 + 12);
115  int local_function_atom_index = READ_32_ALIGNED(table_data + i * 12 + 4 + 12);
116  AtomString module_atom = module_get_atom_string_by_id(this_module, local_module_atom_index, glb);
117  AtomString function_atom = module_get_atom_string_by_id(this_module, local_function_atom_index, glb);
118  uint32_t arity = READ_32_ALIGNED(table_data + i * 12 + 8 + 12);
119 
120  const struct ExportedFunction *bif = bif_registry_get_handler(module_atom, function_atom, arity);
121 
122  if (bif) {
123  this_module->imported_funcs[i] = bif;
124  } else {
125  this_module->imported_funcs[i] = &nifs_get(module_atom, function_atom, arity)->base;
126  }
127 
128  if (!this_module->imported_funcs[i]) {
129  struct UnresolvedFunctionCall *unresolved = malloc(sizeof(struct UnresolvedFunctionCall));
130  if (IS_NULL_PTR(unresolved)) {
131  fprintf(stderr, "Cannot allocate memory while loading module (line: %i).\n", __LINE__);
132  return MODULE_ERROR_FAILED_ALLOCATION;
133  }
134  unresolved->base.type = UnresolvedFunctionCall;
135  unresolved->module_atom_index = this_module->local_atoms_to_global_table[local_module_atom_index];
136  unresolved->function_atom_index = this_module->local_atoms_to_global_table[local_function_atom_index];
137  unresolved->arity = arity;
138 
139  this_module->imported_funcs[i] = &unresolved->base;
140  }
141  }
142 
143  return MODULE_LOAD_OK;
144 }

162 bool module_get_function_from_label(Module *this_module, int label, AtomString *function_name, int *arity, GlobalContext *glb)
163 {
164  int best_label = -1;
165  const uint8_t *export_table_data = (const uint8_t *) this_module->export_table;
166  int exports_count = READ_32_ALIGNED(export_table_data + 8);
167  for (int export_index = exports_count - 1; export_index >= 0; export_index--) {
168  int fun_atom_index = READ_32_ALIGNED(export_table_data + (export_index * 12) + 12);
169  int fun_arity = READ_32_ALIGNED(export_table_data + (export_index * 12) + 4 + 12);
170  int fun_label = READ_32_ALIGNED(export_table_data + (export_index * 12) + 8 + 12);
171  if (fun_label <= label && best_label < fun_label) {
172  best_label = fun_label;
173  *arity = fun_arity;
174  *function_name = module_get_atom_string_by_id(this_module, fun_atom_index, glb);
175  }
176  }
177 
178  const uint8_t *local_table_data = (const uint8_t *) this_module->local_table;
179  int locals_count = READ_32_ALIGNED(local_table_data + 8);
180  for (int local_index = locals_count - 1; local_index >= 0; local_index--) {
181  int fun_atom_index = READ_32_ALIGNED(local_table_data + (local_index * 12) + 12);
182  int fun_arity = READ_32_ALIGNED(local_table_data + (local_index * 12) + 4 + 12);
183  int fun_label = READ_32_ALIGNED(local_table_data + (local_index * 12) + 8 + 12);
184  if (fun_label <= label && best_label < fun_label) {
185  best_label = fun_label;
186  *arity = fun_arity;
187  *function_name = module_get_atom_string_by_id(this_module, fun_atom_index, glb);
188  }
189  }
190  if (UNLIKELY(best_label == -1)) {
191  // Couldn't find the function.
192  return false;
193  }
194  return true;
195 }

204 uint32_t module_search_exported_function(Module *this_module, AtomString func_name, int func_arity, GlobalContext *glb)
205 {
206  size_t functions_count = module_get_exported_functions_count(this_module);
207 
208  const uint8_t *table_data = (const uint8_t *) this_module->export_table;
209  for (unsigned int i = 0; i < functions_count; i++) {
210  AtomString function_atom = module_get_atom_string_by_id(this_module, READ_32_ALIGNED(table_data + i * 12 + 12), glb);
211  int32_t arity = READ_32_ALIGNED(table_data + i * 12 + 4 + 12);
212  if ((func_arity == arity) && atom_are_equals(func_name, function_atom)) {
213  uint32_t label = READ_32_ALIGNED(table_data + i * 12 + 8 + 12);
214  return label;
215  }
216  }
217 
218  return 0;

222 {
223  size_t functions_count = module_get_exported_functions_count(this_module);
224  term result_list = term_nil();

243 Module *module_new_from_iff_binary(GlobalContext *global, const void *iff_binary, unsigned long size)
244 {
245  uint8_t *beam_file = (void *) iff_binary;
246 
247  unsigned long offsets[MAX_OFFS];
248  unsigned long sizes[MAX_SIZES];
249  scan_iff(beam_file, size, offsets, sizes);
250 
251  Module *mod = malloc(sizeof(Module));
252  if (IS_NULL_PTR(mod)) {
253  fprintf(stderr, "Error: Failed to allocate memory: %s:%i.\n", __FILE__, __LINE__);
254  return NULL;
255  }
256  memset(mod, 0, sizeof(Module));
257 
258  mod->module_index = -1;
259 
260 #ifndef AVM_NO_SMP
261  mod->mutex = smp_mutex_create();
262 #endif
263 
264  if (UNLIKELY(module_populate_atoms_table(mod, beam_file + offsets[AT8U], global) != MODULE_LOAD_OK)) {
265  fprintf(stderr, "Error: Failed to populate atoms table: %s:%i.\n", __FILE__, __LINE__);
266  module_destroy(mod);
267  return NULL;
268  }
269 
270  if (UNLIKELY(module_build_imported_functions_table(mod, beam_file + offsets[IMPT], global) != MODULE_LOAD_OK)) {
271  fprintf(stderr, "Error: Failed to build imported functions table: %s:%i.\n", __FILE__, __LINE__);
272  module_destroy(mod);
273  return NULL;
274  }
275 
276 #ifdef ENABLE_ADVANCED_TRACE
277  mod->import_table = beam_file + offsets[IMPT];
278 #endif
279  mod->code = (CodeChunk *) (beam_file + offsets[CODE]);
280  mod->export_table = beam_file + offsets[EXPT];
281  mod->local_table = beam_file + offsets[LOCT];
282  mod->atom_table = beam_file + offsets[AT8U];
283  mod->fun_table = beam_file + offsets[FUNT];
284  mod->str_table = beam_file + offsets[STRT];
285  mod->str_table_len = sizes[STRT];
286  uint32_t num_labels = ENDIAN_SWAP_32(mod->code->labels);
287  mod->labels = calloc(num_labels, sizeof(void *));
288  if (IS_NULL_PTR(mod->labels)) {
289  fprintf(stderr, "Error: Null module labels: %s:%i.\n", __FILE__, __LINE__);
290  module_destroy(mod);
291  return NULL;
292  }
293 
294  parse_line_table(&mod->line_refs, &mod->filenames, beam_file + offsets[LINT] + 8, sizes[LINT]);
295  list_init(&mod->line_ref_offsets);
296 
297  if (offsets[LITT]) {
298  #ifdef WITH_ZLIB
299  mod->literals_data = module_uncompress_literals(beam_file + offsets[LITT], sizes[LITT]);
300  if (IS_NULL_PTR(mod->literals_data)) {
301  module_destroy(mod);
302  return NULL;
303  }
304  #else
305  fprintf(stderr, "Error: zlib required to uncompress literals.\n");
306  module_destroy(mod);
307  return NULL;
308  #endif
309 
310  mod->literals_table = module_build_literals_table(mod->literals_data);
311  mod->free_literals_data = 1;
312 
313  } else if (offsets[LITU]) {
314  mod->literals_data = beam_file + offsets[LITU] + IFF_SECTION_HEADER_SIZE;
315  mod->literals_table = module_build_literals_table(mod->literals_data);
316  mod->free_literals_data = 0;
317 
318  } else {
319  mod->literals_data = NULL;
320  mod->literals_table = NULL;
321  mod->free_literals_data = 0;
322  }
323 
324  mod->end_instruction_ii = read_core_chunk(mod);
325 
326  return mod;
327 }

344 static void *module_uncompress_literals(const uint8_t *litT, int size)
345 {
346  unsigned int required_buf_size = READ_32_ALIGNED(litT + LITT_UNCOMPRESSED_SIZE_OFFSET);
347 
348  uint8_t *outBuf = malloc(required_buf_size);
349  if (IS_NULL_PTR(outBuf)) {
350  fprintf(stderr, "Failed to allocate memory: %s:%i.\n", __FILE__, __LINE__);
351  return NULL;
352  }
353 
354  z_stream infstream;
355  infstream.zalloc = Z_NULL;
356  infstream.zfree = Z_NULL;
357  infstream.opaque = Z_NULL;
358  infstream.avail_in = (uInt) (size - IFF_SECTION_HEADER_SIZE);
359  infstream.next_in = (Bytef *) (litT + LITT_HEADER_SIZE);
360  infstream.avail_out = (uInt) required_buf_size;
361  infstream.next_out = (Bytef *) outBuf;
362 
363  int ret = inflateInit(&infstream);
364  if (ret != Z_OK) {
365  fprintf(stderr, "Failed inflateInit\n");
366  return NULL;
367  }
368  ret = inflate(&infstream, Z_NO_FLUSH);
369  if (ret != Z_OK) {
370  fprintf(stderr, "Failed inflate\n");
371  return NULL;
372  }
373  inflateEnd(&infstream);
374 
375  return outBuf;
376 }

394 
395  pos += term_size + sizeof(uint32_t);
396  }
397 
398  return literals_table;
399 }

449 static uint16_t *parse_line_refs(uint8_t **data, size_t num_refs, size_t len)
450 {
451  uint16_t *ref_table = malloc((num_refs + 1) * sizeof(uint16_t));
452  if (IS_NULL_PTR(ref_table)) {
453  return NULL;
454  }
455 
456  // assert pos >= *data
457  uint8_t *pos = *data;
458  for (size_t i = 0; i < num_refs + 1; ++i) {
459  if ((size_t) (pos - *data) > len) {
460  fprintf(stderr, "Invalid line_ref: expected tag.\n");
461  free(ref_table);
462  return NULL;
463  }
464  uint8_t tag = *pos;
465  switch (tag & 0x0F) {
466  case TAG_COMPACT_INT: {
467  uint16_t line_idx = ((tag & 0xF0) >> 4);
468  ref_table[i] = line_idx;
469  ++pos;
470  break;
471  }
472  case TAG_COMPACT_ATOM: {
473  uint16_t line_idx = ((tag & 0xF0) >> 4);
474  ref_table[i] = line_idx;
475  ++pos;
476  break;
477  }
478  case TAG_EXTENDED_INT: {
479  uint16_t high_order_3_bits = (tag & 0xE0);
480  ++pos;
481  if ((size_t) (pos - *data) > len) {
482  fprintf(stderr, "Invalid line_ref: expected extended int.\n");
483  free(ref_table);
484  return NULL;
485  }
486  uint8_t next_byte = *pos;
487  uint16_t line_idx = ((high_order_3_bits << 3) | next_byte);
488  ++pos;
489  ref_table[i] = line_idx;
490  break;
491  }
492  case TAG_EXTENDED_ATOM: {
493  uint16_t file_idx = ((tag & 0xF0) >> 4);
494  ++pos;
495  if ((size_t) (pos - *data) > len) {
496  fprintf(stderr, "Invalid line_ref: expected extended atom.\n");
497  free(ref_table);
498  return NULL;
499  }
500  uint8_t next_byte = *pos;
501  uint16_t line_idx = ((next_byte & 0xF0) >> 4);
502  ++pos;
503  ref_table[file_idx - 1] = line_idx;
504  break;
505  }
506  default:
507  // TODO handle integer compact encodings > 2048
508  fprintf(stderr, "Unsupported line_ref tag: %u\n", tag);
509  free(ref_table);
510  return NULL;
511  }
512  }
513 
514  *data = pos;
515  return ref_table;
516 }

518 struct ModuleFilename *parse_filename_table(uint8_t **data, size_t num_filenames, size_t len)
519 {
520  struct ModuleFilename *filenames = malloc(num_filenames * sizeof(struct ModuleFilename));
521  if (IS_NULL_PTR(filenames)) {
522  return NULL;
523  }
524 
525  // assert pos >= *data
526  uint8_t *pos = *data;
527  for (size_t i = 0; i < num_filenames; ++i) {
528  if ((size_t) ((pos + 2) - *data) > len) {
529  fprintf(stderr, "Invalid filename: expected 16-bit size.\n");
530  free(filenames);
531  return NULL;
532  }
533  uint16_t size = READ_16_UNALIGNED(pos);
534  pos +=2;
535  if ((size_t) ((pos + size) - *data) > len) {
536  fprintf(stderr, "Invalid filename: expected filename data (%u bytes).\n", size);
537  free(filenames);
538  return NULL;
539  }
540  filenames[i].len = size;
541  filenames[i].data = pos;
542  pos += size;
543  }
544 
545  *data = pos;
546  return filenames;
547 }

549 static void parse_line_table(uint16_t **line_refs, struct ModuleFilename **filenames, uint8_t *data, size_t len)
550 {
551 
552  *line_refs = NULL;
553  *filenames = NULL;
554 
555  if (len == 0) {
556  return;
557  }
558 
559  uint8_t *pos = data;
560 
561  CHECK_FREE_SPACE(4, "Error reading Line chunk: version\n");
562  uint32_t version = READ_32_UNALIGNED(pos);
563  if (UNLIKELY(version != 0)) {
564  fprintf(stderr, "Warning: Unsupported line version %" PRIu32 ". Line information in stacktraces may be missing\n", version);
565  return;
566  }
567  pos += 4;

614 int module_find_line(Module *mod, unsigned int offset)
615 {
616  int i = 0;
617  struct LineRefOffset *head = GET_LIST_ENTRY(&mod->line_ref_offsets, struct LineRefOffset, head);
618  struct ListHead *item;
619  LIST_FOR_EACH (item, &mod->line_ref_offsets) {
620  struct LineRefOffset *ref_offset = GET_LIST_ENTRY(item, struct LineRefOffset, head);
621 
622  if (offset == ref_offset->offset) {
623  return mod->line_refs[ref_offset->line_ref];
624  } else if (i == 0 && offset < ref_offset->offset) {
625  return -1;
626  } else {
627 
628  struct LineRefOffset *prev_ref_offset = GET_LIST_ENTRY(ref_offset->head.prev, struct LineRefOffset, head);
629  if (prev_ref_offset->offset <= offset && offset < ref_offset->offset) {
630  return mod->line_refs[prev_ref_offset->line_ref];
631  }
632 
633  struct LineRefOffset *next_ref_offset = GET_LIST_ENTRY(ref_offset->head.next, struct LineRefOffset, head);
634  if (next_ref_offset == head && ref_offset->offset <= offset) {
635  return mod->line_refs[ref_offset->line_ref];
636  }
637  }
638 
639  ++i;
640  }
641  // should never occur, but return is needed to squelch compiler warnings
642  AVM_ABORT();
643  return -1;
