49 static void cp_to_mod_lbl_off(term cp, Context *ctx, Module **cp_mod, int *label, int *l_off, long *mod_offset)
50 {
51  int module_index = cp >> 24;
52  Module *mod = globalcontext_get_module_by_index(ctx->global, module_index);
53  *mod_offset = (cp & 0xFFFFFF) >> 2;
54 
55  *cp_mod = mod;
56 
57  uint8_t *code = &mod->code->code[0];
58  int labels_count = ENDIAN_SWAP_32(mod->code->labels);
59 
60  int i = 1;
61  uint8_t *l = mod->labels[1];
62  while (*mod_offset > l - code) {
63  i++;
64  if (i >= labels_count) {
65  // last label + 1 is reserved for end of module.
66  *label = i;
67  *l_off = 0;
68  return;
69  }
70  l = mod->labels[i];
71  }
72 
73  *label = i - 1;
74  *l_off = *mod_offset - ((uint8_t *) mod->labels[*label] - code);
75 }

87 term stacktrace_create_raw(Context *ctx, Module *mod, int current_offset, term exception_class)
88 {
89  unsigned int num_frames = 0;
90  unsigned int num_aux_terms = 0;
91  unsigned int filename_lens = 0;
92  Module *prev_mod = NULL;
93  long prev_mod_offset = -1;
94  term *ct = ctx->e;
95  term *stack_base = context_stack_base(ctx);
96 
97  unsigned long stack_size = context_stack_size(ctx);
98  Module **modules = malloc(stack_size * sizeof(Module *));
99  if (IS_NULL_PTR(modules)) {
100  fprintf(stderr, "Unable to allocate space for modules list. No stacktrace will be created\n");
101  return UNDEFINED_ATOM;
102  }
103 
104  size_t num_mods = 0;
105 
106  while (ct != stack_base) {
107  if (term_is_cp(*ct)) {
108 
109  Module *cp_mod;
110  int label;
111  int offset;
112  long mod_offset;
113 
114  cp_to_mod_lbl_off(*ct, ctx, &cp_mod, &label, &offset, &mod_offset);
115  if (mod_offset != cp_mod->end_instruction_ii && !(prev_mod == cp_mod && mod_offset == prev_mod_offset)) {
116  ++num_frames;
117  prev_mod = cp_mod;
118  prev_mod_offset = mod_offset;
119  if (module_has_line_chunk(cp_mod)) {
120  if (!is_module_member(cp_mod, modules, num_mods)) {
121  modules[num_mods] = cp_mod;
122  filename_lens += cp_mod->filenames[0].len;
123  num_mods++;
124  }
125  num_aux_terms++;
126  }
127  }
128  } else if (term_is_catch_label(*ct)) {
129  int module_index;
130  int label = term_to_catch_label_and_module(*ct, &module_index);
131 
132  Module *cl_mod = globalcontext_get_module_by_index(ctx->global, module_index);
133  uint8_t *code = &cl_mod->code->code[0];
134  int mod_offset = ((uint8_t *) cl_mod->labels[label] - code);
135 
136  if (!(prev_mod == cl_mod && mod_offset == prev_mod_offset)) {
137  ++num_frames;
138  prev_mod = cl_mod;
139  prev_mod_offset = mod_offset;
140  if (module_has_line_chunk(cl_mod)) {
141  if (!is_module_member(cl_mod, modules, num_mods)) {
142  modules[num_mods] = cl_mod;
143  filename_lens += cl_mod->filenames[0].len;
144  num_mods++;
145  }
146  num_aux_terms++;
147  }
148  }
149  }
150  ct++;
151  }
152 
153  num_frames++;
154  if (module_has_line_chunk(mod)) {
155  if (!is_module_member(mod, modules, num_mods)) {
156  filename_lens += mod->filenames[0].len;
157  num_mods++;
158  }
159  num_aux_terms++;
160  }
161 
162  free(modules);
163 
164  // {num_frames, num_aux_terms, filename_lens, num_mods, [{module, offset}, ...]}
165  size_t requested_size = TUPLE_SIZE(6) + num_frames * (2 + TUPLE_SIZE(2));
166  if (UNLIKELY(memory_ensure_free(ctx, requested_size) != MEMORY_GC_OK)) {
167  fprintf(stderr, "WARNING: Unable to allocate heap space for raw stacktrace\n");
168  return OUT_OF_MEMORY_ATOM;
169  }
170 
171  term raw_stacktrace = term_nil();
172 
173  term frame_info = term_alloc_tuple(2, &ctx->heap);
174  term_put_tuple_element(frame_info, 0, term_from_int(mod->module_index));
175  term_put_tuple_element(frame_info, 1, term_from_int(current_offset));
176  raw_stacktrace = term_list_prepend(frame_info, raw_stacktrace, &ctx->heap);
177 
178  prev_mod = NULL;
179  prev_mod_offset = -1;
180  // GC may have moved stack
181  ct = ctx->e;
182  stack_base = context_stack_base(ctx);
183  while (ct != stack_base) {
184  if (term_is_cp(*ct)) {
185  Module *cp_mod;
186  int label;
187  int offset;
188  long mod_offset;
189 
190  cp_to_mod_lbl_off(*ct, ctx, &cp_mod, &label, &offset, &mod_offset);
191  if (mod_offset != cp_mod->end_instruction_ii && !(prev_mod == cp_mod && mod_offset == prev_mod_offset)) {
192 
193  prev_mod = cp_mod;
194  prev_mod_offset = mod_offset;
195 
196  term frame_info = term_alloc_tuple(2, &ctx->heap);
197  term_put_tuple_element(frame_info, 0, term_from_int(cp_mod->module_index));
198  term_put_tuple_element(frame_info, 1, term_from_int(mod_offset));
199 
200  raw_stacktrace = term_list_prepend(frame_info, raw_stacktrace, &ctx->heap);
201  }
202  } else if (term_is_catch_label(*ct)) {
203 
204  int module_index;
205  int label = term_to_catch_label_and_module(*ct, &module_index);
206  Module *cl_mod = globalcontext_get_module_by_index(ctx->global, module_index);
207  uint8_t *code = &cl_mod->code->code[0];
208  int mod_offset = ((uint8_t *) cl_mod->labels[label] - code);
209 
210  if (!(prev_mod == cl_mod && mod_offset == prev_mod_offset)) {
211 
212  prev_mod = cl_mod;
213  prev_mod_offset = mod_offset;
214 
215  term frame_info = term_alloc_tuple(2, &ctx->heap);
216  term_put_tuple_element(frame_info, 0, term_from_int(module_index));
217  term_put_tuple_element(frame_info, 1, term_from_int(mod_offset));
218 
219  raw_stacktrace = term_list_prepend(frame_info, raw_stacktrace, &ctx->heap);
220  }
221  }
222  ct++;
223  }
224 
225  term stack_info = term_alloc_tuple(6, &ctx->heap);
226  term_put_tuple_element(stack_info, 0, term_from_int(num_frames));
227  term_put_tuple_element(stack_info, 1, term_from_int(num_aux_terms));
228  term_put_tuple_element(stack_info, 2, term_from_int(filename_lens));
229  term_put_tuple_element(stack_info, 3, term_from_int(num_mods));
230  term_put_tuple_element(stack_info, 4, raw_stacktrace);
231  term_put_tuple_element(stack_info, 5, exception_class);
232 
233  return stack_info;
234 }

257 term stacktrace_build(Context *ctx, term *stack_info)
258 {
259  GlobalContext *glb = ctx->global;
260 
261  if (*stack_info == OUT_OF_MEMORY_ATOM) {
262  return *stack_info;
263  }
264  if (!term_is_tuple(*stack_info)) {
265  return UNDEFINED_ATOM;
266  }
267 
268  int num_frames = term_to_int(term_get_tuple_element(*stack_info, 0));
269  int num_aux_terms = term_to_int(term_get_tuple_element(*stack_info, 1));
270  int filename_lens = term_to_int(term_get_tuple_element(*stack_info, 2));
271  int num_mods = term_to_int(term_get_tuple_element(*stack_info, 3));
272 
273  struct ModulePathPair *module_paths = malloc(num_mods * sizeof(struct ModulePathPair));
274  if (IS_NULL_PTR(module_paths)) {
275  fprintf(stderr, "Unable to allocate space for module paths. Returning raw stacktrace.\n");
276  return *stack_info;
277  }
278 
279  //
280  // [{module, function, arity, [{file, string()}, {line, int}]}, ...]
281  //
282  size_t requested_size = (TUPLE_SIZE(4) + 2) * num_frames + num_aux_terms * (4 + 2 * TUPLE_SIZE(2)) + 2 * filename_lens;
283  if (UNLIKELY(memory_ensure_free(ctx, requested_size) != MEMORY_GC_OK)) {
284  free(module_paths);
285  return OUT_OF_MEMORY_ATOM;
286  }
287 
288  // Note. Safe to get stacktrace after GC when stack_info comes from x[0]
289  term raw_stacktrace = term_get_tuple_element(*stack_info, 4);
290 
291  term stacktrace = term_nil();
292  term el = raw_stacktrace;
293  int module_path_idx = 0;
294  while (!term_is_nil(el)) {
295  term mod_index_tuple = term_get_list_head(el);
296  term cp = module_address(
297  term_to_int(term_get_tuple_element(mod_index_tuple, 0)),
298  term_to_int(term_get_tuple_element(mod_index_tuple, 1)));
299 
300  Module *cp_mod;
301  int label;
302  int offset;
303  long mod_offset;
304  cp_to_mod_lbl_off(cp, ctx, &cp_mod, &label, &offset, &mod_offset);
305 
306  term module_name = module_get_name(cp_mod);
307 
308  term frame_i = term_alloc_tuple(4, &ctx->heap);
309  term_put_tuple_element(frame_i, 0, module_name);
310 
311  term aux_data = term_nil();
312  if (module_has_line_chunk(cp_mod)) {
313  term line_tuple = term_alloc_tuple(2, &ctx->heap);
314  term_put_tuple_element(line_tuple, 0, globalcontext_make_atom(glb, ATOM_STR("\x4", "line")));
315  int line = module_find_line(cp_mod, (unsigned int) mod_offset);
316  term_put_tuple_element(line_tuple, 1, line == -1 ? UNDEFINED_ATOM : term_from_int(line));
317  aux_data = term_list_prepend(line_tuple, aux_data, &ctx->heap);
318 
319  term file_tuple = term_alloc_tuple(2, &ctx->heap);
320  term_put_tuple_element(file_tuple, 0, globalcontext_make_atom(glb, ATOM_STR("\x4", "file")));
321 
322  term path = find_path_created(module_name, module_paths, module_path_idx);
323  if (term_is_invalid_term(path)) {
324  path = term_from_string((const uint8_t *) cp_mod->filenames[0].data, cp_mod->filenames[0].len, &ctx->heap);
325  module_paths[module_path_idx].module = module_name;
326  module_paths[module_path_idx].path = path;
327  module_path_idx++;
328  }
329  term_put_tuple_element(file_tuple, 1, path);
330  aux_data = term_list_prepend(file_tuple, aux_data, &ctx->heap);
331  }
332  term_put_tuple_element(frame_i, 3, aux_data);
333 
334  AtomString function_name = NULL;
335  int arity = 0;
336  bool result = module_get_function_from_label(cp_mod, label, &function_name, &arity, glb);
337 
338  if (LIKELY(result)) {
339  term_put_tuple_element(frame_i, 1, globalcontext_make_atom(glb, function_name));
340  term_put_tuple_element(frame_i, 2, term_from_int(arity));
341  } else {
342  term_put_tuple_element(frame_i, 1, UNDEFINED_ATOM);
343  term_put_tuple_element(frame_i, 2, term_from_int(0));
344  }
345  stacktrace = term_list_prepend(frame_i, stacktrace, &ctx->heap);
346 
347  el = term_get_list_tail(el);
348  }
349  free(module_paths);
350 
351  return stacktrace;
352 }
