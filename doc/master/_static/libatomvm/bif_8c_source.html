347 static term add_boxed_helper(Context *ctx, term arg1, term arg2)
348 {
349  int use_float = 0;
350  int size = 0;
351  if (term_is_boxed_integer(arg1)) {
352  size = term_boxed_size(arg1);
353  } else if (term_is_float(arg1)) {
354  use_float = 1;
355  } else if (!term_is_integer(arg1)) {
356  TRACE("error: arg1: 0x%lx, arg2: 0x%lx\n", arg1, arg2);
357  RAISE_ERROR(BADARITH_ATOM);
358  }
359 
360  if (term_is_boxed_integer(arg2)) {
361  size |= term_boxed_size(arg2);
362  } else if (term_is_float(arg2)) {
363  use_float = 1;
364  } else if (!term_is_integer(arg2)) {
365  TRACE("error: arg1: 0x%lx, arg2: 0x%lx\n", arg1, arg2);
366  RAISE_ERROR(BADARITH_ATOM);
367  }
368 
369  if (use_float) {
370  avm_float_t farg1 = term_conv_to_float(arg1);
371  avm_float_t farg2 = term_conv_to_float(arg2);
372  avm_float_t fresult = farg1 + farg2;
373  if (UNLIKELY(!isfinite(fresult))) {
374  RAISE_ERROR(BADARITH_ATOM);
375  }
376 
377  if (UNLIKELY(memory_ensure_free_opt(ctx, FLOAT_SIZE, MEMORY_CAN_SHRINK) != MEMORY_GC_OK)) {
378  RAISE_ERROR(OUT_OF_MEMORY_ATOM);
379  }
380  return term_from_float(fresult, &ctx->heap);
381  }
382 
383  switch (size) {
384  case 0: {
385  //BUG
386  AVM_ABORT();
387  }
388 
389  case 1: {
390  avm_int_t val1 = term_maybe_unbox_int(arg1);
391  avm_int_t val2 = term_maybe_unbox_int(arg2);
392  avm_int_t res;
393 
394  if (BUILTIN_ADD_OVERFLOW_INT(val1, val2, &res)) {
395  #if BOXED_TERMS_REQUIRED_FOR_INT64 == 2
396  avm_int64_t res64 = (avm_int64_t) val1 + (avm_int64_t) val2;
397  return make_boxed_int64(ctx, res64);
398 
399  #elif BOXED_TERMS_REQUIRED_FOR_INT64 == 1
400  TRACE("overflow: arg1: " AVM_INT64_FMT ", arg2: " AVM_INT64_FMT "\n", arg1, arg2);
401  RAISE_ERROR(OVERFLOW_ATOM);
402  #else
403  #error "Unsupported configuration."
404  #endif
405  }
406 
407  return make_maybe_boxed_int(ctx, res);
408  }
409 
410  #if BOXED_TERMS_REQUIRED_FOR_INT64 == 2
411  case 2:
412  case 3: {
413  avm_int64_t val1 = term_maybe_unbox_int64(arg1);
414  avm_int64_t val2 = term_maybe_unbox_int64(arg2);
415  avm_int64_t res;
416 
417  if (BUILTIN_ADD_OVERFLOW_INT64(val1, val2, &res)) {
418  TRACE("overflow: val1: " AVM_INT64_FMT ", val2: " AVM_INT64_FMT "\n", arg1, arg2);
419  RAISE_ERROR(OVERFLOW_ATOM);
420  }
421 
422  return make_maybe_boxed_int64(ctx, res);
423  }
424  #endif
425 
426  default:
427  RAISE_ERROR(OVERFLOW_ATOM);
428  }
429 }
