297 static term nif_socket_open(Context *ctx, int argc, term argv[])
298 {
299  TRACE("nif_socket_open\n");
300 
301  UNUSED(argc);
302 
303  GlobalContext *global = ctx->global;
304 
305  bool ok = false;
306  int domain = get_domain(global, argv[0], &ok);
307  if (!ok) {
308  RAISE_ERROR(BADARG_ATOM);
309  }
310  int type = get_type(global, argv[1], &ok);
311  if (!ok) {
312  RAISE_ERROR(BADARG_ATOM);
313  }
314  int protocol = get_protocol(global, argv[2], &ok);
315  if (!ok) {
316  RAISE_ERROR(BADARG_ATOM);
317  }
318 
319  int fd = socket(domain, type, protocol);
320  if (UNLIKELY(fd == -1 || fd == CLOSED_FD)) {
321 
322  AVM_LOGE(TAG, "Failed to initialize socket.");
323 
324  if (UNLIKELY(memory_ensure_free(ctx, TUPLE_SIZE(2)) != MEMORY_GC_OK)) {
325  AVM_LOGW(TAG, "Failed to allocate memory: %s:%i.", __FILE__, __LINE__);
326  RAISE_ERROR(OUT_OF_MEMORY_ATOM);
327  }
328 
329  return make_error_tuple(posix_errno_to_term(errno, global), ctx);
330 
331  } else {
332 
333  struct SocketFd *fd_obj = enif_alloc_resource(socket_resource_type, sizeof(struct SocketFd));
334 
335  if (IS_NULL_PTR(fd_obj)) {
336  AVM_LOGW(TAG, "Failed to allocate memory: %s:%i.", __FILE__, __LINE__);
337  RAISE_ERROR(OUT_OF_MEMORY_ATOM);
338  }
339 
340  fd_obj->fd = fd;
341  fd_obj->selecting_process_id = INVALID_PROCESS_ID;
342  TRACE("nif_socket_open: Created socket fd=%i\n", fd_obj->fd);
343 
344  if (UNLIKELY(memory_ensure_free(ctx, TERM_BOXED_RESOURCE_SIZE) != MEMORY_GC_OK)) {
345  AVM_LOGW(TAG, "Failed to allocate memory: %s:%i.", __FILE__, __LINE__);
346  RAISE_ERROR(OUT_OF_MEMORY_ATOM);
347  }
348  term obj = enif_make_resource(erl_nif_env_from_context(ctx), fd_obj);
349  enif_release_resource(fd_obj);
350 
351  size_t requested_size = TUPLE_SIZE(2) + TUPLE_SIZE(2) + REF_SIZE;
352  if (UNLIKELY(memory_ensure_free_with_roots(ctx, requested_size, 1, &obj, MEMORY_CAN_SHRINK) != MEMORY_GC_OK)) {
353  AVM_LOGW(TAG, "Failed to allocate memory: %s:%i.", __FILE__, __LINE__);
354  RAISE_ERROR(OUT_OF_MEMORY_ATOM);
355  }
356 
357  term socket_term = term_alloc_tuple(2, &ctx->heap);
358  uint64_t ref_ticks = globalcontext_get_ref_ticks(ctx->global);
359  term ref = term_from_ref_ticks(ref_ticks, &ctx->heap);
360  term_put_tuple_element(socket_term, 0, obj);
361  term_put_tuple_element(socket_term, 1, ref);
362 
363  term result = term_alloc_tuple(2, &ctx->heap);
364  term_put_tuple_element(result, 0, OK_ATOM);
365  term_put_tuple_element(result, 1, socket_term);
366 
367  return result;
368  }
369 }

392 static term nif_socket_close(Context *ctx, int argc, term argv[])
393 {
394  TRACE("nif_socket_close\n");
395  UNUSED(argc);
396 
397  VALIDATE_VALUE(argv[0], term_is_socket);
398 
399  void *fd_obj_ptr;
400  if (UNLIKELY(!enif_get_resource(erl_nif_env_from_context(ctx), get_socket(argv[0]), socket_resource_type, &fd_obj_ptr))) {
401  RAISE_ERROR(BADARG_ATOM);
402  }
403  struct SocketFd *fd_obj = (struct SocketFd *) fd_obj_ptr;
404  if (fd_obj->fd) {
405  if (fd_obj->selecting_process_id != INVALID_PROCESS_ID) {
406  //
407  // If we are in select, then stop selecting
408  //
409  if (UNLIKELY(enif_select(erl_nif_env_from_context(ctx), fd_obj->fd, ERL_NIF_SELECT_STOP, fd_obj, NULL, term_nil()) < 0)) {
410  RAISE_ERROR(BADARG_ATOM);
411  }
412  //
413  // If there is a process (other than ourself) who is waiting on select
414  // the send a {closed, Ref} message to it, so that it can break
415  // out of its receive statement.
416  //
417  if (fd_obj->selecting_process_id != ctx->process_id) {
418 
419  // send a {closed, Ref | undefined} message to the pid
420  if (UNLIKELY(memory_ensure_free(ctx, TUPLE_SIZE(2) + REF_SIZE) != MEMORY_GC_OK)) {
421  AVM_LOGW(TAG, "Failed to allocate memory: %s:%i.", __FILE__, __LINE__);
422  RAISE_ERROR(OUT_OF_MEMORY_ATOM);
423  }
424 
425  term error_tuple = term_alloc_tuple(2, &ctx->heap);
426  term_put_tuple_element(error_tuple, 0, CLOSED_ATOM);
427  term ref = (fd_obj->ref_ticks == 0) ? UNDEFINED_ATOM : term_from_ref_ticks(fd_obj->ref_ticks, &ctx->heap);
428  term_put_tuple_element(error_tuple, 1, ref);
429 
430  TRACE("nif_socket_close: Sending msg to process %i\n", fd_obj->selecting_process_id);
431  globalcontext_send_message(ctx->global, fd_obj->selecting_process_id, error_tuple);
432  } else {
433  AVM_LOGW(TAG, "Selectable socket %i was closed but no known pid is waiting. This shouldn't happen.", fd_obj->fd);
434  }
435  }
436 
437  int res = close(fd_obj->fd);
438  if (UNLIKELY(res != 0)) {
439  AVM_LOGW(TAG, "Failed to close socket %i", res);
440  }
441 
442  TRACE("nif_socket_close: Clearing pid for socket fd=%i\n", fd_obj->fd);
443  fd_obj->fd = CLOSED_FD;
444  fd_obj->selecting_process_id = INVALID_PROCESS_ID;
445  fd_obj->ref_ticks = 0;
446  } else {
447  TRACE("Double close on socket fd %i", fd_obj->fd);
448  }
449 
450  return OK_ATOM;
451 }

457 static term nif_socket_select(Context *ctx, term argv[], enum ErlNifSelectFlags mode)
458 {
459  TRACE("nif_socket_select\n");
460 
461  VALIDATE_VALUE(argv[0], term_is_socket);
462 
463  term process_pid_term = argv[1];
464  VALIDATE_VALUE(process_pid_term, term_is_pid);
465  int32_t process_pid = term_to_local_process_id(process_pid_term);
466  TRACE("process_pid=%i\n", (int) process_pid);
467 
468  term select_ref_term = argv[2];
469  if (select_ref_term != UNDEFINED_ATOM) {
470  VALIDATE_VALUE(select_ref_term, term_is_reference);
471  }
472  void *fd_obj_ptr;
473  if (UNLIKELY(!enif_get_resource(erl_nif_env_from_context(ctx), get_socket(argv[0]), socket_resource_type, &fd_obj_ptr))) {
474  RAISE_ERROR(BADARG_ATOM);
475  }
476  struct SocketFd *fd_obj = (struct SocketFd *) fd_obj_ptr;
477  TRACE("fd_obj->fd=%i\n", (int) fd_obj->fd);
478 
479  ErlNifEnv *env = erl_nif_env_from_context(ctx);
480  if (fd_obj->selecting_process_id != process_pid && fd_obj->selecting_process_id != INVALID_PROCESS_ID) {
481  if (UNLIKELY(enif_demonitor_process(env, fd_obj, &fd_obj->selecting_process_monitor) != 0)) {
482  // RAISE_ERROR(posix_errno_to_term(EINVAL, global));
483  }
484  fd_obj->selecting_process_id = INVALID_PROCESS_ID;
485  }
486  // Monitor first as select is less likely to fail and it's less expensive to demonitor
487  // if select fails than to stop select if monitor fails
488  if (fd_obj->selecting_process_id != process_pid) {
489  if (UNLIKELY(enif_monitor_process(env, fd_obj, &process_pid, &fd_obj->selecting_process_monitor) != 0)) {
490  RAISE_ERROR(NOPROC_ATOM);
491  }
492  fd_obj->selecting_process_id = process_pid;
493  }
494 
495  if (UNLIKELY(enif_select(erl_nif_env_from_context(ctx), fd_obj->fd, mode, fd_obj, &process_pid, select_ref_term) < 0)) {
496  enif_demonitor_process(env, fd_obj, &fd_obj->selecting_process_monitor);
497  fd_obj->selecting_process_id = INVALID_PROCESS_ID;
498  RAISE_ERROR(BADARG_ATOM);
499  }
500 
501  TRACE("nif_socket_select: Setting pid for socket fd %i to %i\n", fd_obj->fd, process_pid);
502  fd_obj->ref_ticks = (select_ref_term == UNDEFINED_ATOM) ? 0 : term_to_ref_ticks(select_ref_term);
503 
504  return OK_ATOM;
505 }
