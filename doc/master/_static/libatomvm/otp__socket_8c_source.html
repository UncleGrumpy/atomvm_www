232 static void socket_dtor(ErlNifEnv *caller_env, void *obj)
233 {
234  UNUSED(caller_env);
235 
236  struct SocketResource *rsrc_obj = (struct SocketResource *) obj;
237 
238  TRACE("socket_dtor called on resource=%p\n", (void *) rsrc_obj);
239 #if OTP_SOCKET_BSD
240  if (rsrc_obj->fd != CLOSED_FD) {
241  close(rsrc_obj->fd);
242  rsrc_obj->fd = CLOSED_FD;
243  }
244 #elif OTP_SOCKET_LWIP
245  LWIP_BEGIN();
246  if (rsrc_obj->socket_state & SocketStateUDP) {
247  udp_remove(rsrc_obj->udp_pcb);
248  rsrc_obj->udp_pcb = NULL;
249  } else if (rsrc_obj->socket_state & SocketStateTCP) {
250  // Try to nicely close the connection here
251  // NOTE: we can't handle linger if the socket is gone.
252  if (UNLIKELY(tcp_close(rsrc_obj->tcp_pcb) != ERR_OK)) {
253  // The resource will be gone, there is not much we can do here.
254  if (!(rsrc_obj->socket_state & SocketStateTCPListening)) {
255  tcp_abort(rsrc_obj->tcp_pcb);
256  }
257  }
258  rsrc_obj->tcp_pcb = NULL;
259  }
260  LWIP_END();
261 #endif
262 }

282 static void socket_down(ErlNifEnv *caller_env, void *obj, ErlNifPid *pid, ErlNifMonitor *mon)
283 {
284  UNUSED(caller_env);
285  UNUSED(pid);
286  UNUSED(mon);
287 
288  struct SocketResource *rsrc_obj = (struct SocketResource *) obj;
289 
290 #if OTP_SOCKET_BSD
291  TRACE("socket_down called on process_id=%i fd=%i\n", (int) *pid, rsrc_obj->fd);
292 #else
293  TRACE("socket_down called on process_id=%i\n", (int) *pid);
294 #endif
295 
296 #if OTP_SOCKET_BSD
297  if (rsrc_obj->selecting_process_id != INVALID_PROCESS_ID) {
298  // Monitor fired, so make sure we don't try to demonitor in select_stop
299  // as it could crash trying to reacquire lock on process table
300  rsrc_obj->selecting_process_id = INVALID_PROCESS_ID;
301  enif_select(caller_env, rsrc_obj->fd, ERL_NIF_SELECT_STOP, rsrc_obj, NULL, term_nil());
302  }
303 #elif OTP_SOCKET_LWIP
304  // Monitor can be called when we're selecting, accepting or connecting.
305  if (rsrc_obj->selecting_process_id != INVALID_PROCESS_ID) {
306  LWIP_BEGIN();
307  rsrc_obj->selecting_process_id = INVALID_PROCESS_ID;
308  if (rsrc_obj->socket_state & SocketStateTCP) {
309  if (rsrc_obj->socket_state & SocketStateTCPListening) {
310  (void) tcp_close(rsrc_obj->tcp_pcb);
311  } else {
312  tcp_abort(rsrc_obj->tcp_pcb);
313  }
314  rsrc_obj->tcp_pcb = NULL;
315  rsrc_obj->socket_state = SocketStateClosed;
316  } else if (rsrc_obj->socket_state & SocketStateUDP) {
317  udp_remove(rsrc_obj->udp_pcb);
318  rsrc_obj->udp_pcb = NULL;
319  rsrc_obj->socket_state = SocketStateClosed;
320  }
321  LWIP_END();
322  }
323 #endif
324 }

476 static term nif_socket_open(Context *ctx, int argc, term argv[])
477 {
478  TRACE("nif_socket_open\n");
479 
480  UNUSED(argc);
481 
482  GlobalContext *global = ctx->global;
483 
484  bool ok = false;
485  int domain = get_domain(global, argv[0], &ok);
486  if (!ok) {
487  RAISE_ERROR(BADARG_ATOM);
488  }
489  int type = get_type(global, argv[1], &ok);
490  if (!ok) {
491  RAISE_ERROR(BADARG_ATOM);
492  }
493  int protocol = get_protocol(global, argv[2], &ok);
494  if (!ok) {
495  RAISE_ERROR(BADARG_ATOM);
496  }
497 
498  struct SocketResource *rsrc_obj = enif_alloc_resource(socket_resource_type, sizeof(struct SocketResource));
499  if (IS_NULL_PTR(rsrc_obj)) {
500  AVM_LOGW(TAG, "Failed to allocate memory: %s:%i.\n", __FILE__, __LINE__);
501  RAISE_ERROR(OUT_OF_MEMORY_ATOM);
502  }
503 
504 #if OTP_SOCKET_BSD
505  rsrc_obj->fd = socket(domain, type, protocol);
506  if (UNLIKELY(rsrc_obj->fd == -1 || rsrc_obj->fd == CLOSED_FD)) {
507  AVM_LOGE(TAG, "Failed to initialize socket.");
508  enif_release_resource(rsrc_obj);
509  return make_errno_tuple(ctx);
510  } else {
511  TRACE("nif_socket_open: Created socket fd=%i\n", rsrc_obj->fd);
512  rsrc_obj->selecting_process_id = INVALID_PROCESS_ID;
513 
514 #elif OTP_SOCKET_LWIP
515  if (domain == PF_INET && type == SOCK_STREAM && protocol == IPPROTO_TCP) {
516  LWIP_BEGIN();
517  rsrc_obj->tcp_pcb = tcp_new();
518  LWIP_END();
519  rsrc_obj->socket_state = SocketStateTCPNew;
520  } else if (domain == PF_INET && type == SOCK_DGRAM && protocol == IPPROTO_UDP) {
521  LWIP_BEGIN();
522  rsrc_obj->udp_pcb = udp_new();
523  LWIP_END();
524  rsrc_obj->socket_state = SocketStateUDPIdle;
525  } else {
526  enif_release_resource(rsrc_obj);
527  RAISE_ERROR(BADARG_ATOM);
528  }
529  if (UNLIKELY(rsrc_obj->udp_pcb == NULL && rsrc_obj->tcp_pcb == NULL)) {
530  rsrc_obj->socket_state = SocketStateClosed;
531  enif_release_resource(rsrc_obj);
532  RAISE_ERROR(OUT_OF_MEMORY_ATOM);
533  } else {
534  rsrc_obj->selecting_process_id = INVALID_PROCESS_ID;
535  rsrc_obj->linger_on = false;
536  rsrc_obj->linger_sec = 0;
537  rsrc_obj->pos = 0;
538  list_init(&rsrc_obj->received_list);
539  if (rsrc_obj->socket_state & SocketStateTCP) {
540  LWIP_BEGIN();
541  tcp_arg(rsrc_obj->tcp_pcb, rsrc_obj);
542  tcp_recv(rsrc_obj->tcp_pcb, tcp_recv_cb);
543  LWIP_END();
544  } else {
545  LWIP_BEGIN();
546  udp_recv(rsrc_obj->udp_pcb, udp_recv_cb, rsrc_obj);
547  LWIP_END();
548  }
549 #endif
550  rsrc_obj->buf_size = DEFAULT_BUFFER_SIZE;
551 
552  if (UNLIKELY(memory_ensure_free(ctx, TERM_BOXED_RESOURCE_SIZE) != MEMORY_GC_OK)) {
553  AVM_LOGW(TAG, "Failed to allocate memory: %s:%i.", __FILE__, __LINE__);
554  RAISE_ERROR(OUT_OF_MEMORY_ATOM);
555  }
556  term obj = enif_make_resource(erl_nif_env_from_context(ctx), rsrc_obj);
557  enif_release_resource(rsrc_obj);
558 
559  size_t requested_size = TUPLE_SIZE(2) + TUPLE_SIZE(2) + REF_SIZE;
560  if (UNLIKELY(memory_ensure_free_with_roots(ctx, requested_size, 1, &obj, MEMORY_CAN_SHRINK) != MEMORY_GC_OK)) {
561  AVM_LOGW(TAG, "Failed to allocate memory: %s:%i.", __FILE__, __LINE__);
562  RAISE_ERROR(OUT_OF_MEMORY_ATOM);
563  }
564 
565  term socket_term = term_alloc_tuple(2, &ctx->heap);
566  uint64_t ref_ticks = globalcontext_get_ref_ticks(ctx->global);
567  term ref = term_from_ref_ticks(ref_ticks, &ctx->heap);
568  term_put_tuple_element(socket_term, 0, obj);
569  term_put_tuple_element(socket_term, 1, ref);
570 
571  term result = term_alloc_tuple(2, &ctx->heap);
572  term_put_tuple_element(result, 0, OK_ATOM);
573  term_put_tuple_element(result, 1, socket_term);
574 
575  return result;
576  }
577 }

642  }

650  if (UNLIKELY(stop_res < 0)) {

653  // TODO: check if stop_res & ERL_NIF_SELECT_STOP_CALLED or ERL_NIF_SELECT_STOP_SCHEDULED
654  // following what OTP does. Indeed, if we have ERL_NIF_SELECT_STOP_SCHEDULED, we should not close the socket now
655  //
656  // If there is a process (other than ourself) who is waiting on select
657  // the send a {closed, Ref} message to it, so that it can break
658  // out of its receive statement.
659  //
660  if (rsrc_obj->selecting_process_id != ctx->process_id) {

663  if (UNLIKELY(send_closed_notification(ctx, rsrc_obj) < 0)) {

673  AVM_LOGW(TAG, "Failed to close socket %i", res);

677  rsrc_obj->fd = CLOSED_FD;
678  rsrc_obj->selecting_process_id = INVALID_PROCESS_ID;
679  rsrc_obj->ref_ticks = 0;
680  } else {
681  TRACE("Double close on socket fd %i", rsrc_obj->fd);

683 #elif OTP_SOCKET_LWIP

685  if (rsrc_obj->socket_state & SocketStateSelectingRead
686  && rsrc_obj->selecting_process_id != INVALID_PROCESS_ID
687  && rsrc_obj->selecting_process_id != ctx->process_id) {
688  // send a {closed, Ref | undefined} message to the pid
689  if (UNLIKELY(send_closed_notification(ctx, rsrc_obj) < 0)) {
690  RAISE_ERROR(OUT_OF_MEMORY_ATOM);
691  }
692  }
693  if (rsrc_obj->socket_state == SocketStateClosed) {
694  TRACE("Double close on pcb");
695  } else {
696  if (rsrc_obj->socket_state & SocketStateTCP) {
697  LWIP_BEGIN();
698  tcp_arg(rsrc_obj->tcp_pcb, NULL);
699  err_t err = tcp_close(rsrc_obj->tcp_pcb);
700  LWIP_END();
701  if (UNLIKELY(err != ERR_OK)) {
702  AVM_LOGW(TAG, "tcp_close failed with err=%d: %s:%i.\n", err, __FILE__, __LINE__);
703  RAISE_ERROR(OUT_OF_MEMORY_ATOM);
704  }
705  rsrc_obj->tcp_pcb = NULL;
706  } else if (rsrc_obj->socket_state & SocketStateUDP) {
707  LWIP_BEGIN();
708  udp_recv(rsrc_obj->udp_pcb, udp_recv_cb, NULL);
709  udp_remove(rsrc_obj->udp_pcb);
710  LWIP_END();
711  rsrc_obj->udp_pcb = NULL;
712  }
713  rsrc_obj->socket_state = SocketStateClosed;
714  // The resource should not go away until all callbacks are processed.
715  enif_keep_resource(rsrc_obj);
716  struct LWIPEvent event;
717  event.handler = finalize_close_hander;
718  event.finalize_close.rsrc_obj = rsrc_obj;
719  otp_socket_lwip_enqueue(&event);
720  }
721 
722 #endif
723  rsrc_obj->buf_size = DEFAULT_BUFFER_SIZE;
724

770 static err_t tcp_accept_cb(void *arg, struct tcp_pcb *newpcb, err_t err)
771 {
772  UNUSED(err);
773 
774  struct SocketResource *rsrc_obj = (struct SocketResource *) arg;
775  if (rsrc_obj->selecting_process_id != INVALID_PROCESS_ID) {
776  if (newpcb != NULL) {
777  // Because data can come in, we need to immediately set the receive callback
778  // However, we cannot allocate the resource because we can't call malloc from ISR.
779  tcp_arg(newpcb, NULL);
780  tcp_recv(newpcb, tcp_recv_cb);
781 
782  // Delay accepting
783  tcp_backlog_delayed(newpcb);

785  // Enqueue an event to further process accept
786  struct LWIPEvent event;
787  event.handler = tcp_accept_handler;
788  event.tcp_accept.rsrc_obj = rsrc_obj;
789  event.tcp_accept.newpcb = newpcb;

791  } else {

793  AVM_LOGW(TAG, "Insufficient memory to accept connections, client will have to resend SYN.");
794  }
795  }
796  return ERR_OK;
797  } else {
798  // Selecting process died
799  if (newpcb != NULL) {
800  if (rsrc_obj->socket_state & SocketStateTCPListening) {
801  (void) tcp_close(newpcb);
802  } else {
803  tcp_abort(newpcb);
804  }
805  }
806  return ERR_ABRT;
807  }

1588 ssize_t socket_recv(struct SocketResource *rsrc_obj, uint8_t *buf, size_t len, int flags, term *from, Heap *heap)
1589 {
1590 #if OTP_SOCKET_BSD
1591  //
1592  // receive data on the socket
1593  //
1594  struct sockaddr_in addr;
1595  socklen_t addrlen = sizeof(addr);
1596  ssize_t res;
1597  if (from) {
1598  struct RefcBinary *rsrc_refc = refc_binary_from_data(rsrc_obj);
1599  GlobalContext *global = rsrc_refc->resource_type->global;
1600

1602

1604  term port_number = term_from_int(ntohs(addr.sin_port));
1605 
1606  term map = term_alloc_map(2, heap);
1607  term_set_map_assoc(map, 0, ADDR_ATOM, address);
1608  term_set_map_assoc(map, 1, PORT_ATOM, port_number);
1609  *from = map;
1610  } else {
1611  res = recv(rsrc_obj->fd, buf, len, flags);
1612  }
1613  if (res == 0) {
1614  return SocketClosed;
1615  }
1616  if (res < 0) {
1617  if (errno == EAGAIN || errno == EWOULDBLOCK) {
1618  return SocketWouldBlock;
1619  }
1620  return SocketOtherError;
1621  }
1622  return res;
1623 #elif OTP_SOCKET_LWIP
1624  UNUSED(flags);
1625 
1626  uint32_t ip4_u32;
1627  uint16_t port_u16;
1628 
1629  size_t remaining = len;
1630  uint8_t *ptr = buf;
1631  bool closed = false;
1632  err_t err = ERR_OK;
1633  // Use lwIP lock
1634  LWIP_BEGIN();
1635  if (rsrc_obj->socket_state & SocketStateTCP) {
1636  size_t pos = rsrc_obj->pos;
1637  struct ListHead *item;
1638  struct ListHead *tmp;
1639  MUTABLE_LIST_FOR_EACH (item, tmp, &rsrc_obj->received_list) {
1640  struct TCPReceivedItem *received_item = GET_LIST_ENTRY(item, struct TCPReceivedItem, list_head);
1641  if (received_item->buf == NULL || received_item->err != ERR_OK) {
1642  closed = received_item->buf == NULL;
1643  err = received_item->err;
1644  break;
1645  }
1646  if (pos < received_item->buf->tot_len) {
1647  size_t copied = copy_pbuf_data(received_item->buf, pos, remaining, ptr);
1648  ptr += copied;
1649  remaining -= copied;
1650  tcp_recved(rsrc_obj->tcp_pcb, copied);
1651  if (copied + pos == received_item->buf->tot_len) {
1652  // all data was copied.
1653  list_remove(item);
1654  pbuf_free(received_item->buf);
1655  pos = 0;
1656  } else {
1657  pos = pos + copied;
1658  }
1659  if (remaining == 0) {
1660  break;
1661  }
1662  } else {
1663  pos -= received_item->buf->tot_len;
1664  }
1665  }
1666  rsrc_obj->pos = pos;
1667  if (from) {
1668  ip4_u32 = ntohl(ip_addr_get_ip4_u32(&rsrc_obj->tcp_pcb->remote_ip));
1669  port_u16 = rsrc_obj->tcp_pcb->remote_port;
1670  }
1671  } else {

1673  size_t copied = copy_pbuf_data(first_item->buf, 0, remaining, ptr);
1674  remaining -= copied;
1675  if (from) {
1676  ip4_u32 = first_item->addr;
1677  port_u16 = first_item->port;
1678  }
1679  list_remove(&first_item->list_head);
1680  pbuf_free(first_item->buf);
1681  free(first_item);
1682  }
1683  LWIP_END();
1684  if (remaining < len) {
1685  if (from) {
1686  struct RefcBinary *rsrc_refc = refc_binary_from_data(rsrc_obj);
1687  GlobalContext *global = rsrc_refc->resource_type->global;
1688 
1689  term address = inet_make_addr4(ip4_u32, heap);
1690  term port_number = term_from_int(port_u16);
1691 
1692  term map = term_alloc_map(2, heap);
1693  term_set_map_assoc(map, 0, globalcontext_make_atom(global, addr_atom), address);
1694  term_set_map_assoc(map, 1, PORT_ATOM, port_number);
1695
