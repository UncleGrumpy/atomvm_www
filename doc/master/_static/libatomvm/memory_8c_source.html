63 enum MemoryGCResult memory_ensure_free_with_roots(Context *c, size_t size, size_t num_roots, term *roots, enum MemoryAllocMode alloc_mode)
64 {
65  size_t free_space = context_avail_free_memory(c);
66  if (alloc_mode == MEMORY_NO_GC) {
67  if (free_space < size) {
68  HeapFragment *root_fragment = c->heap.root;
69  term *old_end = c->heap.heap_end;
70  term mso_list = root_fragment->mso_list;
71  if (UNLIKELY(memory_init_heap(&c->heap, size) != MEMORY_GC_OK)) {
72  TRACE("Unable to allocate memory fragment. size=%u\n", size);
73  return MEMORY_GC_ERROR_FAILED_ALLOCATION;
74  }
75  // Convert root fragment to non-root fragment.
76  root_fragment->heap_end = old_end; // used to hold mso_list when it was the root fragment
77  c->heap.root->next = root_fragment;
78  c->heap.root->mso_list = mso_list;
79  }
80  } else {
81  size_t maximum_free_space = 2 * (size + MIN_FREE_SPACE_SIZE);
82  if (free_space < size || (alloc_mode == MEMORY_FORCE_SHRINK) || ((alloc_mode == MEMORY_CAN_SHRINK) && free_space > maximum_free_space)) {
83  size_t memory_size = memory_heap_memory_size(&c->heap);
84  if (UNLIKELY(memory_gc(c, memory_size + size + MIN_FREE_SPACE_SIZE, num_roots, roots) != MEMORY_GC_OK)) {
85  // TODO: handle this more gracefully
86  TRACE("Unable to allocate memory for GC. memory_size=%zu size=%u\n", memory_size, size);
87  return MEMORY_GC_ERROR_FAILED_ALLOCATION;
88  }
89  if (alloc_mode != MEMORY_NO_SHRINK) {
90  size_t new_free_space = context_avail_free_memory(c);
91  if (new_free_space > maximum_free_space) {
92  size_t new_memory_size = memory_heap_memory_size(&c->heap);
93  size_t new_requested_size = (new_memory_size - new_free_space) + maximum_free_space;
94  if (!c->has_min_heap_size || (c->min_heap_size < new_requested_size)) {
95  if (UNLIKELY(memory_gc(c, new_requested_size, num_roots, roots) != MEMORY_GC_OK)) {
96  TRACE("Unable to allocate memory for GC shrink. new_memory_size=%zu new_free_space=%zu new_minimum_free_space=%zu size=%u\n", new_memory_size, new_free_space, maximum_free_space, size);
97  return MEMORY_GC_ERROR_FAILED_ALLOCATION;
98  }
99  }
100  }
101  }
102  }
103  }
104 
105  return MEMORY_GC_OK;
106 }

114 static enum MemoryGCResult memory_gc(Context *ctx, size_t new_size, size_t num_roots, term *roots)
115 {
116  TRACE("Going to perform gc on process %i\n", ctx->process_id);
117  size_t min_heap_size = ctx->has_min_heap_size ? ctx->min_heap_size : 0;
118  new_size = MAX(new_size, min_heap_size);
119 
120  if (UNLIKELY(ctx->has_max_heap_size && (new_size > ctx->max_heap_size))) {
121  return MEMORY_GC_DENIED_ALLOCATION;
122  }
123 
124  term old_mso_list = ctx->heap.root->mso_list;
125  term *old_stack_ptr = context_stack_base(ctx);
126  term *old_heap_end = ctx->heap.heap_end;
127  HeapFragment *old_root_fragment = ctx->heap.root;
128 
129  if (UNLIKELY(memory_init_heap(&ctx->heap, new_size) != MEMORY_GC_OK)) {
130  return MEMORY_GC_ERROR_FAILED_ALLOCATION;
131  }
132  // We need old heap fragment to only copy terms that were in the heap (as opposed to in messages)
133  old_root_fragment->heap_end = old_heap_end;
134 
135  term *new_heap = ctx->heap.heap_start;
136  TRACE("- Allocated %i words for new heap at address 0x%p\n", (int) new_size, (void *) new_heap);
137 
138  TRACE("- Running copy GC on registers\n");
139  for (int i = 0; i < MAX_REG; i++) {
140  term new_root = memory_shallow_copy_term(old_root_fragment, ctx->x[i], &ctx->heap.heap_ptr, true);
141  ctx->x[i] = new_root;
142  }
143  TRACE("- after registers, heap.heap_ptr now is at %p, heap.heap_start = %p\n", (void *) ctx->heap.heap_ptr, (void *) ctx->heap.heap_start);
144 
145  TRACE("- Running copy GC on stack (stack size: %i)\n", (int) (old_stack_ptr - ctx->e));
146  term *stack_ptr = new_heap + new_size;
147  while (old_stack_ptr > ctx->e) {
148  term new_root = memory_shallow_copy_term(old_root_fragment, *(--old_stack_ptr), &ctx->heap.heap_ptr, true);
149  push_to_stack(&stack_ptr, new_root);
150  }
151  ctx->e = stack_ptr;
152 
153  struct ListHead *item;
154  TRACE("- Running copy GC on process dictionary\n");
155  LIST_FOR_EACH (item, &ctx->dictionary) {
156  struct DictEntry *entry = GET_LIST_ENTRY(item, struct DictEntry, head);
157  entry->key = memory_shallow_copy_term(old_root_fragment, entry->key, &ctx->heap.heap_ptr, true);
158  entry->value = memory_shallow_copy_term(old_root_fragment, entry->value, &ctx->heap.heap_ptr, true);
159  }
160 
161  TRACE("- Running copy GC on exit reason\n");
162  ctx->exit_reason = memory_shallow_copy_term(old_root_fragment, ctx->exit_reason, &ctx->heap.heap_ptr, true);
163 
164  TRACE("- Running copy GC on provided roots\n");
165  for (size_t i = 0; i < num_roots; i++) {
166  roots[i] = memory_shallow_copy_term(old_root_fragment, roots[i], &ctx->heap.heap_ptr, 1);
167  }
168 
169  term *temp_start = new_heap;
170  term *temp_end = ctx->heap.heap_ptr;
171  term new_mso_list = term_nil();
172  do {
173  term *next_end = temp_end;
174  TRACE("- Running scan and copy GC from %p to %p\n", (void *) temp_start, (void *) temp_end);
175  memory_scan_and_copy(old_root_fragment, temp_start, temp_end, &next_end, &new_mso_list, true);
176  temp_start = temp_end;
177  temp_end = next_end;
178  } while (temp_start != temp_end);
179 
180  ctx->heap.heap_ptr = temp_end;
181 
182  memory_sweep_mso_list(old_mso_list, ctx->global);
183  ctx->heap.root->mso_list = new_mso_list;
184 
185  memory_destroy_heap_fragment(old_root_fragment);
186 
187  return MEMORY_GC_OK;
188 }

208 {
209  TRACE("Copy term tree: %p, heap_ptr: %p\n", (void *) t, (void *) *heap_ptr);
210 
211  term *temp_start = *heap_ptr;
212  term copied_term = memory_shallow_copy_term(NULL, t, heap_ptr, false);
213  term *temp_end = *heap_ptr;
214 
215  do {
216  term *next_end = temp_end;
217  memory_scan_and_copy(NULL, temp_start, temp_end, &next_end, mso_list, false);
218  temp_start = temp_end;
219  temp_end = next_end;
220  } while (temp_start != temp_end);
221 
222  *heap_ptr = temp_end;
223 
224  return copied_term;

241 unsigned long memory_estimate_usage(term t)
242 {
243  unsigned long acc = 0;
244 
245  struct TempStack temp_stack;
246  if (UNLIKELY(temp_stack_init(&temp_stack) != TempStackOk)) {
247  // TODO: handle failed malloc
248  AVM_ABORT();
249  }
250 
251  if (UNLIKELY(temp_stack_push(&temp_stack, t) != TempStackOk)) {
252  // TODO: handle failed malloc
253  AVM_ABORT();
254  }
255 
256  while (!temp_stack_is_empty(&temp_stack)) {
257  if (term_is_atom(t)) {
258  t = temp_stack_pop(&temp_stack);
259 
260  } else if (term_is_integer(t)) {
261  t = temp_stack_pop(&temp_stack);
262 
263  } else if (term_is_nil(t)) {
264  t = temp_stack_pop(&temp_stack);
265 
266  } else if (term_is_pid(t)) {
267  t = temp_stack_pop(&temp_stack);
268 
269  } else if (term_is_nonempty_list(t)) {
270  acc += 2;
271  if (UNLIKELY(temp_stack_push(&temp_stack, term_get_list_tail(t)) != TempStackOk)) {
272  // TODO: handle failed malloc
273  AVM_ABORT();
274  }
275  t = term_get_list_head(t);
276 
277  } else if (term_is_tuple(t)) {
278  int tuple_size = term_get_tuple_arity(t);
279  acc += tuple_size + 1;
280 
281  if (tuple_size > 0) {
282  for (int i = 1; i < tuple_size; i++) {
283  if (UNLIKELY(temp_stack_push(&temp_stack, term_get_tuple_element(t, i)) != TempStackOk)) {
284  // TODO: handle failed malloc
285  AVM_ABORT();
286  }
287  }
288  t = term_get_tuple_element(t, 0);
289 
290  } else {
291  t = term_nil();
292  }
293 
294  } else if (term_is_map(t)) {
295  int map_size = term_get_map_size(t);
296  acc += term_map_size_in_terms(map_size);
297  if (map_size > 0) {
298  for (int i = 1; i < map_size; i++) {
299  if (UNLIKELY(temp_stack_push(&temp_stack, term_get_map_key(t, i)) != TempStackOk)) {
300  // TODO: handle failed malloc
301  AVM_ABORT();
302  }
303  if (UNLIKELY(temp_stack_push(&temp_stack, term_get_map_value(t, i)) != TempStackOk)) {
304  // TODO: handle failed malloc
305  AVM_ABORT();
306  }
307  }
308  if (UNLIKELY(temp_stack_push(&temp_stack, term_get_map_value(t, 0)) != TempStackOk)) {
309  // TODO: handle failed malloc
310  AVM_ABORT();
311  }
312  t = term_get_map_key(t, 0);
313 
314  } else {
315  t = term_nil();
316  }
317 
318  } else if (term_is_boxed(t)) {
319  acc += term_boxed_size(t) + 1;
320  if (term_is_sub_binary(t)) {
321  t = term_get_sub_binary_ref(t);
322  } else {
323  t = temp_stack_pop(&temp_stack);
324  }
325 
326  } else {
327  fprintf(stderr, "bug: found unknown term type: 0x%" TERM_X_FMT "\n", t);
328  if (term_is_boxed(t)) {
329  const term *boxed_value = term_to_const_term_ptr(t);
330  int boxed_size = term_boxed_size(t) + 1;
331  fprintf(stderr, "boxed header: 0x%" TERM_X_FMT ", size: %i\n", boxed_value[0], boxed_size);
332  }
333  AVM_ABORT();
334  }
335  }
336 
337  temp_stack_destroy(&temp_stack);
338 
339  return acc;
340 }

342 static void memory_scan_and_copy(HeapFragment *old_fragment, term *mem_start, const term *mem_end, term **new_heap_pos, term *mso_list, bool move)
343 {
344  term *ptr = mem_start;
345  term *new_heap = *new_heap_pos;
346 
347  while (ptr < mem_end) {
348  term t = *ptr;
349 
350  if (term_is_atom(t)) {
351  TRACE("Found atom (%" TERM_X_FMT ")\n", t);
352  ptr++;
353 
354  } else if (term_is_integer(t)) {
355  TRACE("Found integer (%" TERM_X_FMT ")\n", t);
356  ptr++;
357 
358  } else if (term_is_nil(t)) {
359  TRACE("Found NIL (%" TERM_X_FMT ")\n", t);
360  ptr++;
361 
362  } else if (term_is_pid(t)) {
363  TRACE("Found PID (%" TERM_X_FMT ")\n", t);
364  ptr++;
365 
366  } else if ((t & 0x3) == 0x0) {
367  TRACE("Found boxed header (%" TERM_X_FMT ")\n", t);
368 
369  switch (t & TERM_BOXED_TAG_MASK) {
370  case TERM_BOXED_TUPLE: {
371  int arity = term_get_size_from_boxed_header(t);
372  TRACE("- Boxed is tuple (%" TERM_X_FMT "), arity: %i\n", t, arity);
373 
374  for (int i = 1; i <= arity; i++) {
375  TRACE("-- Elem: %" TERM_X_FMT "\n", ptr[i]);
376  ptr[i] = memory_shallow_copy_term(old_fragment, ptr[i], &new_heap, move);
377  }
378  break;
379  }
380 
381  case TERM_BOXED_BIN_MATCH_STATE: {
382  TRACE("- Found bin match state.\n");
383  ptr[1] = memory_shallow_copy_term(old_fragment, ptr[1], &new_heap, move);
384  break;
385  }
386 
387  case TERM_BOXED_POSITIVE_INTEGER:
388  TRACE("- Found boxed pos int.\n");
389  break;
390 
391  case TERM_BOXED_REF:
392  TRACE("- Found ref.\n");
393  break;
394 
395  case TERM_BOXED_FUN: {
396  int fun_size = term_get_size_from_boxed_header(t);
397  TRACE("- Found fun, size: %i.\n", fun_size);
398 
399  // first term is the boxed header, followed by module and fun index.
400 
401  for (int i = 3; i <= fun_size; i++) {
402  TRACE("-- Frozen: %" TERM_X_FMT "\n", ptr[i]);
403  ptr[i] = memory_shallow_copy_term(old_fragment, ptr[i], &new_heap, move);
404  }
405  break;
406  }
407 
408  case TERM_BOXED_FLOAT:
409  TRACE("- Found float.\n");
410  break;
411 
412  case TERM_BOXED_REFC_BINARY: {
413  TRACE("- Found refc binary.\n");
414  term ref = ((term) ptr) | TERM_BOXED_VALUE_TAG;
415  if (!term_refc_binary_is_const(ref)) {
416  *mso_list = term_list_init_prepend(ptr + REFC_BINARY_CONS_OFFET, ref, *mso_list);
417  }
418  break;
419  }
420 
421  case TERM_BOXED_SUB_BINARY: {
422  TRACE("- Found sub binary.\n");
423  ptr[3] = memory_shallow_copy_term(old_fragment, ptr[3], &new_heap, move);
424  break;
425  }
426 
427  case TERM_BOXED_HEAP_BINARY:
428  TRACE("- Found binary.\n");
429  break;
430 
431  case TERM_BOXED_MAP: {
432  TRACE("- Found map.\n");
433  size_t map_size = term_get_size_from_boxed_header(t) - 1;
434  size_t keys_offset = term_get_map_keys_offset();
435  size_t value_offset = term_get_map_value_offset();
436  TRACE("-- Map keys: %" TERM_X_FMT "\n", ptr[keys_offset]);
437  ptr[keys_offset] = memory_shallow_copy_term(old_fragment, ptr[keys_offset], &new_heap, move);
438  for (size_t i = value_offset; i < value_offset + map_size; ++i) {
439  TRACE("-- Map Value: %" TERM_X_FMT "\n", ptr[i]);
440  ptr[i] = memory_shallow_copy_term(old_fragment, ptr[i], &new_heap, move);
441  }
442  } break;
443 
444  default:
445  fprintf(stderr, "- Found unknown boxed type: %" TERM_X_FMT "\n", (t >> 2) & 0xF);
446  AVM_ABORT();
447  }
448 
449  ptr += term_get_size_from_boxed_header(t) + 1;
450 
451  } else if (term_is_nonempty_list(t)) {
452  TRACE("Found nonempty list (%p)\n", (void *) t);
453  *ptr = memory_shallow_copy_term(old_fragment, t, &new_heap, move);
454  ptr++;
455 
456  } else if (term_is_boxed(t)) {
457  TRACE("Found boxed (%p)\n", (void *) t);
458  *ptr = memory_shallow_copy_term(old_fragment, t, &new_heap, move);
459  ptr++;
460 
461  } else {
462  fprintf(stderr, "bug: found unknown term type: 0x%" TERM_X_FMT "\n", t);
463  AVM_ABORT();
464  }
465  }
466 
467  *new_heap_pos = new_heap;
468 }

481 HOT_FUNC static term memory_shallow_copy_term(HeapFragment *old_fragment, term t, term **new_heap, bool move)
482 {
483  if (term_is_atom(t)) {
484  return t;
485 
486  } else if (term_is_integer(t)) {
487  return t;
488 
489  } else if (term_is_nil(t)) {
490  return t;
491 
492  } else if (term_is_pid(t)) {
493  return t;
494 
495  } else if (term_is_cp(t)) {
496  // CP is valid only on stack
497  return t;
498 
499  } else if (term_is_catch_label(t)) {
500  // catch label is valid only on stack
501  return t;
502 
503  } else if (term_is_boxed(t)) {
504  term *boxed_value = term_to_term_ptr(t);
505  // Do not GC terms from messages until the message is destroyed
506  if (old_fragment != NULL && !memory_heap_fragment_contains_pointer(old_fragment, boxed_value)) {
507  return t;
508  }
509 
510  if (memory_is_moved_marker(boxed_value)) {
511  return memory_dereference_moved_marker(boxed_value);
512  }
513 
514  int boxed_size = term_boxed_size(t) + 1;
515 
516  // It must be an empty tuple, so we are not going to use moved markers.
517  // Empty tuples memory is too small to store moved markers.
518  // However it is also required to avoid boxed terms duplication.
519  // So instead all empty tuples will reference the same boxed term.
520  if (boxed_size == 1) {
521  return ((term) &empty_tuple) | TERM_BOXED_VALUE_TAG;
522  }
523 
524  term *dest = *new_heap;
525  for (int i = 0; i < boxed_size; i++) {
526  dest[i] = boxed_value[i];
527  }
528  *new_heap += boxed_size;
529 
530  term new_term = ((term) dest) | TERM_BOXED_VALUE_TAG;
531 
532  if (move) {
533  memory_replace_with_moved_marker(boxed_value, new_term);
534  } else if (term_is_refc_binary(t)) { // copy, not a move; increment refcount
535  if (!term_refc_binary_is_const(t)) {
536  refc_binary_increment_refcount((struct RefcBinary *) term_refc_binary_ptr(t));
537  }
538  }
539 
540  return new_term;
541 
542  } else if (term_is_nonempty_list(t)) {
543  term *list_ptr = term_get_list_ptr(t);
544  if (old_fragment != NULL && !memory_heap_fragment_contains_pointer(old_fragment, list_ptr)) {
545  return t;
546  }
547 
548  if (memory_is_moved_marker(list_ptr)) {
549  return memory_dereference_moved_marker(list_ptr);
550  }
551 
552  term *dest = *new_heap;
553  dest[0] = list_ptr[0];
554  dest[1] = list_ptr[1];
555  *new_heap += 2;
556 
557  term new_term = ((term) dest) | 0x1;
558 
559  if (move) {
560  memory_replace_with_moved_marker(list_ptr, new_term);
561  }
562 
563  return new_term;
564 
565  } else {
566  fprintf(stderr, "Unexpected term. Term is: %" TERM_X_FMT "\n", t);
567  AVM_ABORT();
568  }
569 }

571 void memory_heap_append_fragment(Heap *heap, HeapFragment *fragment, term mso_list)
572 {
573  // The fragment we are appending may have next fragments
574  // So we take our current next and we add it to the tail of the passed list
575  if (heap->root->next) {
576  HeapFragment *tail = fragment;
577  while (tail->next != NULL) {
