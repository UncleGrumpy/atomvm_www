101 char *interop_list_to_string(term list, int *ok)
102 {
103  int proper;
104  int len = term_list_length(list, &proper);
105  if (UNLIKELY(!proper)) {
106  *ok = 0;
107  return NULL;
108  }
109 
110  char *str = malloc(len + 1);
111  if (IS_NULL_PTR(str)) {
112  return NULL;
113  }
114 
115  term t = list;
116  for (int i = 0; i < len; i++) {
117  term byte_value_term = term_get_list_head(t);
118  if (UNLIKELY(!term_is_integer(byte_value_term))) {
119  *ok = 0;
120  free(str);
121  return NULL;
122  }
123 
124  if (UNLIKELY(!term_is_uint8(byte_value_term))) {
125  *ok = 0;
126  free(str);
127  return NULL;
128  }
129  uint8_t byte_value = term_to_uint8(byte_value_term);
130 
131  str[i] = (char) byte_value;
132  t = term_get_list_tail(t);
133  }
134  str[len] = 0;
135 
136  *ok = 1;
137  return str;
138 }

160 term interop_proplist_get_value_default(term list, term key, term default_value)
161 {
162  term t = list;
163 
164  while (!term_is_nil(t)) {
165  term *t_ptr = term_get_list_ptr(t);
166 
167  term head = t_ptr[1];
168  if (term_is_tuple(head) && term_get_tuple_arity(head) == 2 && term_get_tuple_element(head, 0) == key) {
169  return term_get_tuple_element(head, 1);
170 
171  } else if (term_is_atom(head)) {
172  if (head == key) {
173  return TRUE_ATOM;
174  }
175  }
176 
177  t = *t_ptr;
178  }
179 
180  return default_value;
181 }

183 inline InteropFunctionResult interop_chardata_fold(term t, interop_chardata_fold_fun fold_fun, interop_chardata_rest_fun rest_fun, void *accum)
184 {
185  if (term_is_binary(t)) {
186  return fold_fun(t, accum);
187  }
188 
189  if (UNLIKELY(!term_is_list(t))) {
190  return InteropBadArg;
191  }
192 
193  struct TempStack temp_stack;
194  if (UNLIKELY(temp_stack_init(&temp_stack) != TempStackOk)) {
195  return InteropMemoryAllocFail;
196  }
197 
198  if (UNLIKELY(temp_stack_push(&temp_stack, t) != TempStackOk)) {
199  temp_stack_destroy(&temp_stack);
200  return InteropMemoryAllocFail;
201  }
202 
203  while (!temp_stack_is_empty(&temp_stack)) {
204  if (term_is_integer(t) || term_is_binary(t)) {
205  InteropFunctionResult result = fold_fun(t, accum);
206  if (UNLIKELY(result != InteropOk)) {
207  if (rest_fun) {
208  // we don't pass failed element, fold_fun handles it
209  t = temp_stack_pop(&temp_stack);
210  while (!temp_stack_is_empty(&temp_stack)) {
211  rest_fun(t, accum);
212  t = temp_stack_pop(&temp_stack);
213  }
214  }
215  // we don't process last element either which is the original list
216  temp_stack_destroy(&temp_stack);
217  return result;
218  } else {
219  t = temp_stack_pop(&temp_stack);
220  }
221 
222  } else if (term_is_nil(t)) {
223  t = temp_stack_pop(&temp_stack);
224 
225  } else if (term_is_nonempty_list(t)) {
226  if (UNLIKELY(temp_stack_push(&temp_stack, term_get_list_tail(t)) != TempStackOk)) {
227  temp_stack_destroy(&temp_stack);
228  return InteropMemoryAllocFail;
229  }
230  t = term_get_list_head(t);
231 
232  } else {
233  if (rest_fun) {
234  while (!temp_stack_is_empty(&temp_stack)) {
235  rest_fun(t, accum);
236  t = temp_stack_pop(&temp_stack);
237  }
238  // we don't process last element which was the passed term
239  }
240  temp_stack_destroy(&temp_stack);
241  return InteropBadArg;
242  }
243  }
244 
245  temp_stack_destroy(&temp_stack);
246 
247  return InteropOk;
248 }

286 static enum UnicodeConversionResult interop_binary_conversion(term t, uint8_t *output, size_t *output_len, size_t *rest_crsr, enum CharDataEncoding in_encoding, enum CharDataEncoding out_encoding)
287 {
288  size_t len = term_binary_size(t);
289  if (in_encoding == Latin1Encoding && out_encoding == Latin1Encoding) {
290  if (output) {
291  memcpy(output, term_binary_data(t), len);
292  }
293  *output_len = len;
294  return UnicodeOk;
295  }
296  size_t result = 0;
297  size_t input_index;
298  const uint8_t *input = (const uint8_t *) term_binary_data(t);
299  if (in_encoding == Latin1Encoding) {
300  for (input_index = 0; input_index < len; input_index++) {
301  if (out_encoding == UTF8Encoding) {
302  size_t char_size;
303  if (UNLIKELY(!bitstring_utf8_encode(input[input_index], output, &char_size))) {
304  *rest_crsr = input_index;
305  *output_len = result;
306  return UnicodeError;
307  }
308  result += char_size;
309  if (output) {
310  output += char_size;
311  }
312  } else {
313  // UCS4Native
314  result += sizeof(uint32_t);
315  if (output) {
316  *((uint32_t *) output) = input[input_index];
317  output += sizeof(uint32_t);
318  }
319  }
320  }
321  *output_len = result;
322  return UnicodeOk;
323  }
324  input_index = 0;
325  while (input_index < len) {
326  size_t char_size;
327  uint32_t c;
328  enum UnicodeTransformDecodeResult decode_result = bitstring_utf8_decode(input + input_index, len - input_index, &c, &char_size);
329  if (UNLIKELY(decode_result != UnicodeTransformDecodeSuccess)) {
330  *rest_crsr = input_index;
331  *output_len = result;
332  return decode_result == UnicodeTransformDecodeIncomplete ? UnicodeIncompleteTransform : UnicodeError;
333  }
334  switch (out_encoding) {
335  case Latin1Encoding: {
336  if (c > 255) {
337  *rest_crsr = input_index;
338  *output_len = result;
339  return UnicodeError;
340  }
341  if (output) {
342  *output++ = c;
343  }
344  result++;
345  } break;
346  case UTF8Encoding: {
347  if (output) {
348  memcpy(output, input + input_index, char_size);
349  output += char_size;
350  }
351  result += char_size;
352  } break;
353  case UCS4NativeEncoding: {
354  if (output) {
355  *((uint32_t *) output) = c;
356  output += sizeof(uint32_t);
357  }
358  result += sizeof(uint32_t);
359  } break;
360  }
361  input_index += char_size;
362  }
363  *output_len = result;
364  return UnicodeOk;
365 }

377 static InteropFunctionResult chardata_to_bytes_size_fold_fun(term t, void *accum)
378 {
379  struct CharDataToBytesSizeAcc *acc = (struct CharDataToBytesSizeAcc *) accum;
380  if (term_is_binary(t)) {
381  size_t bin_size;
382  size_t rest_crsr;
383  enum UnicodeConversionResult conv_result = interop_binary_conversion(t, NULL, &bin_size, &rest_crsr, acc->in_encoding, acc->out_encoding);
384  acc->size += bin_size;
385  if (UNLIKELY(conv_result != UnicodeOk)) {
386  acc->rest_size = term_sub_binary_heap_size(t, term_binary_size(t) - rest_crsr);
387  acc->incomplete_transform = conv_result == UnicodeIncompleteTransform;
388  return InteropBadArg;
389  }
390  } else /* term_is_integer(t) */ {
391  avm_int_t c = term_to_int(t);
392  if (c < 0 || c > UNICODE_CHAR_MAX) {
393  return InteropBadArg;
394  }
395  switch (acc->out_encoding) {
396  case Latin1Encoding: {
397  if (c > 255) {
398  return InteropBadArg;
399  }
400  acc->size++;
401  } break;
402  case UTF8Encoding: {
403  size_t char_size;
404  if (UNLIKELY(!bitstring_utf8_encode((uint32_t) c, NULL, &char_size))) {
405  return InteropBadArg;
406  }
407  acc->size += char_size;
408  } break;
409  case UCS4NativeEncoding: {
410  acc->size += sizeof(uint32_t);
411  } break;
412  }
413  }
414  return InteropOk;
415 }

469 static InteropFunctionResult chardata_to_bytes_fold_fun(term t, void *accum)
470 {
471  struct CharDataToBytesAcc *acc = (struct CharDataToBytesAcc *) accum;
472  if (term_is_binary(t)) {
473  size_t bin_size;
474  size_t rest_crsr;
475  enum UnicodeConversionResult conv_result = interop_binary_conversion(t, acc->output, &bin_size, &rest_crsr, acc->in_encoding, acc->out_encoding);
476  acc->output += bin_size;
477  if (UNLIKELY(conv_result != UnicodeOk)) {
478  if (acc->rest) {
479  *acc->rest = term_alloc_sub_binary(t, rest_crsr, term_binary_size(t) - rest_crsr, acc->heap);
480  }
481  if (conv_result == UnicodeIncompleteTransform) {
482  acc->incomplete_transform = true;
483  }
484  return InteropBadArg;
485  }
486  } else /* term_is_integer(t) */ {
487  avm_int_t c = term_to_int(t);
488  if (c < 0 || c > UNICODE_CHAR_MAX) {
489  if (acc->rest) {
490  *acc->rest = t;
491  }
492  return InteropBadArg;
493  }
494  switch (acc->out_encoding) {
495  case Latin1Encoding: {
496  if (c > 255) {
497  if (acc->rest) {
498  *acc->rest = t;
499  }
500  return InteropBadArg;
501  }
502  *acc->output++ = (uint8_t) c;
503  } break;
504  case UTF8Encoding: {
505  size_t char_size;
506  if (UNLIKELY(!bitstring_utf8_encode((uint32_t) c, acc->output, &char_size))) {
507  if (acc->rest) {
508  *acc->rest = t;
509  }
510  return InteropBadArg;
511  }
512  acc->output += char_size;
513  } break;
514  case UCS4NativeEncoding: {
515  // Special case: if input encoding is latin1, only accept characters < 256.
516  if (acc->in_encoding == Latin1Encoding && c > 255) {
517  if (acc->rest) {
518  *acc->rest = t;
519  }
520  return InteropBadArg;
521  }
522  *((uint32_t *) acc->output) = (uint32_t) c;
523  acc->output += sizeof(uint32_t);
524  } break;
525  }
526  }
527  return InteropOk;
528 }
