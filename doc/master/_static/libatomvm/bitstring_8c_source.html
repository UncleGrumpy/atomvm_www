29 bool bitstring_extract_any_integer(const uint8_t *src, size_t offset, avm_int_t n,
30  enum BitstringFlags bs_flags, union maybe_unsigned_int64 *dst)
31 {
32  uint64_t out = 0;
33 
34  int i;
35  for (i = 0; i < n; i++) {
36  int bit_pos = offset + i;
37  int byte_pos = bit_pos >> 3;
38  int shift = 7 - (bit_pos & 7);
39 
40  if (src[byte_pos] & (1 << shift)) {
41  out |= (uint64_t) 1 << (n - i - 1);
42  }
43  }
44 
45  if (bs_flags & LittleEndianIntegerMask) {
46  out = from_le64(out) >> (64 - n);
47  }
48 
49  if ((bs_flags & SignedInteger) && (out & ((uint64_t) 1) << (i - 1))) {
50  dst->u = ((uint64_t) 0xFFFFFFFFFFFFFFFF << i) | out;
51  } else {
52  dst->u = out;
53  }
54 
55  return true;
56 }

58 bool bitstring_insert_any_integer(uint8_t *dst, avm_int_t offset, avm_int64_t value, size_t n, enum BitstringFlags bs_flags)
59 {
60  // TODO support big/little endian signedness flags
61  if (bs_flags != 0) {
62  return false;
63  }
64  // value is truncated to 64 bits
65  if (n > 8 * sizeof(value)) {
66  offset += n - (8 * sizeof(value));
67  n = 8 * sizeof(value);
68  }
69  for (size_t i = 0; i < n; ++i) {
70  size_t k = (n - 1) - i;
71  int bit_val = (value & (0x01LL << k)) >> k;
72  if (bit_val) {
73  int bit_pos = offset + i;
74  int byte_pos = bit_pos >> 3; // div 8
75  uint8_t *pos = (uint8_t *) (dst + byte_pos);
76  int shift = 7 - (bit_pos & 7); // mod 8
77  *pos ^= (0x01 << shift);
78  }
79  }
80  return true;
81 }

104 bool bitstring_utf8_encode(uint32_t c, uint8_t *buf, size_t *out_size)
105 {
106  size_t sz = 0;
107  if (is_invalid_codepoint(c)) {
108  return false;
109  }
110  if (c < 0x80) {
111  if (buf) {
112  *buf++ = c;
113  }
114  sz++;
115  } else if (c < 0x800) {
116  if (buf) {
117  *buf++ = (c >> 6) | 0xC0;
118  *buf++ = (c & 0x3F) | 0x80;
119  }
120  sz += 2;
121  } else if (c < 0x10000) {
122  if (buf) {
123  *buf++ = (c >> 12) | 0xE0;
124  *buf++ = ((c >> 6) & 0x3F) | 0x80;
125  *buf++ = (c & 0x3F) | 0x80;
126  }
127  sz += 3;
128  } else {
129  if (buf) {
130  *buf++ = (c >> 18) | 0xF0;
131  *buf++ = ((c >> 12) & 0x3F) | 0x80;
132  *buf++ = ((c >> 6) & 0x3F) | 0x80;
133  *buf++ = (c & 0x3F) | 0x80;
134  }
135  sz += 4;
136  }
137  *out_size = sz;
138  return true;
139 }

141 enum UnicodeTransformDecodeResult bitstring_utf8_decode(const uint8_t *buf, size_t len, uint32_t *c, size_t *out_size)
142 {
143  if (len == 0) {
144  return UnicodeTransformDecodeFail;
145  } else if (len >= 4 && (buf[0] & 0xF8) == 0xF0 && ((buf[1] & 0xC0) == 0x80) && ((buf[2] & 0xC0) == 0x80) && ((buf[3] & 0xC0) == 0x80)) {
146  uint32_t v = 0;
147  v |= (buf[0] & 0x07) << 18;
148  v |= (buf[1] & 0x3F) << 12;
149  v |= (buf[2] & 0x3F) << 6;
150  v |= (buf[3] & 0x3F);
151  // overlong encoding or invalid codepoint
152  if (v <= 0x10000 || v > 0x10FFFF) {
153  return false;
154  }
155  *c = v;
156  *out_size = 4;
157  return UnicodeTransformDecodeSuccess;
158  } else if (len >= 3 && (buf[0] & 0xF0) == 0xE0 && ((buf[1] & 0xC0) == 0x80) && ((buf[2] & 0xC0) == 0x80)) {
159  uint32_t v = 0;
160  v |= (buf[0] & 0x0F) << 12;
161  v |= (buf[1] & 0x3F) << 6;
162  v |= (buf[2] & 0x3F);
163  // overlong encoding or surrogate
164  if (v < 0x800 || (v >= 0xD800 && v <= 0xDFFF)) {
165  return false;
166  }
167  *c = v;
168  *out_size = 3;
169  return UnicodeTransformDecodeSuccess;
170  } else if (len >= 2 && (buf[0] & 0xE0) == 0xC0 && ((buf[1] & 0xC0) == 0x80)) {
171  uint32_t v = 0;
172  v |= (buf[0] & 0x1F) << 6;
173  v |= (buf[1] & 0x3F);
174  // overlong encoding
175  if (v < 0x80) {
176  return false;
177  }
178  *c = v;
179  *out_size = 2;
180  return UnicodeTransformDecodeSuccess;
181  } else if ((*buf & 0x80) == 0) {
182  uint32_t v = 0;
183  v |= (buf[0] & 0x7F);
184  *c = v;
185  *out_size = 1;
186  return UnicodeTransformDecodeSuccess;
187  } else if (len == 3 && (buf[0] & 0xF8) == 0xF0 && ((buf[1] & 0xC0) == 0x80) && ((buf[2] & 0xC0) == 0x80)) {
188  return UnicodeTransformDecodeIncomplete;
189  } else if (len == 2 && (buf[0] & 0xF8) == 0xF0 && ((buf[1] & 0xC0) == 0x80)) {
190  return UnicodeTransformDecodeIncomplete;
191  } else if (len == 1 && (buf[0] & 0xF8) == 0xF0) {
192  return UnicodeTransformDecodeIncomplete;
193  } else if (len == 2 && (buf[0] & 0xF0) == 0xE0 && ((buf[1] & 0xC0) == 0x80)) {
194  return UnicodeTransformDecodeIncomplete;
195  } else if (len == 1 && (buf[0] & 0xF0) == 0xE0) {
196  return UnicodeTransformDecodeIncomplete;
197  } else if (len == 1 && (buf[0] & 0xE0) == 0xC0) {
198  return UnicodeTransformDecodeIncomplete;
199  }
200 
201  return UnicodeTransformDecodeFail;
202 }

210 bool bitstring_utf16_encode(uint32_t c, uint8_t *buf, enum BitstringFlags bs_flags, size_t *out_size)
211 {
212  size_t sz = 0;
213  if (is_invalid_codepoint(c)) {
214  return false;
215  }
216  if (c < 0x10000) {
217  // Ignore D800-DFFF range
218  if (buf) {
219  if (bs_flags & LittleEndianIntegerMask) {
220  *buf++ = c & 0xFF;
221  *buf++ = c >> 8;
222  } else {
223  *buf++ = c >> 8;
224  *buf++ = c & 0xFF;
225  }
226  }
227  sz += 2;
228  } else {
229  if (buf) {
230  c -= 0x10000;
231  if (bs_flags & LittleEndianIntegerMask) {
232  *buf++ = ((c >> 10) & 0xFF);
233  *buf++ = (c >> 18) | 0xD8;
234  *buf++ = c & 0xFF;
235  *buf++ = ((c >> 8) & 0x03) | 0xDC;
236  } else {
237  *buf++ = (c >> 18) | 0xD8;
238  *buf++ = ((c >> 10) & 0xFF);
239  *buf++ = ((c >> 8) & 0x03) | 0xDC;
240  *buf++ = c & 0xFF;
241  }
242  }
243  sz += 4;
244  }
245  *out_size = sz;
246  return true;

354 void bitstring_copy_bits_incomplete_bytes(uint8_t *dst, size_t bits_offset, const uint8_t *src, size_t bits_count)
355 {
356  size_t byte_offset = bits_offset / 8;
357  size_t bit_offset = bits_offset - (8 * byte_offset);
358  if (bits_offset % 8 == 0 && bits_count >= 8) {
359  size_t bytes_count = bits_count / 8;
360  memcpy(dst + byte_offset, src, bytes_count);
361  src += bytes_count;
362  byte_offset += bytes_count;
363  bits_count -= bytes_count * 8;
364  }
365  // Eventually copy bit by bit
366  dst += byte_offset;
367  uint8_t dest_byte = *dst;
368  uint8_t src_byte = *src++;
369  int dest_bit_ix = 7 - bit_offset;
370  int src_bit_ix = 7;
371  while (bits_count > 0) {
372  if (src_byte & (1 << src_bit_ix)) {
373  dest_byte |= 1 << dest_bit_ix;
374  } else {
375  dest_byte &= ~(1 << dest_bit_ix);
376  }
377  if (dest_bit_ix == 0) {
378  *dst++ = dest_byte;
379  dest_byte = *dst;
380  dest_bit_ix = 8;
381  }
382  if (src_bit_ix == 0) {
383  src_byte = *src++;
384  src_bit_ix = 8;
385  }
386  dest_bit_ix--;
387  src_bit_ix--;
388  bits_count--;
389  }
390  *dst = dest_byte;
391 }
