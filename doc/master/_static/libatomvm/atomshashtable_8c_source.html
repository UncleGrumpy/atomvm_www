37 static unsigned long sdbm_hash(const unsigned char *str, int len)
38 {
39  unsigned long hash = 0;
40  int c;
41 
42  for (int i = 0; i < len; i++) {
43  c = *str++;
44  hash = c + (hash << 6) + (hash << 16) - hash;
45  }
46 
47  return hash;
48 }

68 int atomshashtable_insert(struct AtomsHashTable *hash_table, AtomString string, unsigned long value)
69 {
70  int alen = atom_string_len(string);
71 
72  unsigned long hash = sdbm_hash(string, alen);
73  long index = hash % hash_table->capacity;
74 
75  struct HNode *node = hash_table->buckets[index];
76  if (node) {
77  while (1) {
78  if (atom_are_equals(string, node->key)) {
79  node->value = value;
80  return 1;
81  }
82 
83  if (node->next) {
84  node = node->next;
85  } else {
86  break;
87  }
88  }
89  }
90

92  if (IS_NULL_PTR(new_node)) {
93  return 0;
94  }
95  new_node->next = NULL;
96  new_node->key = string;
97  new_node->value = value;
98 
99  if (node) {
100  node->next = new_node;
101  } else {
102  hash_table->buckets[index] = new_node;
103  }
104 
105  hash_table->count++;
106  return 1;
107 }
