42 GlobalContext *globalcontext_new()
43 {
44  GlobalContext *glb = malloc(sizeof(GlobalContext));
45  if (IS_NULL_PTR(glb)) {
46  return NULL;
47  }
48  list_init(&glb->ready_processes);
49  list_init(&glb->waiting_processes);
50  list_init(&glb->avmpack_data);
51  list_init(&glb->refc_binaries);
52  list_init(&glb->processes_table);
53  glb->registered_processes = NULL;
54 
55  glb->last_process_id = 0;
56 
57  glb->atoms_table = atomshashtable_new();
58  if (IS_NULL_PTR(glb->atoms_table)) {
59  free(glb);
60  return NULL;
61  }
62  glb->atoms_ids_table = valueshashtable_new();
63  if (IS_NULL_PTR(glb->atoms_ids_table)) {
64  free(glb->atoms_table);
65  free(glb);
66  return NULL;
67  }
68 
69  defaultatoms_init(glb);
70 
71  glb->modules_by_index = NULL;
72  glb->loaded_modules_count = 0;
73  glb->modules_table = atomshashtable_new();
74  if (IS_NULL_PTR(glb->modules_table)) {
75  free(glb->atoms_ids_table);
76  free(glb->atoms_table);
77  free(glb);
78  return NULL;
79  }
80 
81  glb->timer_wheel = timer_wheel_new(16);
82  if (IS_NULL_PTR(glb->timer_wheel)) {
83  free(glb->modules_table);
84  free(glb->atoms_ids_table);
85  free(glb->atoms_table);
86  free(glb);
87  return NULL;
88  }
89  glb->last_seen_millis = 0;
90 
91  glb->ref_ticks = 0;
92 
93  sys_init_platform(glb);
94 
95  return glb;
96 }

138 bool globalcontext_unregister_process(GlobalContext *glb, int atom_index)
139 {
140  if (!glb->registered_processes) {
141  return false;
142  }
143 
144  struct RegisteredProcess *registered_processes = GET_LIST_ENTRY(glb->registered_processes, struct RegisteredProcess, registered_processes_list_head);
145 
146  struct RegisteredProcess *p = registered_processes;
147 
148  do {
149  if (p->atom_index == atom_index) {
150  linkedlist_remove(&glb->registered_processes, &p->registered_processes_list_head);
151  free(p);
152  return true;
153  }
154  p = GET_LIST_ENTRY(p->registered_processes_list_head.next, struct RegisteredProcess, registered_processes_list_head);
155  } while (p != registered_processes);
156 
157  return false;
158 }

185 int globalcontext_insert_atom_maybe_copy(GlobalContext *glb, AtomString atom_string, int copy)
186 {
187  struct AtomsHashTable *htable = glb->atoms_table;
188 
189  unsigned long atom_index = atomshashtable_get_value(htable, atom_string, ULONG_MAX);
190  if (atom_index == ULONG_MAX) {
191  if (copy) {
192  uint8_t len = *((uint8_t *) atom_string);
193  uint8_t *buf = malloc(1 + len);
194  if (UNLIKELY(IS_NULL_PTR(buf))) {
195  fprintf(stderr, "Unable to allocate memory for atom string\n");
196  AVM_ABORT();
197  }
198  memcpy(buf, atom_string, 1 + len);
199  atom_string = buf;
200  }
201  atom_index = htable->count;
202  if (!atomshashtable_insert(htable, atom_string, atom_index)) {
203  return -1;
204  }
205  if (!valueshashtable_insert(glb->atoms_ids_table, atom_index, (unsigned long) atom_string)) {
206  return -1;
207  }
208  }
209 
210  return (int) atom_index;
211 }

241 int globalcontext_insert_module(GlobalContext *global, Module *module)
242 {
243  AtomString module_name_atom = module_get_atom_string_by_id(module, 1);
244  if (!atomshashtable_insert(global->modules_table, module_name_atom, TO_ATOMSHASHTABLE_VALUE(module))) {
245  return -1;
246  }
247 
248  int module_index = global->loaded_modules_count;
249 
250  Module **new_modules_by_index = calloc(module_index + 1, sizeof(Module *));
251  if (IS_NULL_PTR(new_modules_by_index)) {
252  fprintf(stderr, "Failed to allocate memory: %s:%i.\n", __FILE__, __LINE__);
253  AVM_ABORT();
254  }
255  if (global->modules_by_index) {
256  for (int i = 0; i < module_index; i++) {
257  new_modules_by_index[i] = global->modules_by_index[i];
258  }
259  free(global->modules_by_index);
260  }
261 
262  module->module_index = module_index;
263 
264  global->modules_by_index = new_modules_by_index;
265  global->modules_by_index[module_index] = module;
266  global->loaded_modules_count++;
267 
268  return module_index;
269 }

271 Module *globalcontext_get_module(GlobalContext *global, AtomString module_name_atom)
272 {
273  Module *found_module = (Module *) atomshashtable_get_value(global->modules_table, module_name_atom, (unsigned long) NULL);
274 
275  if (!found_module) {
276  char *module_name = malloc(256 + 5);
277  if (IS_NULL_PTR(module_name)) {
278  return NULL;
279  }
280 
281  atom_string_to_c(module_name_atom, module_name, 256);
282  strcat(module_name, ".beam");
283  Module *loaded_module = sys_load_module(global, module_name);
284  free(module_name);
285 
286  if (UNLIKELY(!loaded_module || (globalcontext_insert_module(global, loaded_module) < 0))) {
287  return NULL;
288  }
289 
290  return loaded_module;
291  }
292 
293  return found_module;
294 }
