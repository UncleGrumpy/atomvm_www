66 GlobalContext *globalcontext_new()
67 {
68  GlobalContext *glb = malloc(sizeof(GlobalContext));
69  if (IS_NULL_PTR(glb)) {
70  return NULL;
71  }
72  list_init(&glb->ready_processes);
73  list_init(&glb->running_processes);
74  list_init(&glb->waiting_processes);
75 #ifndef AVM_NO_SMP
76  smp_spinlock_init(&glb->processes_spinlock);
77 #endif
78  synclist_init(&glb->avmpack_data);
79  synclist_init(&glb->refc_binaries);
80  synclist_init(&glb->processes_table);
81  synclist_init(&glb->registered_processes);
82  synclist_init(&glb->listeners);
83  synclist_init(&glb->resource_types);
84 
85  glb->last_process_id = 0;
86 
87  glb->atoms_table = atomshashtable_new();
88  if (IS_NULL_PTR(glb->atoms_table)) {
89  free(glb);
90  return NULL;
91  }
92  glb->atoms_ids_table = valueshashtable_new();
93  if (IS_NULL_PTR(glb->atoms_ids_table)) {
94  free(glb->atoms_table);
95  free(glb);
96  return NULL;
97  }
98 
99  defaultatoms_init(glb);
100 
101  glb->modules_by_index = NULL;
102  glb->loaded_modules_count = 0;
103  glb->modules_table = atomshashtable_new();
104  if (IS_NULL_PTR(glb->modules_table)) {
105  free(glb->atoms_ids_table);
106  free(glb->atoms_table);
107  free(glb);
108  return NULL;
109  }
110 #ifndef AVM_NO_SMP
111  glb->modules_lock = smp_rwlock_create();
112  if (IS_NULL_PTR(glb->modules_lock)) {
113  free(glb->modules_table);
114  free(glb->atoms_ids_table);
115  free(glb->atoms_table);
116  free(glb);
117  return NULL;
118  }
119 #endif
120 
121  timer_list_init(&glb->timer_list);
122 #ifndef AVM_NO_SMP
123  smp_spinlock_init(&glb->timer_spinlock);
124 #endif
125 
126  glb->ref_ticks = 0;
127 #if !defined(AVM_NO_SMP) && ATOMIC_LLONG_LOCK_FREE != 2
128  smp_spinlock_init(&glb->ref_ticks_spinlock);
129 #endif
130 
131 #if HAVE_OPEN && HAVE_CLOSE
132  ErlNifEnv env;
133  erl_nif_env_partial_init_from_globalcontext(&env, glb);
134  glb->posix_fd_resource_type = enif_init_resource_type(&env, "posix_fd", &posix_fd_resource_type_init, ERL_NIF_RT_CREATE, NULL);
135  if (IS_NULL_PTR(glb->posix_fd_resource_type)) {
136 #ifndef AVM_NO_SMP
137  smp_rwlock_destroy(glb->modules_lock);
138 #endif
139  free(glb->modules_table);
140  free(glb->atoms_ids_table);
141  free(glb->atoms_table);
142  free(glb);
143  return NULL;
144  }
145 #endif
146 
147  sys_init_platform(glb);
148 
149 #ifndef AVM_NO_SMP
150  glb->schedulers_mutex = smp_mutex_create();
151  if (IS_NULL_PTR(glb->schedulers_mutex)) {
152 #if HAVE_OPEN && HAVE_CLOSE
153  resource_type_destroy(glb->posix_fd_resource_type);
154 #endif
155  smp_rwlock_destroy(glb->modules_lock);
156  free(glb->modules_table);
157  free(glb->atoms_ids_table);
158  free(glb->atoms_table);
159  free(glb);
160  return NULL;
161  }
162  glb->schedulers_cv = smp_condvar_create();
163  if (IS_NULL_PTR(glb->schedulers_cv)) {
164  smp_mutex_destroy(glb->schedulers_mutex);
165 #if HAVE_OPEN && HAVE_CLOSE
166  resource_type_destroy(glb->posix_fd_resource_type);
167 #endif
168  smp_rwlock_destroy(glb->modules_lock);
169  free(glb->modules_table);
170  free(glb->atoms_ids_table);
171  free(glb->atoms_table);
172  free(glb);
173  return NULL;
174  }
175  glb->online_schedulers = smp_get_online_processors();
176  glb->running_schedulers = 0;
177  glb->waiting_scheduler = false;
178 
179  smp_spinlock_init(&glb->env_spinlock);
180 #endif
181  glb->scheduler_stop_all = false;
182 
183  return glb;
184 }

189 
190 #ifndef AVM_NO_SMP
191  smp_condvar_destroy(glb->schedulers_cv);
192  smp_mutex_destroy(glb->schedulers_mutex);
193  smp_rwlock_destroy(glb->modules_lock);
194 #endif
195  synclist_destroy(&glb->registered_processes);
196  synclist_destroy(&glb->processes_table);
197  // Destroy every remaining refc binaries.
198  struct ListHead *item;
199  struct ListHead *tmp;
200  struct ListHead *refc_binaries = synclist_nolock(&glb->refc_binaries);
201  MUTABLE_LIST_FOR_EACH (item, tmp, refc_binaries) {
202  struct RefcBinary *refc = GET_LIST_ENTRY(item, struct RefcBinary, head);
203  refc_binary_destroy(refc, glb);
204  }
205  synclist_destroy(&glb->refc_binaries);
206 
207  struct ListHead *open_avm_packs = synclist_nolock(&glb->avmpack_data);
208  MUTABLE_LIST_FOR_EACH (item, tmp, open_avm_packs) {
209  struct AVMPackData *avmpack_data = GET_LIST_ENTRY(item, struct AVMPackData, avmpack_head);
210  avmpack_data_destroy(avmpack_data, glb);
211  }
212  synclist_destroy(&glb->avmpack_data);
213 
214  struct ListHead *listeners = synclist_nolock(&glb->listeners);
215  MUTABLE_LIST_FOR_EACH (item, tmp, listeners) {
216  sys_listener_destroy(item);
217  }
218  synclist_destroy(&glb->listeners);
219 
220  // Destroy resource types
221  struct ListHead *resource_types = synclist_nolock(&glb->resource_types);
222  MUTABLE_LIST_FOR_EACH (item, tmp, resource_types) {
223  struct ResourceType *resource_type = GET_LIST_ENTRY(item, struct ResourceType, head);
224  resource_type_destroy(resource_type);

226  synclist_destroy(&glb->resource_types);
227 
228  free(glb);
229 }

308 bool globalcontext_register_process(GlobalContext *glb, int atom_index, int local_process_id)
309 {
310  struct ListHead *registered_processes_list = synclist_wrlock(&glb->registered_processes);
311  struct ListHead *item;
312  LIST_FOR_EACH (item, registered_processes_list) {
313  const struct RegisteredProcess *registered_process = GET_LIST_ENTRY(item, struct RegisteredProcess, registered_processes_list_head);
314  if (registered_process->atom_index == atom_index) {
315  synclist_unlock(&glb->registered_processes);
316  return false;
317  }
318  }
319 
320  struct RegisteredProcess *registered_process = malloc(sizeof(struct RegisteredProcess));
321  if (IS_NULL_PTR(registered_process)) {
322  fprintf(stderr, "Failed to allocate memory: %s:%i.\n", __FILE__, __LINE__);
323  AVM_ABORT();
324  }
325  registered_process->atom_index = atom_index;
326  registered_process->local_process_id = local_process_id;
327 
328  list_append(registered_processes_list, &registered_process->registered_processes_list_head);
329  synclist_unlock(&glb->registered_processes);
330 
331  return true;
332 }

393 int globalcontext_insert_atom_maybe_copy(GlobalContext *glb, AtomString atom_string, int copy)
394 {
395  struct AtomsHashTable *htable = glb->atoms_table;
396 
397  unsigned long atom_index = atomshashtable_get_value(htable, atom_string, ULONG_MAX);
398  if (atom_index == ULONG_MAX) {
399  if (copy) {
400  uint8_t len = *((uint8_t *) atom_string);
401  uint8_t *buf = malloc(1 + len);
402  if (UNLIKELY(IS_NULL_PTR(buf))) {
403  fprintf(stderr, "Unable to allocate memory for atom string\n");
404  AVM_ABORT();
405  }
406  memcpy(buf, atom_string, 1 + len);
407  atom_string = buf;
408  }
409  atom_index = htable->count;
410  if (!atomshashtable_insert(htable, atom_string, atom_index)) {
411  return -1;
412  }
413  if (!valueshashtable_insert(glb->atoms_ids_table, atom_index, (unsigned long) atom_string)) {
414  return -1;
415  }
416  }
417 
418  return (int) atom_index;
419 }

451 int globalcontext_insert_module(GlobalContext *global, Module *module)
452 {
453  SMP_RWLOCK_WRLOCK(global->modules_lock);
454  AtomString module_name_atom = module_get_atom_string_by_id(module, 1);
455  if (!atomshashtable_insert(global->modules_table, module_name_atom, TO_ATOMSHASHTABLE_VALUE(module))) {
456  SMP_RWLOCK_UNLOCK(global->modules_lock);
457  return -1;
458  }
459 
460  int module_index = global->loaded_modules_count;
461 
462  Module **new_modules_by_index = calloc(module_index + 1, sizeof(Module *));
463  if (IS_NULL_PTR(new_modules_by_index)) {
464  fprintf(stderr, "Failed to allocate memory: %s:%i.\n", __FILE__, __LINE__);
465  SMP_RWLOCK_UNLOCK(global->modules_lock);
466  AVM_ABORT();
467  }
468  if (global->modules_by_index) {
469  for (int i = 0; i < module_index; i++) {
470  new_modules_by_index[i] = global->modules_by_index[i];
471  }
472  free(global->modules_by_index);
473  }
474 
475  module->module_index = module_index;
476 
477  global->modules_by_index = new_modules_by_index;
478  global->modules_by_index[module_index] = module;
479  global->loaded_modules_count++;
480  SMP_RWLOCK_UNLOCK(global->modules_lock);
481 
482  return module_index;
483 }

486 {

488 
489  if (!found_module) {

491  if (IS_NULL_PTR(module_name)) {
492  return NULL;

496  strcat(module_name, ".beam");

501  return NULL;

503

505  }
