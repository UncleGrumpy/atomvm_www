1005  if (ok) {
1006  printf("%s", str);
1007  free(str);
1008 
1009  term refcopy = term_from_ref_ticks(ref_ticks, &ctx->heap);
1010 
1011  term reply = term_alloc_tuple(3, &ctx->heap);
1012  term_put_tuple_element(reply, 0, IO_REPLY_ATOM);
1013  term_put_tuple_element(reply, 1, refcopy);
1014  term_put_tuple_element(reply, 2, OK_ATOM);
1015 
1016  port_send_message(ctx->global, pid, reply);
1017  }
1018  }
1019 
1020  } else if (port_is_standard_port_command(msg)) {
1021 
1022  term pid = term_get_tuple_element(msg, 0);
1023  term ref = term_get_tuple_element(msg, 1);

1039  port_send_reply(ctx, pid, ref, OK_ATOM);

1041  free(str);
1042  } else {
1043  term error = port_create_error_tuple(ctx, BADARG_ATOM);

1129  }
