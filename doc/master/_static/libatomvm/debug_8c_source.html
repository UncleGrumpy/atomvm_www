23 static COLD_FUNC void debug_display_type(term t, const Context *ctx)
24 {
25  if (term_is_atom(t) || term_is_integer(t) || term_is_nil(t) || term_is_pid(t)) {
26  term_display(stderr, t, ctx);
27  } else if ((t & 0x3F) == 0) {
28  fprintf(stderr, "tuple(%i)", term_get_size_from_boxed_header(t));
29  } else if (term_is_boxed(t)) {
30  fprintf(stderr, "boxed(0x%lx)", (unsigned long) term_to_term_ptr(t));
31  } else if ((t & 0x3) == 0x1) {
32  fprintf(stderr, "list(0x%lx)", (unsigned long) term_to_term_ptr(t));
33  } else if (term_is_catch_label(t)) {
34  int module_index;
35  int catch_label = term_to_catch_label_and_module(t, &module_index);
36  fprintf(stderr, "catch label(%i:%i)", module_index, catch_label);
37  } else if (term_is_cp(t)) {
38  fprintf(stderr, "continuation pointer");
39  } else {
40  fprintf(stderr, "unknown");
41  }
42 }

44 static COLD_FUNC void debug_dump_binary_mem(char *buf, term val, unsigned n)
45 {
46  for (unsigned int i = 0; i < n; ++i) {
47  int bit_i = val >> i & 0x1;
48  buf[(n - 1) - i] = bit_i ? '1' : '0';
49  }
50  buf[n] = '\0';
51 }

53 static COLD_FUNC void debug_dump_term(Context *ctx, term *pos, const char *region, unsigned i)

64 COLD_FUNC void debug_dump_memory(Context *ctx, term *start, term *end, const char *region)
65 {
66  unsigned long size = end - start;

69  fprintf(stderr, "DEBUG: %s end: 0x%lx\n", region, (unsigned long) end);
70  fprintf(stderr, "DEBUG: %s size: %li words\n", region, size);
71  term *pos = start;
