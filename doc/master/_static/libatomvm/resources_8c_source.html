34 ErlNifResourceType *enif_init_resource_type(ErlNifEnv *env, const char *name, const ErlNifResourceTypeInit *init, ErlNifResourceFlags flags, ErlNifResourceFlags *tried)
35 {
36  if (tried) {
37  *tried = flags;
38  }
39  struct ResourceType *result = malloc(sizeof(struct ResourceType));
40  if (UNLIKELY(result == NULL)) {
41  return NULL;
42  }
43  result->name = strdup(name);
44  result->global = env->global;
45  list_init(&result->head);
46  result->dtor = NULL;
47  if (init->members >= 1) {
48  result->dtor = init->dtor;
49  }
50  synclist_append(&env->global->resource_types, &result->head);
51 
52  return result;
53 }

55 void *enif_alloc_resource(ErlNifResourceType *type, unsigned size)
56 {
57  struct RefcBinary *refc = refc_binary_create_resource(size, type);
58  if (UNLIKELY(refc == NULL)) {
59  return NULL;
60  }
61  // We add it now to the list of refc binaries, so resource is destroyed at
62  // the latest when globalcontext is destroyed
63  synclist_append(&type->global->refc_binaries, &refc->head);
64  return (void *) refc_binary_get_data(refc);
65 }

67 int enif_get_resource(ErlNifEnv *env, ERL_NIF_TERM t, ErlNifResourceType *type, void **objp)
68 {
69  UNUSED(env);
70 
71  if (UNLIKELY(!term_is_refc_binary(t))) {
72  return false;
73  }
74  if (UNLIKELY(term_refc_binary_is_const(t))) {
75  return false;
76  }
77  const term *boxed_value = term_to_const_term_ptr(t);
78  struct RefcBinary *refc = (struct RefcBinary *) boxed_value[3];
79  if (UNLIKELY(refc->resource_type != type)) {
80  return false;
81  }
82  *objp = (void *) refc_binary_get_data(refc);
83  return true;
84 }
