35 {
36  if (tried) {
37  *tried = flags;
38  }
39  struct ResourceType *result = malloc(sizeof(struct ResourceType));
40  if (UNLIKELY(result == NULL)) {
41  return NULL;
42  }
43  result->name = strdup(name);
44  result->global = env->global;
45  list_init(&result->head);
46  list_init(&result->monitors);
47  result->dtor = NULL;
48  result->stop = NULL;
49  result->down = NULL;
50  if (init->members >= 1) {
51  result->dtor = init->dtor;
52  if (init->members >= 2) {
53  result->stop = init->stop;
54  if (init->members >= 3) {
55  result->down = init->down;
56  }
57  }
58  }
59  synclist_append(&env->global->resource_types, &result->head);
60 
61  return result;
62 }

119 int enif_select(ErlNifEnv *env, ErlNifEvent event, enum ErlNifSelectFlags mode, void *obj, const ErlNifPid *pid, ERL_NIF_TERM ref)
120 {
121  if (!(mode & (ERL_NIF_SELECT_STOP | ERL_NIF_SELECT_READ | ERL_NIF_SELECT_WRITE))) {
122  return ERL_NIF_SELECT_BADARG;
123  }
124  if (UNLIKELY(mode & (ERL_NIF_SELECT_READ | ERL_NIF_SELECT_WRITE) && !term_is_reference(ref) && ref != UNDEFINED_ATOM)) {
125  return ERL_NIF_SELECT_BADARG;
126  }
127 
128  struct RefcBinary *resource = refc_binary_from_data(obj);
129  // Search for event and obj
130  struct ListHead *item;
131  struct ListHead *select_events = synclist_wrlock(&env->global->select_events);
132  struct SelectEvent *select_event = NULL;
133  LIST_FOR_EACH (item, select_events) {
134  select_event = GET_LIST_ENTRY(item, struct SelectEvent, head);
135  if (select_event->event == event && select_event->resource == resource) {
136  break;
137  }
138  select_event = NULL;
139  }
140  if (mode & ERL_NIF_SELECT_STOP) {
141  if (select_event == NULL) {
142  synclist_unlock(&env->global->select_events);
143  return ERL_NIF_SELECT_INVALID_EVENT;
144  }
145  bool was_read = select_event->read;
146  bool was_write = select_event->write;
147  if (!was_read && !was_write) {
148  list_remove(&select_event->head);
149  synclist_unlock(&env->global->select_events);
150  // We can call stop now.
151  if (resource->resource_type->stop) {
152  resource->resource_type->stop(env, obj, event, true);
153  }
154  refc_binary_decrement_refcount(resource, env->global);
155  free((void *) select_event);
156  return ERL_NIF_SELECT_STOP_CALLED;
157  }
158  // We cannot call stop now because scheduler loop unlocks after building
159  // the select set but before calling select (or equivalent)
160  // So instead we flag the event.
161  select_event->close = 1;
162  select_event->read = 0;
163  select_event->write = 0;
164  synclist_unlock(&env->global->select_events);
165  // Platform loop should check close flag after unregister is called
166  if (was_read) {
167  sys_unregister_select_event(env->global, event, false);
168  }
169  if (was_write) {
170  sys_unregister_select_event(env->global, event, true);
171  }
172  return ERL_NIF_SELECT_STOP_SCHEDULED;
173  }
174  // Create new event if it doesn't exist.
175  if (select_event == NULL) {
176  select_event = malloc(sizeof(struct SelectEvent));
177  select_event->event = event;
178  select_event->resource = resource;
179  // Resource is used in select_event, so we increase refcount.
180  refc_binary_increment_refcount(resource);
181  list_init(&select_event->head);
182  list_append(select_events, &select_event->head);
183  }
184  // Second read or second write overwrite ref & pid.
185  if (ref == UNDEFINED_ATOM) {
186  select_event->ref_ticks = 0;
187  select_event->undefined_ref = 1;
188  } else {
189  select_event->ref_ticks = term_to_ref_ticks(ref);
190  select_event->undefined_ref = 0;
191  }
192  select_event->local_pid = *pid;
193  select_event->read = mode & ERL_NIF_SELECT_READ;
194  select_event->write = mode & ERL_NIF_SELECT_WRITE;
195  select_event->close = 0;
196  synclist_unlock(&env->global->select_events);
197  if (select_event->read) {
198  sys_register_select_event(env->global, event, false);
199  }
200  if (select_event->write) {
201  sys_register_select_event(env->global, event, true);
202  }
203  return 0;
204 }

206 static void select_event_send_notification(struct SelectEvent *select_event, bool is_write, GlobalContext *global)
207 {
208  BEGIN_WITH_STACK_HEAP(TUPLE_SIZE(4) + REF_SIZE + TERM_BOXED_REFC_BINARY_SIZE, heap)
209  term notification = term_alloc_tuple(4, &heap);
210  term_put_tuple_element(notification, 0, SELECT_ATOM);
211  term_put_tuple_element(notification, 1, term_from_resource(select_event->resource->data, &heap));
212  refc_binary_increment_refcount(select_event->resource);
213  term ref;
214  if (select_event->undefined_ref) {
215  ref = UNDEFINED_ATOM;
216  } else {
217  ref = term_from_ref_ticks(select_event->ref_ticks, &heap);
218  }
219  term_put_tuple_element(notification, 2, ref);
220  term_put_tuple_element(notification, 3, is_write ? READY_OUTPUT_ATOM : READY_INPUT_ATOM);
221  globalcontext_send_message(global, select_event->local_pid, notification);
222  if (is_write) {
223  select_event->write = 0;
224  } else {
225  select_event->read = 0;
226  }
227  END_WITH_STACK_HEAP(heap, global)
228  sys_unregister_select_event(global, select_event->event, is_write);
229 }

231 bool select_event_notify(ErlNifEvent event, bool is_read, bool is_write, GlobalContext *global)
232 {
233  bool result = false;
234  struct SelectEvent *select_event = NULL;
235  struct ListHead *item;
236  struct ListHead *select_events = synclist_wrlock(&global->select_events);
237  LIST_FOR_EACH (item, select_events) {
238  select_event = GET_LIST_ENTRY(item, struct SelectEvent, head);
239  if (select_event->event == event) {
240  break;
241  }
242  select_event = NULL;
243  }
244  synclist_unlock(&global->select_events);
245  if (select_event) {
246  if (is_read && select_event->read) {
247  select_event_send_notification(select_event, false, global);
248  result = true;
249  }
250  if (is_write && select_event->write) {
251  select_event_send_notification(select_event, true, global);
252  result = true;
253  }
254  }
255  return result;
256 }

269 void select_event_count_and_destroy_closed(struct ListHead *select_events, size_t *read, size_t *write, size_t *either, GlobalContext *global)
270 {
271  size_t read_count = 0;
272  size_t write_count = 0;
273  size_t either_count = 0;
274 
275  struct ListHead *item;
276  struct ListHead *tmp;
277  MUTABLE_LIST_FOR_EACH (item, tmp, select_events) {
278  struct SelectEvent *select_event = GET_LIST_ENTRY(item, struct SelectEvent, head);
279  if (select_event->close) {
280  list_remove(&select_event->head);
281  select_event_destroy(select_event, global);
282  } else {
283  if (select_event->read || select_event->write) {
284  if (select_event->read) {
285  read_count++;
286  }
287  if (select_event->write) {
288  write_count++;
289  }
290  either_count++;
291  }
292  }
293  }
294  if (read) {
295  *read = read_count;
296  }
297  if (write) {
298  *write = write_count;
299  }
300  if (either) {
301  *either = either_count;
302  }
303 }

305 int enif_monitor_process(ErlNifEnv *env, void *obj, const ErlNifPid *target_pid, ErlNifMonitor *mon)
306 {
307  struct RefcBinary *resource = refc_binary_from_data(obj);
308  if (resource->resource_type == NULL || resource->resource_type->down == NULL) {
309  return -1;
310  }
311 
312  Context *target = globalcontext_get_process_lock(env->global, *target_pid);
313  if (IS_NULL_PTR(target)) {
314  return 1;
315  }
316 
317  struct ResourceMonitor *monitor = context_resource_monitor(target, obj);
318  list_append(&resource->resource_type->monitors, &monitor->resource_list_head);
319  globalcontext_get_process_unlock(env->global, target);
320 
321  if (mon) {
322  *mon = monitor->base.ref_ticks;
323  }
324 
325  return 0;
326 }

329 {

352
