164 int mbedtls_ssl_send_cb(void *ctx, const unsigned char *buf, size_t len)
165 {
166  TRACE("%s\n", __func__);
167  ssize_t res = socket_send((struct SocketResource *) ctx, buf, len, term_invalid_term());
168  if (res == SocketWouldBlock) {
169  return MBEDTLS_ERR_SSL_WANT_WRITE;
170  }
171  return res;
172 }

174 int mbedtls_ssl_recv_cb(void *ctx, unsigned char *buf, size_t len)
175 {
176  TRACE("%s\n", __func__);
177  ssize_t res = socket_recv((struct SocketResource *) ctx, buf, len, 0, NULL, NULL);
178  if (res == SocketWouldBlock) {
179  return MBEDTLS_ERR_SSL_WANT_READ;
180  }
181  return res;
182 }

354 static term nif_ssl_config_defaults(Context *ctx, int argc, term argv[])
355 {
356  TRACE("%s\n", __func__);
357  UNUSED(argc);
358  VALIDATE_VALUE(argv[1], term_is_atom);
359  VALIDATE_VALUE(argv[2], term_is_atom);
360 
361  void *rsrc_obj_ptr;
362  if (UNLIKELY(!enif_get_resource(erl_nif_env_from_context(ctx), argv[0], sslconfig_resource_type, &rsrc_obj_ptr))) {
363  RAISE_ERROR(BADARG_ATOM);
364  }
365  struct SSLConfigResource *rsrc_obj = (struct SSLConfigResource *) rsrc_obj_ptr;
366 
367  int endpoint = interop_atom_term_select_int(endpoint_table, argv[1], ctx->global);
368  if (UNLIKELY(endpoint == UNKNOWN_TABLE_VALUE)) {
369  RAISE_ERROR(BADARG_ATOM);
370  }
371 
372  enum inet_type transport_type = inet_atom_to_type(argv[2], ctx->global);
373  if (UNLIKELY(transport_type != InetStreamType && transport_type != InetDgramType)) {
374  RAISE_ERROR(BADARG_ATOM);
375  }
376  int transport = transport_type == InetStreamType ? MBEDTLS_SSL_TRANSPORT_STREAM : MBEDTLS_SSL_TRANSPORT_DATAGRAM;
377 
378  int err = mbedtls_ssl_config_defaults(&rsrc_obj->config, endpoint, transport, MBEDTLS_SSL_PRESET_DEFAULT);
379  if (UNLIKELY(err != 0)) {
380  RAISE_ERROR(OUT_OF_MEMORY_ATOM);
381  }
382 
383  return OK_ATOM;
384 }

488 static term make_err_result(int err, Context *ctx)
489 {
490  switch (err) {
491  case 0:
492  return OK_ATOM;
493  case MBEDTLS_ERR_SSL_WANT_READ:
494  return globalcontext_make_atom(ctx->global, ATOM_STR("\x9", "want_read"));
495  case MBEDTLS_ERR_SSL_WANT_WRITE:
496  return globalcontext_make_atom(ctx->global, ATOM_STR("\xA", "want_write"));
497 #if MBEDTLS_VERSION_NUMBER >= 0x020B0000
498  case MBEDTLS_ERR_SSL_ASYNC_IN_PROGRESS:
499  return globalcontext_make_atom(ctx->global, ATOM_STR("\xA", "async_in_progress"));
500 #if MBEDTLS_VERSION_NUMBER >= 0x020E0000
501  case MBEDTLS_ERR_SSL_CRYPTO_IN_PROGRESS:
502  return globalcontext_make_atom(ctx->global, ATOM_STR("\xA", "crypto_in_progress"));
503 #endif
504 #endif
505  default: {
506  if (UNLIKELY(memory_ensure_free(ctx, TUPLE_SIZE(2)) != MEMORY_GC_OK)) {
507  AVM_LOGW(TAG, "Failed to allocate memory: %s:%i.", __FILE__, __LINE__);
508  RAISE_ERROR(OUT_OF_MEMORY_ATOM);
509  }
510  term error_tuple = term_alloc_tuple(2, &ctx->heap);
511  term_put_tuple_element(error_tuple, 0, ERROR_ATOM);
512  term_put_tuple_element(error_tuple, 1, term_from_int(err));
513  return error_tuple;
514  }
515  }

571 
572  const uint8_t *buffer = (const uint8_t *) term_binary_data(data);
573  size_t len = term_binary_size(data);
574 
575  int res = mbedtls_ssl_write(&context_rsrc->context, buffer, len);
576 
577  if (res == (int) len) {
578  return OK_ATOM;
579  }
580  if (LIKELY(res >= 0)) { // ensure we don't return OK if res is 0
581  size_t rest_len = len - res;
582  size_t requested_size = term_sub_binary_heap_size(data, rest_len);
583  if (UNLIKELY(memory_ensure_free_with_roots(ctx, TUPLE_SIZE(2) + requested_size, 1, &data, MEMORY_CAN_SHRINK) != MEMORY_GC_OK)) {
584  AVM_LOGW(TAG, "Failed to allocate memory: %s:%i.", __FILE__, __LINE__);
585  RAISE_ERROR(OUT_OF_MEMORY_ATOM);
586  }
587

589  return port_create_tuple2(ctx, OK_ATOM, rest);
590  }

595 static term nif_ssl_read(Context *ctx, int argc, term argv[])
596 {
597  TRACE("%s\n", __func__);
598  UNUSED(argc);
599  VALIDATE_VALUE(argv[1], term_is_integer);
600 
601  void *rsrc_obj_ptr;
602  if (UNLIKELY(!enif_get_resource(erl_nif_env_from_context(ctx), argv[0], sslcontext_resource_type, &rsrc_obj_ptr))) {
603  RAISE_ERROR(BADARG_ATOM);
604  }
605  struct SSLContextResource *context_rsrc = (struct SSLContextResource *) rsrc_obj_ptr;
606 
607  avm_int_t len = term_to_int(argv[1]);
608  if (len < 0) {
609  RAISE_ERROR(BADARG_ATOM);
610  }
611 #if MBEDTLS_VERSION_NUMBER >= 0x03000000
612  if (len == 0) {
613  len = mbedtls_ssl_get_max_in_record_payload(&context_rsrc->context);
614  }
615 #endif
616  if (len <= 0) {
617  len = DEFAULT_READ_BUFFER_FALLBACK;
618  }
619  size_t ensure_packet_avail = term_binary_data_size_in_terms(len) + BINARY_HEADER_SIZE;
620  size_t requested_size = TUPLE_SIZE(2) + ensure_packet_avail;
621 
622  if (UNLIKELY(memory_ensure_free(ctx, requested_size) != MEMORY_GC_OK)) {
623  AVM_LOGW(TAG, "Failed to allocate memory: %s:%i.", __FILE__, __LINE__);
624  RAISE_ERROR(OUT_OF_MEMORY_ATOM);
625  }
626 
627  term data = term_create_uninitialized_binary(len, &ctx->heap, ctx->global);
628  uint8_t *buffer = (uint8_t *) term_binary_data(data);
629 
630  int res = mbedtls_ssl_read(&context_rsrc->context, buffer, len);
631 
632  if (res == len) {
633  return port_create_tuple2(ctx, OK_ATOM, data);
634  }
635 
636  if (res >= 0 && res < len) {
637  size_t requested_size = term_sub_binary_heap_size(data, res);
638  if (UNLIKELY(memory_ensure_free_with_roots(ctx, TUPLE_SIZE(2) + requested_size, 1, &data, MEMORY_CAN_SHRINK) != MEMORY_GC_OK)) {
639  AVM_LOGW(TAG, "Failed to allocate memory: %s:%i.", __FILE__, __LINE__);
640  RAISE_ERROR(OUT_OF_MEMORY_ATOM);
641  }
642 
643  term rest = term_maybe_create_sub_binary(data, 0, res, &ctx->heap, ctx->global);
644  return port_create_tuple2(ctx, OK_ATOM, rest);
645  }
646 
647  return make_err_result(res, ctx);
648 }
