33 void mailbox_send(Context *c, term t)
34 {
35  TRACE("Sending 0x%lx to pid %i\n", t, c->process_id);
36 
37  unsigned long estimated_mem_usage = memory_estimate_usage(t);
38 
39  Message *m = malloc(sizeof(Message) + estimated_mem_usage * sizeof(term));
40  if (IS_NULL_PTR(m)) {
41  fprintf(stderr, "Failed to allocate memory: %s:%i.\n", __FILE__, __LINE__);
42  return;
43  }
44  m->mso_list = term_nil();
45 
46  term *heap_pos = mailbox_message_memory(m);
47  m->message = memory_copy_term_tree(&heap_pos, t, &m->mso_list);
48  m->msg_memory_size = estimated_mem_usage;
49 
50  list_append(&c->mailbox, &m->mailbox_list_head);
51 
52  if (c->jump_to_on_restore) {
53  c->saved_ip = c->jump_to_on_restore;
54  c->jump_to_on_restore = NULL;
55  }
56  scheduler_make_ready(c->global, c);
57 }

69 term mailbox_peek(Context *c)
70 {
71  Message *m = GET_LIST_ENTRY(list_first(&c->mailbox), Message, mailbox_list_head);
72 
73  TRACE("Pid %i is peeking 0x%lx.\n", c->process_id, m->message);
74 
75  if (c->e - c->heap_ptr < m->msg_memory_size) {
76  // ADDITIONAL_PROCESSING_MEMORY_SIZE: ensure some additional memory for message processing, so there is
77  // no need to run GC again.
78  if (UNLIKELY(memory_ensure_free(c, m->msg_memory_size + ADDITIONAL_PROCESSING_MEMORY_SIZE) != MEMORY_GC_OK)) {
79  fprintf(stderr, "Failed to allocate memory: %s:%i.\n", __FILE__, __LINE__);
80  }
81  }
82 
83  term rt = memory_copy_term_tree(&c->heap_ptr, m->message, &c->mso_list);
84 
85  return rt;
