69 term refc_binary_create_binary_info(Context *ctx)
70 {
71  size_t len = 0;
72  struct ListHead *item;
73  LIST_FOR_EACH (item, &ctx->global->refc_binaries) {
74  len++;
75  }
76  if (len == 0) {
77  return term_nil();
78  }
79  if (memory_ensure_free(ctx, len * (2 + TUPLE_SIZE(2))) != MEMORY_GC_OK) {
80  return term_invalid_term();
81  }
82  term ret = term_nil();
83  LIST_FOR_EACH (item, &ctx->global->refc_binaries) {
84  struct RefcBinary *refc = GET_LIST_ENTRY(item, struct RefcBinary, head);
85  term t = term_alloc_tuple(2, ctx);
86  term_put_tuple_element(t, 0, term_from_int(refc->size));
87  term_put_tuple_element(t, 1, term_from_int(refc->ref_count));
88  ret = term_list_prepend(t, ret, ctx);
89  }
90  return ret;
91 }
