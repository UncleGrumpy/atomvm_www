65 #define DEFINE_HASH_FOLD(ALGORITHM, SUFFIX) \
66  static InteropFunctionResult ALGORITHM##_hash_fold_fun(term t, void *accum) \
67  { \
68  mbedtls_##ALGORITHM##_context *md_ctx = (mbedtls_##ALGORITHM##_context *) accum; \
69  if (term_is_integer(t)) { \
70  avm_int64_t tmp = term_maybe_unbox_int64(t); \
71  if (tmp < 0 || tmp > 255) { \
72  return InteropBadArg; \
73  } \
74  uint8_t val = (uint8_t) tmp; \
75  if (UNLIKELY(mbedtls_##ALGORITHM##_update##SUFFIX(md_ctx, &val, 1) != 0)) { \
76  return InteropBadArg; \
77  } \
78  } else /* term_is_binary(t) */ { \
79  if (UNLIKELY(mbedtls_##ALGORITHM##_update##SUFFIX(md_ctx, (uint8_t *) term_binary_data(t), term_binary_size(t)) != 0)) { \
80  return InteropBadArg; \
81  } \
82  } \
83  return InteropOk; \
84  }

86 #define DEFINE_HASH_FOLD_NORET(ALGORITHM, SUFFIX) \
87  static InteropFunctionResult ALGORITHM##_hash_fold_fun(term t, void *accum) \
88  { \
89  mbedtls_##ALGORITHM##_context *md_ctx = (mbedtls_##ALGORITHM##_context *) accum; \
90  if (term_is_integer(t)) { \
91  avm_int64_t tmp = term_maybe_unbox_int64(t); \
92  if (tmp < 0 || tmp > 255) { \
93  return InteropBadArg; \
94  } \
95  uint8_t val = (uint8_t) tmp; \
96  mbedtls_##ALGORITHM##_update##SUFFIX(md_ctx, &val, 1); \
97  } else /* term_is_binary(t) */ { \
98  mbedtls_##ALGORITHM##_update##SUFFIX(md_ctx, (uint8_t *) term_binary_data(t), term_binary_size(t)); \
99  } \
100  return InteropOk; \
101  }

103 #define DEFINE_DO_HASH(ALGORITHM, SUFFIX) \
104  static bool do_##ALGORITHM##_hash(term data, unsigned char *dst) \
105  { \
106  mbedtls_##ALGORITHM##_context md_ctx; \
107  \
108  mbedtls_##ALGORITHM##_init(&md_ctx); \
109  mbedtls_##ALGORITHM##_starts##SUFFIX(&md_ctx); \
110  \
111  InteropFunctionResult result = interop_chardata_fold(data, ALGORITHM##_hash_fold_fun, NULL, (void *) &md_ctx); \
112  if (UNLIKELY(result != InteropOk)) { \
113  return false; \
114  } \
115  \
116  if (UNLIKELY(mbedtls_##ALGORITHM##_finish##SUFFIX(&md_ctx, dst) != 0)) { \
117  return false; \
118  } \
119  \
120  return true; \
121  }

226 static term nif_crypto_hash(Context *ctx, int argc, term argv[])
227 {
228  UNUSED(argc);
229  term type = argv[0];
230  VALIDATE_VALUE(type, term_is_atom);
231  term data = argv[1];
232 
233  unsigned char digest[MAX_MD_SIZE];
234  size_t digest_len = 0;
235 
236  enum crypto_algorithm algo = interop_atom_term_select_int(crypto_algorithm_table, type, ctx->global);
237  switch (algo) {
238  case CryptoMd5: {
239  if (UNLIKELY(!do_md5_hash(data, digest))) {
240  RAISE_ERROR(BADARG_ATOM)
241  }
242  digest_len = 16;
243  break;
244  }
245  case CryptoSha1: {
246  if (UNLIKELY(!do_sha1_hash(data, digest))) {
247  RAISE_ERROR(BADARG_ATOM)
248  }
249  digest_len = 20;
250  break;
251  }
252  case CryptoSha224: {
253  if (UNLIKELY(!do_sha256_hash_true(data, digest))) {
254  RAISE_ERROR(BADARG_ATOM)
255  }
256  digest_len = 28;
257  break;
258  }
259  case CryptoSha256: {
260  if (UNLIKELY(!do_sha256_hash_false(data, digest))) {
261  RAISE_ERROR(BADARG_ATOM)
262  }
263  digest_len = 32;
264  break;
265  }
266  case CryptoSha384: {
267  if (UNLIKELY(!do_sha512_hash_true(data, digest))) {
268  RAISE_ERROR(BADARG_ATOM)
269  }
270  digest_len = 48;
271  break;
272  }
273  case CryptoSha512: {
274  if (UNLIKELY(!do_sha512_hash_false(data, digest))) {
275  RAISE_ERROR(BADARG_ATOM)
276  }
277  digest_len = 64;
278  break;
279  }
280  default:
281  RAISE_ERROR(BADARG_ATOM);
282  }
283 
284  if (UNLIKELY(memory_ensure_free(ctx, term_binary_heap_size(digest_len)) != MEMORY_GC_OK)) {
285  RAISE_ERROR(OUT_OF_MEMORY_ATOM);
286  }
287  return term_from_literal_binary(digest, digest_len, &ctx->heap, ctx->global);
288 }

316 static term handle_iodata(term iodata, const void **data, size_t *len, void **allocated_ptr)
317 {
318  *allocated_ptr = NULL;
319 
320  if (term_is_binary(iodata)) {
321  *data = term_binary_data(iodata);
322  *len = term_binary_size(iodata);
323  return OK_ATOM;
324  } else if (term_is_list(iodata)) {
325  InteropFunctionResult result = interop_iolist_size(iodata, len);
326  switch (result) {
327  case InteropOk:
328  break;
329  case InteropMemoryAllocFail:
330  return OUT_OF_MEMORY_ATOM;
331  case InteropBadArg:
332  return BADARG_ATOM;
333  }
334  void *allocated_buf = malloc(*len);
335  if (IS_NULL_PTR(allocated_buf)) {
336  return OUT_OF_MEMORY_ATOM;
337  }
338  result = interop_write_iolist(iodata, allocated_buf);
339  switch (result) {
340  case InteropOk:
341  break;
342  case InteropMemoryAllocFail:
343  free(allocated_buf);
344  return OUT_OF_MEMORY_ATOM;
345  case InteropBadArg:
346  free(allocated_buf);
347  return BADARG_ATOM;
348  }
349  *data = allocated_buf;
350  *allocated_ptr = allocated_buf;
351  return OK_ATOM;
352  } else {
353  return BADARG_ATOM;
354  }
355 }

358 {

367  return false;

371 static term make_crypto_error(const char *file, int line, const char *message, Context *ctx)
372 {
373  int err_needed_mem = (strlen(file) * CONS_SIZE) + TUPLE_SIZE(2) + (strlen(message) * CONS_SIZE)
374  + TUPLE_SIZE(3);
375 
376  if (UNLIKELY(memory_ensure_free(ctx, err_needed_mem) != MEMORY_GC_OK)) {
377  RAISE_ERROR(OUT_OF_MEMORY_ATOM);
378  }
379 
380  term file_t = interop_bytes_to_list(file, strlen(file), &ctx->heap);
381  term file_line_t = term_alloc_tuple(2, &ctx->heap);
382  term_put_tuple_element(file_line_t, 0, file_t);
383  term_put_tuple_element(file_line_t, 1, term_from_int(line));
384 
385  term message_t = interop_bytes_to_list(message, strlen(message), &ctx->heap);
386 
387  term err_t = term_alloc_tuple(3, &ctx->heap);
388  term_put_tuple_element(err_t, 0, BADARG_ATOM);
389  term_put_tuple_element(err_t, 1, file_line_t);
390  term_put_tuple_element(err_t, 2, message_t);
391 
392  return err_t;
393 }

395 static term nif_crypto_crypto_one_time(Context *ctx, int argc, term argv[])
396 {
397  bool has_iv = argc == 5;
398  const AtomStringIntPair *cipher_table;
399  term key;
400  term iv;
401  term data;
402  term flag_or_options;
403  if (has_iv) {
404  cipher_table = cipher_iv_table;
405  key = argv[1];
406  iv = argv[2];
407  data = argv[3];
408  flag_or_options = argv[4];
409  } else {
410  cipher_table = cipher_no_iv_table;
411  key = argv[1];
412  data = argv[2];
413  flag_or_options = argv[3];
414  }
415 
416  term cipher_term = argv[0];
417  mbedtls_cipher_type_t cipher
418  = interop_atom_term_select_int(cipher_table, cipher_term, ctx->global);
419  if (UNLIKELY(cipher == MBEDTLS_CIPHER_NONE)) {
420  RAISE_ERROR(make_crypto_error(__FILE__, __LINE__, "Unknown cipher", ctx));
421  }
422 
423  // from this point onward use `goto raise_error` in order to raise and free all buffers
424  term error_atom = UNDEFINED_ATOM;
425 
426  void *allocated_key_data = NULL;
427  void *allocated_iv_data = NULL;
428  void *allocated_data_data = NULL;
429 
430  const void *key_data;
431  size_t key_len;
432  term result_t = handle_iodata(key, &key_data, &key_len, &allocated_key_data);
433  if (UNLIKELY(result_t != OK_ATOM)) {
434  error_atom = result_t;
435  goto raise_error;
436  }
437 
438  const void *iv_data = NULL;
439  size_t iv_len = 0;
440  if (has_iv) {
441  result_t = handle_iodata(iv, &iv_data, &iv_len, &allocated_iv_data);
442  if (UNLIKELY(result_t != OK_ATOM)) {
443  error_atom = result_t;
444  goto raise_error;
445  }
446  }
447 
448  const void *data_data;
449  size_t data_size;
450  result_t = handle_iodata(data, &data_data, &data_size, &allocated_data_data);
451  if (UNLIKELY(result_t != OK_ATOM)) {
452  error_atom = result_t;
453  goto raise_error;
454  }
455 
456  mbedtls_operation_t operation;
457  mbedtls_cipher_padding_t padding = MBEDTLS_PADDING_NONE;
458  bool padding_has_been_set = false;
459 
460  if (term_is_list(flag_or_options)) {
461  term encrypt_flag = interop_kv_get_value_default(
462  flag_or_options, ATOM_STR("\x7", "encrypt"), UNDEFINED_ATOM, ctx->global);
463  if (UNLIKELY(!bool_to_mbedtls_operation(encrypt_flag, &operation))) {
464  error_atom = BADARG_ATOM;
465  goto raise_error;
466  }
467 
468  term padding_term = interop_kv_get_value_default(
469  flag_or_options, ATOM_STR("\x7", "padding"), UNDEFINED_ATOM, ctx->global);
470 
471  if (padding_term != UNDEFINED_ATOM) {
472  padding_has_been_set = true;
473 
474  int padding_int = interop_atom_term_select_int(padding_table, padding_term, ctx->global);
475  if (UNLIKELY(padding_int < 0)) {
476  error_atom = BADARG_ATOM;
477  goto raise_error;
478  }
479  padding = (mbedtls_cipher_padding_t) padding_int;
480  }
481 
482  } else {
483  if (UNLIKELY(!bool_to_mbedtls_operation(flag_or_options, &operation))) {
484  error_atom = BADARG_ATOM;
485  goto raise_error;
486  }
487  }
488 
489  const mbedtls_cipher_info_t *cipher_info = mbedtls_cipher_info_from_type(cipher);
490 
491  mbedtls_cipher_context_t cipher_ctx;
492 
493  void *temp_buf = NULL;
494 
495  int source_line;
496  int result = mbedtls_cipher_setup(&cipher_ctx, cipher_info);
497  if (UNLIKELY(result != 0)) {
498  source_line = __LINE__;
499  goto mbed_error;
500  }
501 
502  result = mbedtls_cipher_setkey(&cipher_ctx, key_data, key_len * 8, operation);
503  if (UNLIKELY(result != 0)) {
504  source_line = __LINE__;
505  goto mbed_error;
506  }
507 
508  // we know that mbedtls supports padding just for CBC, so it makes sense to change to OTP
509  // default (none) just for it. However in case a padding is set for other modes let mbedtls
510  // decide which error should be raised.
511  if (mbedtls_cipher_get_cipher_mode(&cipher_ctx) == MBEDTLS_MODE_CBC || padding_has_been_set) {
512  result = mbedtls_cipher_set_padding_mode(&cipher_ctx, padding);
513  if (UNLIKELY(result != 0)) {
514  source_line = __LINE__;
515  goto mbed_error;
516  }
517  }
518 
519  unsigned int block_size = mbedtls_cipher_get_block_size(&cipher_ctx);
520 
521  size_t temp_buf_size = data_size + block_size;
522  temp_buf = malloc(temp_buf_size);
523  if (IS_NULL_PTR(temp_buf)) {
524  error_atom = OUT_OF_MEMORY_ATOM;
525  goto raise_error;
526  }
527 
528  // from this point onward use `mbed_error` in order to raise and free all buffers
529 
530  result = mbedtls_cipher_crypt(
531  &cipher_ctx, iv_data, iv_len, data_data, data_size, temp_buf, &temp_buf_size);
532  if (result != 0 && result != MBEDTLS_ERR_CIPHER_FULL_BLOCK_EXPECTED) {
533  source_line = __LINE__;
534  goto mbed_error;
535  }
536  mbedtls_cipher_free(&cipher_ctx);
537 
538  free(allocated_key_data);
539  free(allocated_iv_data);
540  free(allocated_data_data);
541 
542  if (UNLIKELY(memory_ensure_free(ctx, temp_buf_size) != MEMORY_GC_OK)) {
543  free(temp_buf);
544  RAISE_ERROR(OUT_OF_MEMORY_ATOM);
545  }
546 
547  term out = term_from_literal_binary(temp_buf, temp_buf_size, &ctx->heap, ctx->global);
548 
549  free(temp_buf);
550 
551  return out;
552 
553 raise_error:
554  free(allocated_key_data);
555  free(allocated_iv_data);
556  free(allocated_data_data);
557  RAISE_ERROR(error_atom);
558 
559 mbed_error:
560  free(temp_buf);
561  free(allocated_key_data);
562  free(allocated_iv_data);
563  free(allocated_data_data);
564 
565  char err_msg[24];
566  snprintf(err_msg, sizeof(err_msg), "Error %x", -result);
567  RAISE_ERROR(make_crypto_error(__FILE__, source_line, err_msg, ctx));
568 }
