145  UNUSED(event);
146  UNUSED(is_direct_call);
147 
148  struct PosixFd *fd_obj = (struct PosixFd *) obj;

193 static term nif_atomvm_posix_open(Context *ctx, int argc, term argv[])
194 {
195  GlobalContext *glb = ctx->global;
196 
197  term path_term = argv[0];
198  term flags = argv[1];
199  VALIDATE_VALUE(flags, term_is_list);
200  int posix_flags = O_NONBLOCK; // Always be non-blocking
201  while (term_is_nonempty_list(flags)) {
202  term flag = term_get_list_head(flags);
203  VALIDATE_VALUE(flag, term_is_atom);
204  // SUSv1 flags
205  if (globalcontext_is_term_equal_to_atom_string(glb, flag, O_RDONLY_ATOM_STR)) {
206  posix_flags |= O_RDONLY;
207  } else if (globalcontext_is_term_equal_to_atom_string(glb, flag, O_WRONLY_ATOM_STR)) {
208  posix_flags |= O_WRONLY;
209  } else if (globalcontext_is_term_equal_to_atom_string(glb, flag, O_RDWR_ATOM_STR)) {
210  posix_flags |= O_RDWR;
211  } else if (globalcontext_is_term_equal_to_atom_string(glb, flag, O_APPEND_ATOM_STR)) {
212  posix_flags |= O_APPEND;
213  } else if (globalcontext_is_term_equal_to_atom_string(glb, flag, O_CREAT_ATOM_STR)) {
214  posix_flags |= O_CREAT;
215  } else if (globalcontext_is_term_equal_to_atom_string(glb, flag, O_EXCL_ATOM_STR)) {
216  posix_flags |= O_EXCL;
217  } else if (globalcontext_is_term_equal_to_atom_string(glb, flag, O_NOCTTY_ATOM_STR)) {
218  posix_flags |= O_NOCTTY;
219  // O_NONBLOCK is not optional
220  } else if (globalcontext_is_term_equal_to_atom_string(glb, flag, O_SYNC_ATOM_STR)) {
221  posix_flags |= O_SYNC;
222  } else if (globalcontext_is_term_equal_to_atom_string(glb, flag, O_TRUNC_ATOM_STR)) {
223  posix_flags |= O_TRUNC;
224  // SUSv4/2018 edition flags
225 #if HAVE_O_EXEC
226  } else if (globalcontext_is_term_equal_to_atom_string(glb, flag, O_EXEC_ATOM_STR)) {
227  posix_flags |= O_EXEC;
228 #endif
229 #if HAVE_O_SEARCH
230  } else if (globalcontext_is_term_equal_to_atom_string(glb, flag, O_SEARCH_ATOM_STR)) {
231  posix_flags |= O_SEARCH;
232 #endif
233 #if HAVE_O_CLOEXEC
234  } else if (globalcontext_is_term_equal_to_atom_string(glb, flag, O_CLOEXEC_ATOM_STR)) {
235  posix_flags |= O_CLOEXEC;
236 #endif
237 #if HAVE_O_DIRECTORY
238  } else if (globalcontext_is_term_equal_to_atom_string(glb, flag, O_DIRECTORY_ATOM_STR)) {
239  posix_flags |= O_DIRECTORY;
240 #endif
241 #if HAVE_O_DSYNC
242  } else if (globalcontext_is_term_equal_to_atom_string(glb, flag, O_DSYNC_ATOM_STR)) {
243  posix_flags |= O_DSYNC;
244 #endif
245 #if HAVE_O_NOFOLLOW
246  } else if (globalcontext_is_term_equal_to_atom_string(glb, flag, O_NOFOLLOW_ATOM_STR)) {
247  posix_flags |= O_NOFOLLOW;
248 #endif
249 #if HAVE_O_RSYNC
250  } else if (globalcontext_is_term_equal_to_atom_string(glb, flag, O_RSYNC_ATOM_STR)) {
251  posix_flags |= O_RSYNC;
252 #endif
253 #if HAVE_O_TTY_INIT
254  } else if (globalcontext_is_term_equal_to_atom_string(glb, flag, O_TTY_INIT_ATOM_STR)) {
255  posix_flags |= O_TTY_INIT;
256 #endif
257  } else {
258  RAISE_ERROR(BADARG_ATOM);
259  }
260  flags = term_get_list_tail(flags);
261  }
262 
263  int ok;
264  char *path = interop_term_to_string(path_term, &ok);
265  if (UNLIKELY(!ok)) {
266  RAISE_ERROR(BADARG_ATOM);
267  }
268 
269  term result;
270  int fd;
271  if (argc == 3) {
272  term mode_term = argv[2];
273  if (UNLIKELY(!term_is_integer(mode_term))) {
274  free(path);
275  RAISE_ERROR(BADARG_ATOM);
276  }
277  int mode = term_to_int(mode_term);
278  fd = open(path, posix_flags, mode);
279  } else {
280  if (UNLIKELY(posix_flags & O_CREAT)) {
281  free(path);
282  RAISE_ERROR(BADARG_ATOM);
283  }
284  fd = open(path, posix_flags);
285  }
286  free(path);
287  if (UNLIKELY(fd < 0)) {
288  // Return an error.
289  if (UNLIKELY(memory_ensure_free_opt(ctx, TUPLE_SIZE(2), MEMORY_CAN_SHRINK) != MEMORY_GC_OK)) {
290  RAISE_ERROR(OUT_OF_MEMORY_ATOM);
291  }
292  result = term_alloc_tuple(2, &ctx->heap);
293  term_put_tuple_element(result, 0, ERROR_ATOM);
294  term_put_tuple_element(result, 1, posix_errno_to_term(errno, glb));
295  } else {
296  // Return a resource object
297  struct PosixFd *fd_obj = enif_alloc_resource(glb->posix_fd_resource_type, sizeof(struct PosixFd));
298  if (IS_NULL_PTR(fd_obj)) {
299  RAISE_ERROR(OUT_OF_MEMORY_ATOM);
300  }
301  fd_obj->fd = fd;
302  fd_obj->selecting_process_id = INVALID_PROCESS_ID;
303  if (UNLIKELY(memory_ensure_free_opt(ctx, TUPLE_SIZE(2) + TERM_BOXED_RESOURCE_SIZE, MEMORY_CAN_SHRINK) != MEMORY_GC_OK)) {
304  RAISE_ERROR(OUT_OF_MEMORY_ATOM);
305  }
306  term obj = term_from_resource(fd_obj, &ctx->heap);
307  result = term_alloc_tuple(2, &ctx->heap);
308  term_put_tuple_element(result, 0, OK_ATOM);
309  term_put_tuple_element(result, 1, obj);
310  }
311 
312  return result;
313 }

315 static term nif_atomvm_posix_close(Context *ctx, int argc, term argv[])
316 {
317  UNUSED(argc);
318 
319  term result = OK_ATOM;
320 
321  void *fd_obj_ptr;
322  if (UNLIKELY(!enif_get_resource(erl_nif_env_from_context(ctx), argv[0], ctx->global->posix_fd_resource_type, &fd_obj_ptr))) {
323  RAISE_ERROR(BADARG_ATOM);
324  }
325  struct PosixFd *fd_obj = (struct PosixFd *) fd_obj_ptr;
326  if (fd_obj->fd != CLOSED_FD) {
327  if (fd_obj->selecting_process_id != INVALID_PROCESS_ID) {
328  fprintf(stderr, "Calling close on a selectable posix file, missing call to posix_select_stop?\n");
329  }
330  if (UNLIKELY(close(fd_obj->fd) < 0)) {
331  fd_obj->fd = CLOSED_FD; // even if bad things happen, do not close twice.
332  if (UNLIKELY(memory_ensure_free_opt(ctx, TUPLE_SIZE(2), MEMORY_CAN_SHRINK) != MEMORY_GC_OK)) {
333  RAISE_ERROR(OUT_OF_MEMORY_ATOM);
334  }
335  result = term_alloc_tuple(2, &ctx->heap);
336  term_put_tuple_element(result, 0, ERROR_ATOM);
337  term_put_tuple_element(result, 1, posix_errno_to_term(errno, ctx->global));
338  }
339  fd_obj->fd = CLOSED_FD;
340  }
341 
342  return result;
343 }

345 static term nif_atomvm_posix_read(Context *ctx, int argc, term argv[])
346 {
347  GlobalContext *glb = ctx->global;
348  UNUSED(argc);
349  term count_term = argv[1];
350  VALIDATE_VALUE(count_term, term_is_integer);
351  int count = term_to_int(count_term);
352 
353  void *fd_obj_ptr;
354  if (UNLIKELY(!enif_get_resource(erl_nif_env_from_context(ctx), argv[0], glb->posix_fd_resource_type, &fd_obj_ptr))) {
355  RAISE_ERROR(BADARG_ATOM);
356  }
357  struct PosixFd *fd_obj = (struct PosixFd *) fd_obj_ptr;
358  if (UNLIKELY(memory_ensure_free_opt(ctx, term_binary_data_size_in_terms(count) + BINARY_HEADER_SIZE + TERM_BOXED_SUB_BINARY_SIZE + TUPLE_SIZE(2), MEMORY_CAN_SHRINK) != MEMORY_GC_OK)) {
359  RAISE_ERROR(OUT_OF_MEMORY_ATOM);
360  }
361  term bin_term = term_create_uninitialized_binary(count, &ctx->heap, glb);
362  int res = read(fd_obj->fd, (void *) term_binary_data(bin_term), count);
363  if (UNLIKELY(res < 0)) {
364  // Return an error.
365  if (UNLIKELY(memory_ensure_free_opt(ctx, TUPLE_SIZE(2), MEMORY_CAN_SHRINK) != MEMORY_GC_OK)) {
366  RAISE_ERROR(OUT_OF_MEMORY_ATOM);
367  }
368  term ret = term_alloc_tuple(2, &ctx->heap);
369  term_put_tuple_element(ret, 0, ERROR_ATOM);
370  term_put_tuple_element(ret, 1, posix_errno_to_term(errno, glb));
371  return ret;
372  }
373  if (res == 0) {
374  return globalcontext_make_atom(glb, ATOM_STR("\x3", "eof"));
375  }
376  if (res < count) {
377  bin_term = term_alloc_sub_binary(bin_term, 0, res, &ctx->heap);
378  }
379  term result = term_alloc_tuple(2, &ctx->heap);
380  term_put_tuple_element(result, 0, OK_ATOM);
381  term_put_tuple_element(result, 1, bin_term);
382 
383  return result;
384 }

386 static term nif_atomvm_posix_write(Context *ctx, int argc, term argv[])
387 {
388  UNUSED(argc);
389  term data_term = argv[1];
390  VALIDATE_VALUE(data_term, term_is_binary);
391 
392  void *fd_obj_ptr;
393  if (UNLIKELY(!enif_get_resource(erl_nif_env_from_context(ctx), argv[0], ctx->global->posix_fd_resource_type, &fd_obj_ptr))) {
394  RAISE_ERROR(BADARG_ATOM);
395  }
396  struct PosixFd *fd_obj = (struct PosixFd *) fd_obj_ptr;
397  const char *data = term_binary_data(data_term);
398  unsigned long n = term_binary_size(data_term);
399  term result;
400  int res = write(fd_obj->fd, data, n);
401  if (UNLIKELY(memory_ensure_free_opt(ctx, TUPLE_SIZE(2), MEMORY_CAN_SHRINK) != MEMORY_GC_OK)) {
402  RAISE_ERROR(OUT_OF_MEMORY_ATOM);
403  }
404  result = term_alloc_tuple(2, &ctx->heap);
405  if (res < 0) {
406  term_put_tuple_element(result, 0, ERROR_ATOM);
407  term_put_tuple_element(result, 1, posix_errno_to_term(errno, ctx->global));
408  } else {
409  term_put_tuple_element(result, 0, OK_ATOM);
410  term_put_tuple_element(result, 1, term_from_int(res));
411  }
412 
413  return result;
414 }

416 static term nif_atomvm_posix_select(Context *ctx, term argv[], enum ErlNifSelectFlags mode)
417 {
418  term process_pid_term = argv[1];
419  VALIDATE_VALUE(process_pid_term, term_is_pid);
420  int32_t process_pid = term_to_local_process_id(process_pid_term);
421  term select_ref_term = argv[2];
422  if (select_ref_term != UNDEFINED_ATOM) {
423  VALIDATE_VALUE(select_ref_term, term_is_reference);
424  }
425  void *fd_obj_ptr;
426  if (UNLIKELY(!enif_get_resource(erl_nif_env_from_context(ctx), argv[0], ctx->global->posix_fd_resource_type, &fd_obj_ptr))) {
427  RAISE_ERROR(BADARG_ATOM);
428  }
429  struct PosixFd *fd_obj = (struct PosixFd *) fd_obj_ptr;
430  ErlNifEnv *env = erl_nif_env_from_context(ctx);
431  if (fd_obj->selecting_process_id != process_pid && fd_obj->selecting_process_id != INVALID_PROCESS_ID) {
432  if (UNLIKELY(enif_demonitor_process(env, fd_obj, &fd_obj->selecting_process_monitor) != 0)) {
433  RAISE_ERROR(BADARG_ATOM);
434  }
435  fd_obj->selecting_process_id = INVALID_PROCESS_ID;
436  }
437  // Monitor first as select is less likely to fail and it's less expensive to demonitor
438  // if select fails than to stop select if monitor fails
439  if (fd_obj->selecting_process_id != process_pid) {
440  if (UNLIKELY(enif_monitor_process(env, fd_obj, &process_pid, &fd_obj->selecting_process_monitor) != 0)) {
441  RAISE_ERROR(BADARG_ATOM);
442  }
443  fd_obj->selecting_process_id = process_pid;
444  }
445  if (UNLIKELY(enif_select(env, fd_obj->fd, mode, fd_obj, &process_pid, select_ref_term) < 0)) {
446  enif_demonitor_process(env, fd_obj, &fd_obj->selecting_process_monitor);
447  fd_obj->selecting_process_id = INVALID_PROCESS_ID;
448  RAISE_ERROR(BADARG_ATOM);
449  }
450 
451  return OK_ATOM;
452 }

484 {
485  UNUSED(argc);
486  term path_term = argv[0];
487  term mode_term = argv[1];
488  VALIDATE_VALUE(mode_term, term_is_integer);
