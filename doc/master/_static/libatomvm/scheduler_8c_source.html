29 static void update_timer_wheel(GlobalContext *global)
30 {
31  struct TimerWheel *tw = global->timer_wheel;
32  uint32_t last_seen_millis = global->last_seen_millis;
33 
34  if (timer_wheel_is_empty(tw)) {
35  sys_stop_millis_timer();
36  return;
37  }
38 
39  uint32_t millis_now = sys_millis();
40 
41  if (millis_now < last_seen_millis) {
42  for (uint32_t i = last_seen_millis; i < UINT32_MAX; i++) {
43  timer_wheel_tick(tw);
44  }
45  }
46  for (uint32_t i = last_seen_millis; i < millis_now; i++) {
47  timer_wheel_tick(tw);
48  }
49  global->last_seen_millis = millis_now;
50 }

62 Context *scheduler_do_wait(GlobalContext *global)
63 {
64  do {
65  update_timer_wheel(global);
66  sys_consume_pending_events(global);
67  scheduler_execute_native_handlers(global);
68 
69  update_timer_wheel(global);
70  if (list_is_empty(&global->ready_processes)) {
71  sys_sleep(global);
72  }
73  } while (list_is_empty(&global->ready_processes));
74 
75  struct ListHead *next_ready = list_first(&global->ready_processes);
76  list_remove(next_ready);
77  list_append(&global->ready_processes, next_ready);
78 
79  return GET_LIST_ENTRY(next_ready, Context, processes_list_head);
80 }

142 void scheduler_set_timeout(Context *ctx, avm_int64_t timeout)
143 {
144  GlobalContext *glb = ctx->global;
145 
146  ctx->flags |= WaitingTimeout;
147 
148  struct TimerWheel *tw = glb->timer_wheel;
149 
150  if (timer_wheel_is_empty(tw)) {
151  sys_start_millis_timer();
152  }
153 
154  struct TimerWheelItem *twi = &ctx->timer_wheel_head;
155  if (UNLIKELY(twi->callback)) {
156  AVM_ABORT();
157  }
158 
159  uint64_t expiry = timer_wheel_expiry_to_monotonic(tw, timeout);
160  timer_wheel_item_init(twi, scheduler_timeout_callback, expiry);
161 
162  timer_wheel_insert(tw, twi);
163 }
