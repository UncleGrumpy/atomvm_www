48 static int update_timer_list(GlobalContext *global)
49 {
50  struct TimerList *tw = &global->timer_list;
51  if (timer_list_is_empty(tw)) {
52  // Do not fetch the current date if there is no timer
53  return -1;
54  }
55  uint64_t millis_now = sys_monotonic_millis();
56  timer_list_next(tw, millis_now, scheduler_timeout_callback);
57  if (tw->next_timer == 0) {
58  return -1;
59  }
60  uint64_t wait_timeout = tw->next_timer - millis_now;
61  if (wait_timeout > INT_MAX) {
62  wait_timeout = INT_MAX;
63  }
64  return (int) wait_timeout;
65 }

67 Context *scheduler_wait(Context *ctx)
68 {
69 #ifdef DEBUG_PRINT_READY_PROCESSES
70  debug_print_processes_list(global->ready_processes);
71 #endif
72  GlobalContext *global = ctx->global;
73  SMP_SPINLOCK_LOCK(&global->processes_spinlock);
74  context_update_flags(ctx, ~Running, NoFlags);
75  // Remove from running list, but do not remove from ready list.
76  if (!context_get_flags(ctx, Ready)) {
77  list_remove(&ctx->processes_list_head);
78  list_append(&global->waiting_processes, &ctx->processes_list_head);
79  }
80  SMP_SPINLOCK_UNLOCK(&global->processes_spinlock);
81 
82  return scheduler_run(global);
83 }

85 static void scheduler_process_native_signal_messages(Context *ctx)
86 {
87  // This mirrors PROCESS_SIGNAL_MESSAGES macro of emulated processes, but
88  // for native processes.
89  MailboxMessage *signal_message = mailbox_process_outer_list(&ctx->mailbox);
90  while (signal_message) {
91  if (signal_message->type == KillSignal) {
92  struct TermSignal *kill_signal = CONTAINER_OF(signal_message, struct TermSignal, base);
93  context_process_kill_signal(ctx, kill_signal);
94  } else if (signal_message->type == ProcessInfoRequestSignal) {
95  struct BuiltInAtomRequestSignal *request_signal
96  = CONTAINER_OF(signal_message, struct BuiltInAtomRequestSignal, base);
97  context_process_process_info_request_signal(ctx, request_signal);
98  }
99  MailboxMessage *next = signal_message->next;
100  mailbox_message_dispose(signal_message, &ctx->heap);
101  signal_message = next;
102  }
103 }

105 static Context *scheduler_run0(GlobalContext *global)
106 {
107  // This function should return a new process to run.
108  // If running_schedulers is greater than online_schedulers, take the
109  // opportunity to end the scheduler, in which case the function returns
110  // NULL
111  Context *result = NULL;
112 
113 #ifndef AVM_NO_SMP
114  SMP_MUTEX_LOCK(global->schedulers_mutex);
115  bool is_waiting = !global->waiting_scheduler;
116  if (is_waiting) {
117  global->waiting_scheduler = true;
118  }
119  bool main_thread = smp_is_main_thread(global);
120 #endif
121  do {
122 #ifndef AVM_NO_SMP
123  // We keep every scheduler threads but one in condition variable.
124  // `is_waiting` holds for the one that leaves this loop and it
125  // processes the timer, the scheduling and the system events in this
126  // order. If a process is to be scheduled (native or not), it signals
127  // the condition variable so another thread can take over.
128  // When there is nothing to do, the scheduler thread that runs the
129  // timer (for which is_waiting is true) waits into `sys_poll_events`
130  // until the timer expires or it is signaled with `sys_signal`.
131  do {
132  if (main_thread && global->scheduler_stop_all) {
133  while (global->running_schedulers > 1) {
134  // Wake a thread and join it.
135  if (is_waiting) {
136  global->waiting_scheduler = false;
137  } else {
138  sys_signal(global);
139  }
140  smp_condvar_signal(global->schedulers_cv);
141  smp_condvar_wait(global->schedulers_cv, global->schedulers_mutex);
142  }
143  global->running_schedulers = 0;
144  global->waiting_scheduler = false;
145  SMP_MUTEX_UNLOCK(global->schedulers_mutex);
146  return NULL;
147  }
148  if (!main_thread
149  && (global->scheduler_stop_all
150  || global->running_schedulers > global->online_schedulers)) {
151  global->running_schedulers--;
152  if (is_waiting) {
153  global->waiting_scheduler = false;
154  } else {
155  sys_signal(global);
156  }
157  smp_condvar_signal(global->schedulers_cv);
158  SMP_MUTEX_UNLOCK(global->schedulers_mutex);
159  return NULL;
160  }
161  if (!is_waiting) {
162  // Before entering the condition variable, signal the poll events
163  // so the thread polling on events can check the ready queue.
164  sys_signal(global);
165  smp_condvar_wait(global->schedulers_cv, global->schedulers_mutex);
166  is_waiting = !global->waiting_scheduler;
167  if (is_waiting) {
168  global->waiting_scheduler = true;
169  }
170  }
171  } while (!is_waiting);
172  SMP_MUTEX_UNLOCK(global->schedulers_mutex);
173 #else
174  if (global->scheduler_stop_all) {
175  return NULL;
176  }
177 #endif
178  // Only one scheduler is running in this section, using the
179  // condition variable. It nevertheless needs to wait on the timer_mutex
180  // as the timer wheel can be modified by another process.
181  SMP_SPINLOCK_LOCK(&global->timer_spinlock);
182  int32_t wait_timeout = update_timer_list(global);
183  SMP_SPINLOCK_UNLOCK(&global->timer_spinlock);
184 
185  SMP_SPINLOCK_LOCK(&global->processes_spinlock);
186  // Pick first ready which is not running.
187  struct ListHead *next_ready = list_first(&global->ready_processes);
188  while (next_ready != &global->ready_processes) {
189  result = GET_LIST_ENTRY(next_ready, Context, processes_list_head);
190  if (!(result->flags & Running)) {
191  list_remove(next_ready);
192  context_update_flags(result, ~Ready, Running);
193  if (result->native_handler) {
194  // Native handlers are marked as waiting
195  list_append(&global->waiting_processes, next_ready);
196  } else {
197  list_append(&global->running_processes, next_ready);
198  }
199  break;
200  }
201  next_ready = next_ready->next;
202  result = NULL;
203  }
204  SMP_SPINLOCK_UNLOCK(&global->processes_spinlock);
205 
206  if (result == NULL && !global->scheduler_stop_all) {
207  sys_poll_events(global, wait_timeout);
208  } else {
209  sys_poll_events(global, SYS_POLL_EVENTS_DO_NOT_WAIT);
210  }
211  SMP_MUTEX_LOCK(global->schedulers_mutex);
212  } while (result == NULL);
213 
214 #ifndef AVM_NO_SMP
215  global->waiting_scheduler = false;
216  smp_condvar_signal(global->schedulers_cv);
217  SMP_MUTEX_UNLOCK(global->schedulers_mutex);
218 #endif
219 
220  return result;
221 }

223 Context *scheduler_run(GlobalContext *global)
224 {
225  // Outer loop to process native contexts.
226  Context *result = NULL;
227  do {
228  result = scheduler_run0(global);
229  if (result == NULL) {
230  break;
231  }
232 
233  if (result->native_handler) {
234  // process signal messages and also empty outer list to inner list.
235  scheduler_process_native_signal_messages(result);
236  if (!(result->flags & Killed)) {
237  if (mailbox_has_next(&result->mailbox)) {
238  if (result->native_handler(result) == NativeContinue) {
239  // If native handler has memory fragments, garbage collect
240  // them
241  if (result->heap.root->next) {
242  if (UNLIKELY(memory_ensure_free_opt(result, 0, MEMORY_FORCE_SHRINK) != MEMORY_GC_OK)) {
243  fprintf(stderr, "Out of memory error in native handler\n");
244  AVM_ABORT();
245  }
246  }
247  context_update_flags(result, ~Running, NoFlags);
248  // The context was marked ready for the first message
249  // However, another message may have arrived before it
250  // was marked running in scheduler_run0, so there can
251  // be several messages in the mailbox. Yet, the handler
252  // may have processed only one message. So we rechedule
253  // to make sure the handler process all messages.
254  if (mailbox_has_next(&result->mailbox)) {
255  scheduler_make_ready(result);
256  }
257  } else {
258  scheduler_terminate(result);
259  }
260  } else {
261  // Don't call the native handler if the mailbox is empty
262  // so native handlers can safely expect the first call to
263  // `mailbox_first` will return a message.
264  //
265  // Indeed, a native process can be signaled and be made
266  // ready, and in this case the mailbox may only contain
267  // signal messages that are processed and removed by
268  // `scheduler_process_native_signal_messages`.
269  context_update_flags(result, ~Running, NoFlags);
270  }
271  }
272  result = NULL; // Schedule next process (native or not)
273  }
274  } while (result == NULL);
275  return result;
276 }

278 Context *scheduler_next(GlobalContext *global, Context *c)
279 {
280  c->reductions += DEFAULT_REDUCTIONS_AMOUNT;
281 
282  // Remove c from running and append it at the end of ready list
283  // c could already be in ready queue, if it received a message.
284  SMP_SPINLOCK_LOCK(&global->processes_spinlock);
285  list_remove(&c->processes_list_head);
286  list_append(&global->ready_processes, &c->processes_list_head);
287  context_update_flags(c, ~Running, NoFlags);
288  SMP_SPINLOCK_UNLOCK(&global->processes_spinlock);
289 
290  // Schedule.
291  return scheduler_run(global);
292 }

294 static void scheduler_make_ready(Context *ctx)
295 {
296  GlobalContext *global = ctx->global;
297  SMP_SPINLOCK_LOCK(&global->processes_spinlock);
298  if (context_get_flags(ctx, Killed)) {
299  SMP_SPINLOCK_UNLOCK(&global->processes_spinlock);
300  return;
301  }
302  list_remove(&ctx->processes_list_head);
303 #ifndef AVM_NO_SMP
304  bool waiting_scheduler = global->waiting_scheduler;
305  if (!waiting_scheduler) {
306  // Start a new scheduler if none are going to take this process.
307  if (SMP_MUTEX_TRYLOCK(global->schedulers_mutex)) {
308  if (global->running_schedulers > 0
309  && global->running_schedulers < global->online_schedulers
310  && !context_get_flags(ctx, Running)) {
311  global->running_schedulers++;
312  smp_scheduler_start(global);
313  }
314  SMP_MUTEX_UNLOCK(global->schedulers_mutex);
315  }
316  }
317 #endif
318  // Move to ready queue (from waiting or running)
319  // The process may be running (it would be signaled), so mark it
320  // as ready
321  context_update_flags(ctx, ~NoFlags, Ready);
322  list_append(&global->ready_processes, &ctx->processes_list_head);
323  SMP_SPINLOCK_UNLOCK(&global->processes_spinlock);
324 #ifndef AVM_NO_SMP
325  if (waiting_scheduler) {
326  sys_signal(global);
327  }
328 #endif
329 }

367 void scheduler_set_timeout(Context *ctx, avm_int64_t timeout)
368 {
369  GlobalContext *glb = ctx->global;
370  uint64_t millis_now = sys_monotonic_millis();
371  uint64_t expiry = millis_now + timeout;
372 
373  context_update_flags(ctx, ~NoFlags, WaitingTimeout);
374  struct TimerList *tw = &glb->timer_list;
375  struct TimerListItem *twi = &ctx->timer_list_head;
376  timer_list_item_init(twi, expiry);
377 
378  SMP_SPINLOCK_LOCK(&glb->timer_spinlock);
379  timer_list_insert(tw, twi);
380  SMP_SPINLOCK_UNLOCK(&glb->timer_spinlock);
381 
382 #ifndef AVM_NO_SMP
383  if (glb->waiting_scheduler) {
384  sys_signal(glb);
