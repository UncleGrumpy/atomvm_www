66GlobalContext *globalcontext_new()
67{
68 GlobalContext *glb = malloc(sizeof(GlobalContext));
69 if (IS_NULL_PTR(glb)) {
70 return NULL;
71 }
72 list_init(&glb->ready_processes);
73 list_init(&glb->running_processes);
74 list_init(&glb->waiting_processes);
75#ifndef AVM_NO_SMP
76 smp_spinlock_init(&glb->processes_spinlock);
77#endif
78 synclist_init(&glb->avmpack_data);
79 synclist_init(&glb->refc_binaries);
80 synclist_init(&glb->processes_table);
81 synclist_init(&glb->registered_processes);
82 synclist_init(&glb->listeners);
83 synclist_init(&glb->resource_types);
84 synclist_init(&glb->select_events);
85
86 glb->last_process_id = 0;
87
88 glb->atoms_table = atomshashtable_new();
89 if (IS_NULL_PTR(glb->atoms_table)) {
90 free(glb);
91 return NULL;
92 }
93 glb->atoms_ids_table = valueshashtable_new();
94 if (IS_NULL_PTR(glb->atoms_ids_table)) {
95 free(glb->atoms_table);
96 free(glb);
97 return NULL;
98 }
99
100 defaultatoms_init(glb);
101
102 glb->modules_by_index = NULL;
103 glb->loaded_modules_count = 0;
104 glb->modules_table = atomshashtable_new();
105 if (IS_NULL_PTR(glb->modules_table)) {
106 free(glb->atoms_ids_table);
107 free(glb->atoms_table);
108 free(glb);
109 return NULL;
110 }
111#ifndef AVM_NO_SMP
112 glb->modules_lock = smp_rwlock_create();
113 if (IS_NULL_PTR(glb->modules_lock)) {
114 free(glb->modules_table);
115 free(glb->atoms_ids_table);
116 free(glb->atoms_table);
117 free(glb);
118 return NULL;
119 }
120#endif
121
122 timer_list_init(&glb->timer_list);
123#ifndef AVM_NO_SMP
124 smp_spinlock_init(&glb->timer_spinlock);
125#endif
126
127 glb->ref_ticks = 0;
128#if !defined(AVM_NO_SMP) && ATOMIC_LLONG_LOCK_FREE != 2
129 smp_spinlock_init(&glb->ref_ticks_spinlock);
130#endif
131
132#if HAVE_OPEN && HAVE_CLOSE
133 ErlNifEnv env;
134 erl_nif_env_partial_init_from_globalcontext(&env, glb);
135 glb->posix_fd_resource_type = enif_init_resource_type(&env, "posix_fd", &posix_fd_resource_type_init, ERL_NIF_RT_CREATE, NULL);
136 if (IS_NULL_PTR(glb->posix_fd_resource_type)) {
137#ifndef AVM_NO_SMP
138 smp_rwlock_destroy(glb->modules_lock);
139#endif
140 free(glb->modules_table);
141 free(glb->atoms_ids_table);
142 free(glb->atoms_table);
143 free(glb);
144 return NULL;
145 }
146#endif
147
148 sys_init_platform(glb);
149
150#ifndef AVM_NO_SMP
151 glb->schedulers_mutex = smp_mutex_create();
152 if (IS_NULL_PTR(glb->schedulers_mutex)) {
153#if HAVE_OPEN && HAVE_CLOSE
154 resource_type_destroy(glb->posix_fd_resource_type);
155#endif
156 smp_rwlock_destroy(glb->modules_lock);
157 free(glb->modules_table);
158 free(glb->atoms_ids_table);
159 free(glb->atoms_table);
160 free(glb);
161 return NULL;
162 }
163 glb->schedulers_cv = smp_condvar_create();
164 if (IS_NULL_PTR(glb->schedulers_cv)) {
165 smp_mutex_destroy(glb->schedulers_mutex);
166#if HAVE_OPEN && HAVE_CLOSE
167 resource_type_destroy(glb->posix_fd_resource_type);
168#endif
169 smp_rwlock_destroy(glb->modules_lock);
170 free(glb->modules_table);
171 free(glb->atoms_ids_table);
172 free(glb->atoms_table);
173 free(glb);
174 return NULL;
175 }
176 glb->online_schedulers = smp_get_online_processors();
177 glb->running_schedulers = 0;
178 glb->waiting_scheduler = false;
179
180 smp_spinlock_init(&glb->env_spinlock);
181#endif
182 glb->scheduler_stop_all = false;
183
184 return glb;
185}

187COLD_FUNC void globalcontext_destroy(GlobalContext *glb)
188{
189 sys_free_platform(glb);
190
191 struct ListHead *item;
192 struct ListHead *tmp;
193
194 struct ListHead *open_avm_packs = synclist_nolock(&glb->avmpack_data);
195 MUTABLE_LIST_FOR_EACH (item, tmp, open_avm_packs) {
196 struct AVMPackData *avmpack_data = GET_LIST_ENTRY(item, struct AVMPackData, avmpack_head);
197 avmpack_data_destroy(avmpack_data, glb);
198 }
199 synclist_destroy(&glb->avmpack_data);
200
201 struct ListHead *listeners = synclist_nolock(&glb->listeners);
202 MUTABLE_LIST_FOR_EACH (item, tmp, listeners) {
203 sys_listener_destroy(item);
204 }
205 synclist_destroy(&glb->listeners);
206
207 // Destroy select events before resources
208 struct ListHead *select_events = synclist_nolock(&glb->select_events);
209 MUTABLE_LIST_FOR_EACH (item, tmp, select_events) {
210 struct SelectEvent *select_event = GET_LIST_ENTRY(item, struct SelectEvent, head);
211 free((void *) select_event);
212 }
213 synclist_destroy(&glb->select_events);
214
215 // Destroy refc binaries including resources
216 struct ListHead *refc_binaries = synclist_nolock(&glb->refc_binaries);
217 MUTABLE_LIST_FOR_EACH (item, tmp, refc_binaries) {
218 struct RefcBinary *refc = GET_LIST_ENTRY(item, struct RefcBinary, head);
219 refc_binary_destroy(refc, glb);
220 }
221 synclist_destroy(&glb->refc_binaries);
222
223 // Destroy resource types
224 struct ListHead *resource_types = synclist_nolock(&glb->resource_types);
225 MUTABLE_LIST_FOR_EACH (item, tmp, resource_types) {
226 struct ResourceType *resource_type = GET_LIST_ENTRY(item, struct ResourceType, head);
227 resource_type_destroy(resource_type);
228 }
229 synclist_destroy(&glb->resource_types);
230
231#ifndef AVM_NO_SMP
232 smp_condvar_destroy(glb->schedulers_cv);
233 smp_mutex_destroy(glb->schedulers_mutex);
234 smp_rwlock_destroy(glb->modules_lock);
235#endif
236 synclist_destroy(&glb->registered_processes);
237 synclist_destroy(&glb->processes_table);
238
239 free(glb);
240}

242Context *globalcontext_get_process_nolock(GlobalContext *glb, int32_t process_id)
243{
244 struct ListHead *item;
245 Context *p = NULL;
246
247 struct ListHead *processes_table_list = synclist_nolock(&glb->processes_table);
248 LIST_FOR_EACH (item, processes_table_list) {
249 p = GET_LIST_ENTRY(item, Context, processes_table_head);
250
251 if (p->process_id == process_id) {
252 return p;
253 }
254 }
255
256 return NULL;
257}

319bool globalcontext_register_process(GlobalContext *glb, int atom_index, int local_process_id)
320{
321 struct ListHead *registered_processes_list = synclist_wrlock(&glb->registered_processes);
322 struct ListHead *item;
323 LIST_FOR_EACH (item, registered_processes_list) {
324 const struct RegisteredProcess *registered_process = GET_LIST_ENTRY(item, struct RegisteredProcess, registered_processes_list_head);
325 if (registered_process->atom_index == atom_index) {
326 synclist_unlock(&glb->registered_processes);
327 return false;
328 }
329 }
330
331 struct RegisteredProcess *registered_process = malloc(sizeof(struct RegisteredProcess));
332 if (IS_NULL_PTR(registered_process)) {
333 fprintf(stderr, "Failed to allocate memory: %s:%i.\n", __FILE__, __LINE__);
334 AVM_ABORT();
335 }
336 registered_process->atom_index = atom_index;
337 registered_process->local_process_id = local_process_id;
338
339 list_append(registered_processes_list, &registered_process->registered_processes_list_head);
340 synclist_unlock(&glb->registered_processes);
341
342 return true;
343}

402int globalcontext_insert_atom_maybe_copy(GlobalContext *glb, AtomString atom_string, int copy)
403{
404 struct AtomsHashTable *htable = glb->atoms_table;
405
406 unsigned long atom_index = atomshashtable_get_value(htable, atom_string, ULONG_MAX);
407 if (atom_index == ULONG_MAX) {
408 if (copy) {
409 uint8_t len = *((uint8_t *) atom_string);
410 uint8_t *buf = malloc(1 + len);
411 if (UNLIKELY(IS_NULL_PTR(buf))) {
412 fprintf(stderr, "Unable to allocate memory for atom string\n");
413 AVM_ABORT();
414 }
415 memcpy(buf, atom_string, 1 + len);
416 atom_string = buf;
417 }
418 atom_index = htable->count;
419 if (!atomshashtable_insert(htable, atom_string, atom_index)) {
420 return -1;
421 }
422 if (!valueshashtable_insert(glb->atoms_ids_table, atom_index, (unsigned long) atom_string)) {
423 return -1;
424 }
425 }
426
427 return (int) atom_index;
428}

460int globalcontext_insert_module(GlobalContext *global, Module *module)
461{
462 SMP_RWLOCK_WRLOCK(global->modules_lock);
463 AtomString module_name_atom = module_get_atom_string_by_id(module, 1);
464 if (!atomshashtable_insert(global->modules_table, module_name_atom, TO_ATOMSHASHTABLE_VALUE(module))) {
465 SMP_RWLOCK_UNLOCK(global->modules_lock);
466 return -1;
467 }
468
469 int module_index = global->loaded_modules_count;
470
471 Module **new_modules_by_index = calloc(module_index + 1, sizeof(Module *));
472 if (IS_NULL_PTR(new_modules_by_index)) {
473 fprintf(stderr, "Failed to allocate memory: %s:%i.\n", __FILE__, __LINE__);
474 SMP_RWLOCK_UNLOCK(global->modules_lock);
475 AVM_ABORT();
476 }
477 if (global->modules_by_index) {
478 for (int i = 0; i < module_index; i++) {
479 new_modules_by_index[i] = global->modules_by_index[i];
480 }
481 free(global->modules_by_index);
482 }
483
484 module->module_index = module_index;
485
486 global->modules_by_index = new_modules_by_index;
487 global->modules_by_index[module_index] = module;
488 global->loaded_modules_count++;
489 SMP_RWLOCK_UNLOCK(global->modules_lock);
490
491 return module_index;
492}

494Module *globalcontext_get_module(GlobalContext *global, AtomString module_name_atom)
495{
496 Module *found_module = (Module *) atomshashtable_get_value(global->modules_table, module_name_atom, (unsigned long) NULL);
497
498 if (!found_module) {
499 char *module_name = malloc(256 + 5);
500 if (IS_NULL_PTR(module_name)) {
501 return NULL;
502 }
503
504 atom_string_to_c(module_name_atom, module_name, 256);
505 strcat(module_name, ".beam");
506 Module *loaded_module = sys_load_module(global, module_name);
507 free(module_name);
508
509 if (UNLIKELY(!loaded_module || (globalcontext_insert_module(global, loaded_module) < 0))) {
510 return NULL;
511 }
512
513 return loaded_module;
514 }
515
516 return found_module;
517}

527bool globalcontext_demonitor(GlobalContext *global, uint64_t ref_ticks)
528{
529 struct ListHead *pitem;
530
531 struct ListHead *processes_table_list = synclist_wrlock(&global->processes_table);
532 LIST_FOR_EACH (pitem, processes_table_list) {
533 Context *p = GET_LIST_ENTRY(pitem, Context, processes_table_head);
534
535 struct ListHead *item;
536 LIST_FOR_EACH (item, &p->monitors_head) {
537 struct Monitor *monitor = GET_LIST_ENTRY(item, struct Monitor, monitor_list_head);
538 if (monitor->ref_ticks == ref_ticks) {
539 list_remove(&monitor->monitor_list_head);
540 free(monitor);
541 synclist_unlock(&global->processes_table);
542 return true;
543 }
544 }
545 }
546
547 synclist_unlock(&global->processes_table);
548 return false;
549}
