144static inline size_t memory_heap_fragment_memory_size(const HeapFragment *fragment)
145{
146 size_t result = 0;
147 do {
148 result += fragment->heap_end - fragment->storage;
149 fragment = fragment->next;
150 } while (fragment);
151 return result;
152}
153
160static inline size_t memory_heap_memory_size(const Heap *heap)
161{
162 size_t result = heap->heap_end - heap->heap_start;
163 if (heap->root->next) {
164 result += memory_heap_fragment_memory_size(heap->root->next);
165 }
166 return result;
167}

177static inline MALLOC_LIKE term *memory_heap_alloc(Heap *heap, size_t size)
178{
179 term *allocated = heap->heap_ptr;
180 heap->heap_ptr += size;
181#ifdef DEBUG_HEAP_ALLOC
182 if (UNLIKELY(heap->heap_ptr > heap->heap_end)) {
183 fprintf(stderr, "Tried to allocate heap terms beyond end of heap, size=%u, overflow=%u, heap_size=%u\n", (unsigned) size, (unsigned) (heap->heap_ptr - heap->heap_end), (unsigned) (heap->heap_end - heap->heap_start));
184 AVM_ABORT();
185 }
186#endif

188 return allocated;

321static inline void memory_destroy_heap_fragment(HeapFragment *fragment)
322{
323 while (fragment->next) {
324 HeapFragment *next = fragment->next;
325 free(fragment);
326 fragment = next;
327 }
328 free(fragment);
329}
