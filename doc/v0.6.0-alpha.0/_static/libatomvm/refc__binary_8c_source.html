36struct RefcBinary *refc_binary_create_resource(size_t size, struct ResourceType *resource_type)
37{
38 size_t n = sizeof(struct RefcBinary) + size;
39 struct RefcBinary *refc = malloc(n);
40 if (IS_NULL_PTR(refc)) {
41 return NULL;
42 }
43 list_init(&refc->head);
44 refc->ref_count = 1;
45 refc->size = size;
46 refc->resource_type = resource_type;
47
48 return refc;
49}

66bool refc_binary_decrement_refcount(struct RefcBinary *refc, struct GlobalContext *global)
67{
68 if (--refc->ref_count == 0) {
69 synclist_remove(&global->refc_binaries, &refc->head);
70 refc_binary_destroy(refc, global);
71 return true;
72 }
73 return false;
74}

76void refc_binary_destroy(struct RefcBinary *refc, struct GlobalContext *global)
77{
78 UNUSED(global);
79
80 if (refc->resource_type) {
81 if (refc->resource_type->down) {
82 // There may be monitors associated with this resource.
83 destroy_resource_monitors(refc, global);
84 }
85 if (refc->resource_type->dtor) {
86 ErlNifEnv env;
87 erl_nif_env_partial_init_from_globalcontext(&env, global);
88 refc->resource_type->dtor(&env, (void *) &refc->data);
89 }
90 }
91 free(refc);
92}

94term refc_binary_create_binary_info(Context *ctx)
95{
96 size_t len = 0;
97 struct ListHead *item;
98 struct ListHead *refc_binaries = synclist_wrlock(&ctx->global->refc_binaries);
99 LIST_FOR_EACH (item, refc_binaries) {
100 len++;
101 }
102 if (len == 0) {
103 synclist_unlock(&ctx->global->refc_binaries);
104 return term_nil();
105 }
106 if (memory_ensure_free(ctx, len * (2 + TUPLE_SIZE(2))) != MEMORY_GC_OK) {
107 synclist_unlock(&ctx->global->refc_binaries);
108 return term_invalid_term();
109 }
110 term ret = term_nil();
111 LIST_FOR_EACH (item, refc_binaries) {
112 struct RefcBinary *refc = GET_LIST_ENTRY(item, struct RefcBinary, head);
113 term t = term_alloc_tuple(2, &ctx->heap);
114 term_put_tuple_element(t, 0, term_from_int(refc->size));
115 term_put_tuple_element(t, 1, term_from_int(refc->ref_count));
116 ret = term_list_prepend(t, ret, &ctx->heap);
117 }
118 synclist_unlock(&ctx->global->refc_binaries);
119 return ret;
120}

122size_t refc_binary_total_size(Context *ctx)
123{
124 size_t size = 0;
125 struct ListHead *item;
126 struct ListHead *refc_binaries = synclist_rdlock(&ctx->global->refc_binaries);
127 LIST_FOR_EACH (item, refc_binaries) {
128 struct RefcBinary *refc = GET_LIST_ENTRY(item, struct RefcBinary, head);
129 size += refc->size;
130 }
131 synclist_unlock(&ctx->global->refc_binaries);
132 return size;
133}
