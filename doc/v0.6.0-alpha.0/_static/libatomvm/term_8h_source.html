356{
357 /* boxed: 10 */

359 const term *boxed_value = term_to_const_term_ptr(t);
360 int masked_value = boxed_value[0] & TERM_BOXED_TAG_MASK;

786static inline term term_make_boxed_int(avm_int_t value, Heap *heap)
787{
788 term *boxed_int = memory_heap_alloc(heap, 1 + BOXED_TERMS_REQUIRED_FOR_INT);
789 boxed_int[0] = (BOXED_TERMS_REQUIRED_FOR_INT << 6) | TERM_BOXED_POSITIVE_INTEGER; // OR sign bit
790 boxed_int[1] = value;
791 return ((term) boxed_int) | TERM_BOXED_VALUE_TAG;
792}

794static inline term term_make_boxed_int64(avm_int64_t large_int64, Heap *heap)
795{
796 term *boxed_int = memory_heap_alloc(heap, 1 + BOXED_TERMS_REQUIRED_FOR_INT64);
797 boxed_int[0] = (BOXED_TERMS_REQUIRED_FOR_INT64 << 6) | TERM_BOXED_POSITIVE_INTEGER; // OR sign bit
798 #if BOXED_TERMS_REQUIRED_FOR_INT64 == 1
799 boxed_int[1] = large_int64;
800 #elif BOXED_TERMS_REQUIRED_FOR_INT64 == 2
801 #if __BYTE_ORDER__ == __ORDER_LITTLE_ENDIAN__
802 boxed_int[1] = large_int64;
803 boxed_int[2] = large_int64 >> 32;
804 #elif __BYTE_ORDER__ == __ORDER_BIG_ENDIAN__
805 boxed_int[2] = large_int64;
806 boxed_int[1] = large_int64 >> 32;
807 #else
808 #error "unsupported endianness."
809 #endif
810 #else
811 #error "unsupported configuration."
812 #endif
813 return ((term) boxed_int) | TERM_BOXED_VALUE_TAG;
814}

947{
948 TERM_DEBUG_ASSERT(term_is_binary(t));

977{
