39void port_ensure_available(Context *ctx, size_t size)
40{
41 if (context_avail_free_memory(ctx) < size) {
42 switch (memory_ensure_free_opt(ctx, size, MEMORY_NO_GC)) {
43 case MEMORY_GC_OK:
44 break;
45 case MEMORY_GC_ERROR_FAILED_ALLOCATION:
46 // TODO Improve error handling
47 fprintf(stderr, "Failed to allocate additional heap storage: [%s:%i]\n", __FILE__, __LINE__);
48 AVM_ABORT();
49 case MEMORY_GC_DENIED_ALLOCATION:
50 // TODO Improve error handling
51 fprintf(stderr, "Not permitted to allocate additional heap storage: [%s:%i]\n", __FILE__, __LINE__);
52 AVM_ABORT();
53 }
54 }
55}

57int port_is_standard_port_command(term t)
58{
59 if (!term_is_tuple(t)) {
60 return 0;
61 } else if (term_get_tuple_arity(t) != 3) {
62 return 0;
63 } else {
64 term pid = term_get_tuple_element(t, 0);
65 term ref = term_get_tuple_element(t, 1);
66 if (!term_is_pid(pid)) {
67 return 0;
68 } else if (!term_is_reference(ref)) {
69 return 0;
70 } else {
71 return 1;
72 }
73 }
74}

95term port_heap_create_tuple_n(Heap *heap, size_t num_terms, term *terms)
96{
97 term ret = term_alloc_tuple(num_terms, heap);
98
99 for (size_t i = 0; i < num_terms; ++i) {
100 term_put_tuple_element(ret, i, terms[i]);
101 }
102
103 return ret;
104}
