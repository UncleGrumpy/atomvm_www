44void scan_iff(const void *iff_binary, int buf_size, unsigned long *offsets, unsigned long *sizes)
45{
46 const uint8_t *data = iff_binary;
47
48 memset(offsets, 0, sizeof(unsigned long) * MAX_OFFS);
49 memset(sizes, 0, sizeof(unsigned long) * MAX_SIZES);
50
51 int current_pos = 12;
52
53 uint32_t iff_size = READ_32_ALIGNED(data + 4);
54 int file_size = iff_size;
55 if (UNLIKELY(buf_size < file_size)) {
56 fprintf(stderr, "warning: buffer holding IFF is smaller than IFF size: %i", buf_size);
57 }
58
59 do {
60 struct IFFRecord *current_record = (struct IFFRecord *) (data + current_pos);
61
62 if (!memcmp(current_record->name, "AtU8", 4)) {
63 offsets[AT8U] = current_pos;
64 sizes[AT8U] = ENDIAN_SWAP_32(current_record->size);
65
66 } else if (!memcmp(current_record->name, "Code", 4)) {
67 offsets[CODE] = current_pos;
68 sizes[CODE] = ENDIAN_SWAP_32(current_record->size);
69
70 } else if (!memcmp(current_record->name, "ExpT", 4)) {
71 offsets[EXPT] = current_pos;
72 sizes[EXPT] = ENDIAN_SWAP_32(current_record->size);
73
74 } else if (!memcmp(current_record->name, "LocT", 4)) {
75 offsets[LOCT] = current_pos;
76 sizes[LOCT] = ENDIAN_SWAP_32(current_record->size);
77
78 } else if (!memcmp(current_record->name, "LitT", 4)) {
79 offsets[LITT] = current_pos;
80 sizes[LITT] = ENDIAN_SWAP_32(current_record->size);
81
82 } else if (!memcmp(current_record->name, "LitU", 4)) {
83 offsets[LITU] = current_pos;
84 sizes[LITU] = ENDIAN_SWAP_32(current_record->size);
85
86 } else if (!memcmp(current_record->name, "ImpT", 4)) {
87 offsets[IMPT] = current_pos;

89 } else if (!memcmp(current_record->name, "FunT", 4)) {
90 offsets[FUNT] = current_pos;
91 sizes[FUNT] = ENDIAN_SWAP_32(current_record->size);
92 } else if (!memcmp(current_record->name, "StrT", 4)) {
93 offsets[STRT] = current_pos;
94 sizes[STRT] = ENDIAN_SWAP_32(current_record->size);
95 } else if (!memcmp(current_record->name, "Line", 4)) {
96 offsets[LINT] = current_pos;
97 sizes[LINT] = ENDIAN_SWAP_32(current_record->size);
98 }
99
100 current_pos += iff_align(ENDIAN_SWAP_32(current_record->size) + 8);
101 } while (current_pos < file_size);
102}
