48static unsigned long sdbm_hash(const unsigned char *str, int len)
49{
50 unsigned long hash = 0;
51 int c;
52
53 for (int i = 0; i < len; i++) {
54 c = *str++;
55 hash = c + (hash << 6) + (hash << 16) - hash;
56 }
57
58 return hash;
59}

61struct AtomsHashTable *atomshashtable_new()
62{
63 struct AtomsHashTable *htable = malloc(sizeof(struct AtomsHashTable));
64 if (IS_NULL_PTR(htable)) {
65 return NULL;
66 }
67 htable->buckets = calloc(DEFAULT_SIZE, sizeof(struct HNode *));
68 if (IS_NULL_PTR(htable->buckets)) {
69 free(htable);
70 return NULL;
71 }
72
73 htable->count = 0;
74 htable->capacity = DEFAULT_SIZE;
75
76#ifndef AVM_NO_SMP
77 htable->lock = smp_rwlock_create();
78#endif
79
80 return htable;
81}

83int atomshashtable_insert(struct AtomsHashTable *hash_table, AtomString string, unsigned long value)
84{
85 int alen = atom_string_len(string);
86
87 unsigned long hash = sdbm_hash(string, alen);
88 SMP_WRLOCK(hash_table);
89 long index = hash % hash_table->capacity;
90
91 struct HNode *node = hash_table->buckets[index];
92 if (node) {
93 while (1) {
94 if (atom_are_equals(string, node->key)) {
95 node->value = value;
96 SMP_UNLOCK(hash_table);
97 return 1;
98 }
99
100 if (node->next) {
101 node = node->next;
102 } else {
103 break;
104 }
105 }
106 }
107
108 struct HNode *new_node = malloc(sizeof(struct HNode));
109 if (IS_NULL_PTR(new_node)) {
110 SMP_UNLOCK(hash_table);
111 return 0;
112 }
113 new_node->next = NULL;
114 new_node->key = string;
115 new_node->value = value;
116
117 if (node) {
118 node->next = new_node;
119 } else {
120 hash_table->buckets[index] = new_node;
121 }
122
123 hash_table->count++;
124 SMP_UNLOCK(hash_table);
125 return 1;
126}

128unsigned long atomshashtable_get_value(const struct AtomsHashTable *hash_table, const AtomString string, unsigned long default_value)
129{
130 unsigned long hash = sdbm_hash(string, atom_string_len(string));
131 SMP_RDLOCK(hash_table);
132 long index = hash % hash_table->capacity;
133
134 const struct HNode *node = hash_table->buckets[index];
135 while (node) {
136 if (atom_are_equals(string, node->key)) {
137 unsigned long result = node->value;
138 SMP_UNLOCK(hash_table);
139 return result;
140 }
141
142 node = node->next;
143 }
144
145 SMP_UNLOCK(hash_table);
146 return default_value;
147}
