40NO_DISCARD static inline TempStackResult temp_stack_init(struct TempStack *temp_stack)
41{
42 temp_stack->size = 8;
43 temp_stack->stack_end = ((term *) malloc(temp_stack->size * sizeof(term))) + temp_stack->size;
44 if (IS_NULL_PTR(temp_stack->stack_end)) {
45 return TempStackFailedAlloc;
46 }
47 temp_stack->stack_pos = temp_stack->stack_end;
48
49 return TempStackOk;
50}

52static inline void temp_stack_destroy(struct TempStack *temp_stack)
53{
54 free(temp_stack->stack_end - temp_stack->size);
55}

57NO_DISCARD static TempStackResult temp_stack_grow(struct TempStack *temp_stack)
58{
59 int old_used_size = temp_stack->stack_end - temp_stack->stack_pos;
60 int new_size = temp_stack->size * 2;
61 term *new_stack_end = ((term *) malloc(new_size * sizeof(term))) + new_size;
62 if (IS_NULL_PTR(new_stack_end)) {
63 return TempStackFailedAlloc;
64 }
65 term *new_stack_pos = new_stack_end - old_used_size;
66 memcpy(new_stack_pos, temp_stack->stack_pos, old_used_size * sizeof(term));
67
68 free(temp_stack->stack_end - temp_stack->size);
69 temp_stack->stack_end = new_stack_end;
70 temp_stack->stack_pos = new_stack_pos;
71 temp_stack->size = new_size;
72
73 return TempStackOk;
74}

76static inline int temp_stack_is_empty(const struct TempStack *temp_stack)
77{
78 return temp_stack->stack_end == temp_stack->stack_pos;
79}

81NO_DISCARD static inline TempStackResult temp_stack_push(struct TempStack *temp_stack, term value)
82{
83 if (temp_stack->stack_end - temp_stack->stack_pos == temp_stack->size - 1) {
84 TempStackResult ret = temp_stack_grow(temp_stack);
85 if (UNLIKELY(ret != TempStackOk)) {
86 return ret;
87 }
88 }
89
90 temp_stack->stack_pos--;
91 *temp_stack->stack_pos = value;
92
93 return TempStackOk;
94}
95
96static inline term temp_stack_pop(struct TempStack *temp_stack)
97{
98 term value = *temp_stack->stack_pos;
99 temp_stack->stack_pos++;
100
101 return value;
102}
