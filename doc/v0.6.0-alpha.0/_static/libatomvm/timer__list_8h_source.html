56static inline void timer_list_insert(struct TimerList *tw, struct TimerListItem *item)
57{
58 uint64_t expiry_time = item->expiry_time;
59
60 if (tw->timers == 0 || expiry_time < tw->next_timer) {
61 tw->next_timer = expiry_time;
62 }
63
64 tw->timers++;
65
66 list_append(&tw->head, &item->head);
67}

69static inline void timer_list_remove(struct TimerList *tw, struct TimerListItem *item)
70{
71 if (item->head.next != &item->head) {
72 tw->timers--;
73 list_remove(&item->head);
74 list_init(&item->head);
75 }
76}
