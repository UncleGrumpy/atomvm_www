80 struct AtomTable *atom_table_new()
81 {
82  struct AtomTable *htable = malloc(sizeof(struct AtomTable));
83  if (IS_NULL_PTR(htable)) {
84  return NULL;
85  }
86  htable->buckets = calloc(DEFAULT_SIZE, sizeof(struct HNode *));
87  if (IS_NULL_PTR(htable->buckets)) {
88  free(htable);
89  return NULL;
90  }
91 
92  htable->count = 0;
93  htable->capacity = DEFAULT_SIZE;
94 
95  htable->last_node_group = NULL;
96  htable->first_node_group = new_node_group(htable, DEFAULT_SIZE);
97 
98 #ifndef AVM_NO_SMP
99  htable->lock = smp_rwlock_create();
100 #endif
101 
102  return htable;

148 static unsigned long sdbm_hash(const unsigned char *str, int len)
149 {
150  unsigned long hash = 0;
151  int c;
152 
153  for (int i = 0; i < len; i++) {
154  c = *str++;
155  hash = c + (hash << 6) + (hash << 16) - hash;
156  }
157 
158  return hash;
159 }

239 int atom_table_cmp_using_atom_index(struct AtomTable *table, int t_atom_index, int other_atom_index)
240 {
241  AtomString t_atom_string = atom_table_get_atom_string(table, t_atom_index);
242 
243  int t_atom_len = atom_string_len(t_atom_string);
244  const char *t_atom_data = (const char *) atom_string_data(t_atom_string);
245 
246  AtomString other_atom_string = atom_table_get_atom_string(table, other_atom_index);
247 
248  int other_atom_len = atom_string_len(other_atom_string);
249  const char *other_atom_data = (const char *) atom_string_data(other_atom_string);
250 
251  int cmp_size = (t_atom_len > other_atom_len) ? other_atom_len : t_atom_len;
252 
253  int memcmp_result = memcmp(t_atom_data, other_atom_data, cmp_size);
254 
255  if (memcmp_result == 0) {
256  if (t_atom_len == other_atom_len) {
257  return 0;
258  } else {
259  return (t_atom_len > other_atom_len) ? 1 : -1;
260  }
261  }
262 
263  return memcmp_result;
264 }

322 static inline long insert_node(struct AtomTable *table, struct HNodeGroup *node_group,
323  unsigned long bucket_index, AtomString string)
324 {
325  long new_index = table->count;
326  table->count++;
327 
328  struct HNode *node = &node_group->nodes[new_index - node_group->first_index];
329  table->last_node_group_avail--;
330  init_node(node, string, new_index);
331  insert_node_into_bucket(table, bucket_index, node);

336 static bool do_rehash(struct AtomTable *table, int new_capacity)
337 {
338  int new_size_bytes = sizeof(struct HNode *) * new_capacity;
339  struct HNode **new_buckets = realloc(table->buckets, new_size_bytes);
340  if (IS_NULL_PTR(new_buckets)) {
341  // Allocation failure can be ignored, the hash table will continue with the previous bucket
342  return false;
343  }
344  memset(new_buckets, 0, new_size_bytes);
345  table->buckets = new_buckets;
346  table->capacity = new_capacity;
347 
348  struct HNodeGroup *group = table->first_node_group;
349 
350  while (group) {
351  int group_count;
352  if (group == table->last_node_group) {
353  group_count = group->len - table->last_node_group_avail;
354  } else {
355  group_count = group->len;
356  }
357 
358  for (int i = 0; i < group_count; i++) {
359  struct HNode *node = &group->nodes[i];
360  AtomString key = node->key;
361 
362  unsigned long hash = sdbm_hash(key, atom_string_len(key));
363  unsigned long bucket_index = hash % table->capacity;
364 
365  insert_node_into_bucket(table, bucket_index, node);
366  }
367 
368  group = group->next;
369  }
370 
371  return true;
372 }

374 static inline bool maybe_rehash(struct AtomTable *table, int new_entries)
375 {
376  int new_count = table->count + new_entries;
377  int threshold = ATOM_TABLE_THRESHOLD(table->capacity);
378  if (new_count > threshold) {
379  return false;
380  }
381 
382  int new_capacity = ATOM_TABLE_NEW_CAPACITY(new_count);
383  return do_rehash(table, new_capacity);

386 long atom_table_ensure_atom(struct AtomTable *table, AtomString string, enum AtomTableCopyOpt opts)
387 {
388  unsigned long hash = sdbm_hash(string, atom_string_len(string));
389  SMP_WRLOCK(table);
390  unsigned long bucket_index = hash % table->capacity;
391 
392  struct HNode *node = get_node_from_bucket(table, bucket_index, string);
393  if (node) {
394  SMP_UNLOCK(table);
395  return node->index;
396  }
397  if (opts & AtomTableAlreadyExisting) {
398  SMP_UNLOCK(table);
399  return ATOM_TABLE_NOT_FOUND;
400  }
401 
402  struct HNodeGroup *node_group = table->last_node_group;
403  if (!table->last_node_group_avail) {
404  node_group = new_node_group(table, DEFAULT_SIZE);
405  if (IS_NULL_PTR(node_group)) {
406  SMP_UNLOCK(table);
407  return ATOM_TABLE_ALLOC_FAIL;
408  }
409  }
410 
411  AtomString maybe_copied = string;
412  if (opts & AtomTableCopyAtom) {
413  uint8_t len = *((uint8_t *) string);
414  uint8_t *buf = malloc(1 + len);
415  if (IS_NULL_PTR(buf)) {
416  SMP_UNLOCK(table);
417  return ATOM_TABLE_ALLOC_FAIL;
418  }
419  memcpy(buf, string, 1 + len);
420  maybe_copied = buf;
421  }
422 
423  if (maybe_rehash(table, 1)) {
424  bucket_index = hash % table->capacity;
425  }
426 
427  long new_index = insert_node(table, node_group, bucket_index, maybe_copied);
428 
429  SMP_UNLOCK(table);
430  return new_index;
431 }

433 int atom_table_ensure_atoms(
434  struct AtomTable *table, const void *atoms, int count, int *translate_table)
435 {
436  SMP_WRLOCK(table);
437 
438  int new_atoms_count = 0;
439 
440  const uint8_t *current_atom = atoms;
441 
442  for (int i = 0; i < count; i++) {
443  struct HNode *node = get_node(table, current_atom);
444  if (node) {
445  translate_table[i] = node->index;
446  } else {
447  new_atoms_count++;
448  translate_table[i] = ATOM_TABLE_NOT_FOUND;
449  }
450 
451  uint8_t atom_len = current_atom[0];
452  current_atom += 1 + atom_len;
453  }
454 
455  maybe_rehash(table, new_atoms_count);
456 
457  current_atom = atoms;
458  int remaining_atoms = new_atoms_count;
459  struct HNodeGroup *node_group = table->last_node_group;
460  for (int i = 0; i < count; i++) {
461  if (translate_table[i] == ATOM_TABLE_NOT_FOUND) {
462  if (!table->last_node_group_avail) {
463  node_group = new_node_group(table, remaining_atoms);
464  if (IS_NULL_PTR(node_group)) {
465  SMP_UNLOCK(table);
466  return ATOM_TABLE_ALLOC_FAIL;
467  }
468  }
469 
470  unsigned long hash = sdbm_hash(current_atom, atom_string_len(current_atom));
471  unsigned long bucket_index = hash % table->capacity;
472 
473  translate_table[i] = insert_node(table, node_group, bucket_index, current_atom);
474  remaining_atoms--;
475  if (remaining_atoms == 0) {
476  break;
477  }
478  }
479  uint8_t atom_len = current_atom[0];
480  current_atom += 1 + atom_len;
481  }
482 
483  SMP_UNLOCK(table);
484 
485  return 0;
486 }
