41 // Convert a mailbox message (struct Message or struct TermSignal) to a heap
42 // fragment (HeapFragment) so it can be owned by the recipient.
43 // We assert this layout mapping is correct.
44 _Static_assert(offsetof(struct Message, base) + offsetof(struct MailboxMessage, next) == offsetof(HeapFragment, next) ? 1 : 0,
45  "Message.base.next doesn't match HeapFragment.next");
46 _Static_assert(offsetof(struct Message, base) + offsetof(struct MailboxMessage, type) == offsetof(HeapFragment, heap_end) ? 1 : 0,
47  "Message.base.type doesn't match HeapFragment.heap_end");
48 _Static_assert(offsetof(struct Message, message) == offsetof(HeapFragment, storage) ? 1 : 0,
49  "Message.message doesn't match HeapFragment.storage[0]");
50 _Static_assert(offsetof(struct Message, heap_end) == offsetof(HeapFragment, storage[1]) ? 1 : 0,
51  "Message.heap_end doesn't match HeapFragment.storage[1]");
52 _Static_assert(sizeof(struct Message) == sizeof(HeapFragment) + 2 * sizeof(term) ? 1 : 0,
53  "sizeof(Message) doesn't match sizeof(HeapFragment) + 2 terms");
54 _Static_assert(offsetof(struct TermSignal, base) + offsetof(struct MailboxMessage, next) == offsetof(HeapFragment, next) ? 1 : 0,
55  "TermSignal.base.next doesn't match HeapFragment.next");

57  "TermSignal.base.type doesn't match HeapFragment.heap_end");
58 _Static_assert(offsetof(struct TermSignal, signal_term) == offsetof(HeapFragment, storage) ? 1 : 0,

65 HeapFragment *mailbox_message_to_heap_fragment(void *m, term *heap_end)
66 {
67  HeapFragment *fragment = (HeapFragment *) m;
68  fragment->next = NULL; // MailboxMessage.next
69  fragment->heap_end = heap_end; // MailboxMessage.type/heap_fragment_end
70  // We don't need to erase Message.message/TermSignal.signal_term as they are valid terms
71  // Message.heap_end or TrapSignal.heap_end are not valid terms, put nil
72  fragment->storage[1] = term_nil(); // Message/TrapSignal.heap_end
73 
74  return fragment;
75 }

77 // Dispose message. Normal / signal messages are not destroyed, instead they
78 // are appended to the current heap.
79 void mailbox_message_dispose(MailboxMessage *m, Heap *heap)
80 {
81  switch (m->type) {
82  case NormalMessage: {
83  Message *normal_message = CONTAINER_OF(m, Message, base);
84  term mso_list = normal_message->storage[STORAGE_MSO_LIST_INDEX];
85  HeapFragment *fragment = mailbox_message_to_heap_fragment(normal_message, normal_message->heap_end);
86  memory_heap_append_fragment(heap, fragment, mso_list);
87  break;
88  }
89  case KillSignal:
90  case TrapAnswerSignal: {
91  struct TermSignal *term_signal = CONTAINER_OF(m, struct TermSignal, base);
92  term mso_list = term_signal->storage[STORAGE_MSO_LIST_INDEX];
93  HeapFragment *fragment = mailbox_message_to_heap_fragment(term_signal, term_signal->heap_end);
94  memory_heap_append_fragment(heap, fragment, mso_list);
95  break;
96  }
97  case ProcessInfoRequestSignal: {
98  struct BuiltInAtomRequestSignal *request_signal
99  = CONTAINER_OF(m, struct BuiltInAtomRequestSignal, base);
100  free(request_signal);
101  break;
102  }
103  case TrapExceptionSignal: {
104  struct BuiltInAtomSignal *atom_signal = CONTAINER_OF(m, struct BuiltInAtomSignal, base);
105  free(atom_signal);
106  break;
107  }
108  case FlushMonitorSignal:
109  case FlushInfoMonitorSignal: {
110  struct RefSignal *ref_signal = CONTAINER_OF(m, struct RefSignal, base);
111  free(ref_signal);
112  break;
113  }
114  case GCSignal:
115  free(m);
116  break;
117  }
118 }

120 void mailbox_destroy(Mailbox *mbox, Heap *heap)
121 {
122  MailboxMessage *msg = mbox->outer_first;
123  while (msg) {
124  MailboxMessage *next = msg->next;
125  mailbox_message_dispose(msg, heap);
126  msg = next;
127  }
128  msg = mbox->inner_first;
129  while (msg) {
130  MailboxMessage *next = msg->next;
131  mailbox_message_dispose(msg, heap);
132  msg = next;
133  }
134 }

152 size_t mailbox_size(Mailbox *mbox)
153 {
154  size_t result = 0;
155  MailboxMessage *msg = mbox->outer_first;
156  while (msg) {
157  // We don't count signals.
158  if (msg->type == NormalMessage) {
159  Message *normal_message = CONTAINER_OF(msg, Message, base);
160  result += sizeof(Message) + normal_message->heap_end - normal_message->storage;
161  }
162  msg = msg->next;
163  }
164  msg = mbox->inner_first;
165  while (msg) {
166  Message *normal_message = CONTAINER_OF(msg, Message, base);
167  result += sizeof(Message) + normal_message->heap_end - normal_message->storage;
168  msg = msg->next;
169  }
170  return result;
171 }

173 static void mailbox_post_message(Context *c, MailboxMessage *m)
174 {
175  m->next = NULL;
176 
177  // Append message at the beginning of outer_first.
178 #ifndef AVM_NO_SMP
179  MailboxMessage *current_first = NULL;
180  do {
181  m->next = current_first;
182  } while (!ATOMIC_COMPARE_EXCHANGE_WEAK(&c->mailbox.outer_first, &current_first, m));
183 #else
184  m->next = c->mailbox.outer_first;
185  c->mailbox.outer_first = m;
186 #endif
187 
188  scheduler_signal_message(c);
189 }

191 void mailbox_send(Context *c, term t)
192 {
193  unsigned long estimated_mem_usage = memory_estimate_usage(t) + 1; // mso_list
194 
195  Message *msg = malloc(sizeof(Message) + estimated_mem_usage * sizeof(term));
196  if (IS_NULL_PTR(msg)) {
197  fprintf(stderr, "Failed to allocate memory: %s:%i.\n", __FILE__, __LINE__);
198  return;
199  }
200  msg->base.type = NormalMessage;
201  msg->message = memory_copy_term_tree_to_storage(msg->storage, &msg->heap_end, t);
202 
203  TRACE("Sending %p to pid %i\n", (void *) msg->message, c->process_id);
204 
205  mailbox_post_message(c, &msg->base);
206 }

282 MailboxMessage *mailbox_process_outer_list(Mailbox *mbox)
283 {
284  // Empty outer list using CAS
285  MailboxMessage *current = mbox->outer_first;
286 #ifndef AVM_NO_SMP
287  while (!ATOMIC_COMPARE_EXCHANGE_WEAK(&mbox->outer_first, &current, NULL)) {
288  };
289 #else
290  mbox->outer_first = NULL;
291 #endif
292  // Reverse the list
293  MailboxMessage *previous_normal = NULL;
294  MailboxMessage *previous_signal = NULL;
295  MailboxMessage *last_normal = NULL;
296  while (current) {
297  MailboxMessage *next = current->next;
298  if (current->type == NormalMessage) {
299  // Get last normal to update inner_last.
300  if (last_normal == NULL) {
301  last_normal = current;
302  }
303  current->next = previous_normal;
304  previous_normal = current;
305  } else {
306  current->next = previous_signal;
307  previous_signal = current;
308  }
309  current = next;
310  }
311  // If we did enqueue some normal messages, lastNormal is the first
312  // one in outer list (last received one)
313  if (last_normal) {
314  // previousNormal is new list head
315  // If we had no receive_pointer, it should be this list head
316  if (mbox->receive_pointer == NULL) {
317  mbox->receive_pointer = previous_normal;
318  // If we had a prev, set the prev's next to the new current.
319  if (mbox->receive_pointer_prev) {
320  mbox->receive_pointer_prev->next = previous_normal;
321  } else if (mbox->inner_first == NULL) {
322  // If we had no first, this is the first message.
323  mbox->inner_first = previous_normal;
324  }
325  }
326 
327  // Update last and previous last's next.
328  // Append these new items at the end of inner list.
329  if (mbox->inner_last) {
330  // This may be mbox->receive_pointer_prev which we
331  // are updating a second time here.
332  mbox->inner_last->next = previous_normal;
333  }
334  mbox->inner_last = last_normal;
335  }
336 
337  return previous_signal;
338 }

340 void mailbox_next(Mailbox *mbox)
341 {
342  // This is called from OP_LOOP_REC_END opcode, so we cannot make any
343  // assumption about the state and should perform a nop if moving cursor
344  // beyond last position.
345  if (UNLIKELY(mbox->receive_pointer == NULL)) {
346  fprintf(stderr, "OP_LOOP_REC_END beyond mailbox end\n");
347  return;
348  }
349 
350  mbox->receive_pointer_prev = mbox->receive_pointer;
351  mbox->receive_pointer = mbox->receive_pointer->next;
352 }

370 MailboxMessage *mailbox_take_message(Mailbox *mbox)
371 {
372  // This is called from OP_REMOVE_MESSAGE opcode, so we cannot make any
373  // assumption about the state and should perform a nop if the mailbox
374  // is empty.
375  if (UNLIKELY(mbox->receive_pointer == NULL)) {
376  fprintf(stderr, "OP_REMOVE_MESSAGE on empty mailbox\n");
377  return NULL;
378  }
379  MailboxMessage *removed = mbox->receive_pointer;
380  if (mbox->receive_pointer_prev) {
381  // We did not remove first message.
382  mbox->receive_pointer_prev->next = removed->next;
383  // If we removed last messages, update inner last.
384  if (mbox->inner_last == removed) {
385  mbox->inner_last = mbox->receive_pointer_prev;
386  }
387  } else {
388  // We did remove first message.
389  mbox->inner_first = removed->next;
390  if (mbox->inner_first == NULL) {
391  // If this also the last, update inner_last.
392  mbox->inner_last = NULL;
393  }
394  }
395 
396  // Reset receive pointers
397  mailbox_reset(mbox);
398 
399  return removed;
400 }
