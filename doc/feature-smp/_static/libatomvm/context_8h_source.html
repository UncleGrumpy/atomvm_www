238 static inline term *context_stack_base(const Context *ctx)
239 {
240  // Find which fragment the stack belongs to.
241  if (ctx->e >= ctx->heap.heap_start && ctx->e <= ctx->heap.heap_end) {
242  return ctx->heap.heap_end;
243  }
244  HeapFragment *fragment = ctx->heap.root->next;
245  while (fragment) {
246  if (ctx->e >= fragment->storage && ctx->e <= fragment->heap_end) {
247  return fragment->heap_end;
248  }
249  fragment = fragment->next;
250  }
251  fprintf(stderr, "Context heap is corrupted, cannot find fragment with stack pointer.\n");
252  AVM_ABORT();
253 }
