90 void port_ensure_available(Context *ctx, size_t size)
91 {
92  if (context_avail_free_memory(ctx) < size) {
93  switch (memory_ensure_free(ctx, size)) {
94  case MEMORY_GC_OK:
95  break;
96  case MEMORY_GC_ERROR_FAILED_ALLOCATION:
97  // TODO Improve error handling
98  fprintf(stderr, "Failed to allocate additional heap storage: [%s:%i]\n", __FILE__, __LINE__);
99  AVM_ABORT();
100  case MEMORY_GC_DENIED_ALLOCATION:
101  // TODO Improve error handling
102  fprintf(stderr, "Not permitted to allocate additional heap storage: [%s:%i]\n", __FILE__, __LINE__);
103  AVM_ABORT();
104  }
105  }
106 }

108 int port_is_standard_port_command(term t)
109 {
110  if (!term_is_tuple(t)) {
111  return 0;
112  } else if (term_get_tuple_arity(t) != 3) {
113  return 0;
114  } else {
115  term pid = term_get_tuple_element(t, 0);
116  term ref = term_get_tuple_element(t, 1);
117  if (!term_is_pid(pid)) {
118  return 0;
119  } else if (!term_is_reference(ref)) {
120  return 0;
121  } else {
122  return 1;
123  }
124  }
125 }
