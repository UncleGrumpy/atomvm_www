59 char *interop_list_to_string(term list, int *ok)
60 {
61  int proper;
62  int len = term_list_length(list, &proper);
63  if (UNLIKELY(!proper)) {
64  *ok = 0;
65  return NULL;
66  }
67 
68  char *str = malloc(len + 1);
69  if (IS_NULL_PTR(str)) {
70  return NULL;
71  }
72 
73  term t = list;
74  for (int i = 0; i < len; i++) {
75  term byte_value_term = term_get_list_head(t);
76  if (UNLIKELY(!term_is_integer(byte_value_term))) {
77  *ok = 0;
78  free(str);
79  return NULL;
80  }
81 
82  if (UNLIKELY(!term_is_uint8(byte_value_term))) {
83  *ok = 0;
84  free(str);
85  return NULL;
86  }
87  uint8_t byte_value = term_to_uint8(byte_value_term);
88 
89  str[i] = (char) byte_value;
90  t = term_get_list_tail(t);
91  }
92  str[len] = 0;
93 
94  *ok = 1;
95  return str;
96 }

118 term interop_proplist_get_value_default(term list, term key, term default_value)
119 {
120  term t = list;
121 
122  while (!term_is_nil(t)) {
123  term *t_ptr = term_get_list_ptr(t);
124 
125  term head = t_ptr[1];
126  if (term_is_tuple(head) && term_get_tuple_arity(head) == 2 && term_get_tuple_element(head, 0) == key) {
127  return term_get_tuple_element(head, 1);
128 
129  } else if (term_is_atom(head)) {
130  if (head == key) {
131  return TRUE_ATOM;
132  }
133  }
134 
135  t = *t_ptr;
136  }
137 
138  return default_value;
139 }

141 InteropFunctionResult interop_iolist_size(term t, size_t *size)
142 {
143  if (term_is_binary(t)) {
144  *size = term_binary_size(t);
145  return InteropOk;
146  }
147 
148  if (UNLIKELY(!term_is_list(t))) {
149  return InteropBadArg;
150  }
151 
152  unsigned long acc = 0;
153 
154  struct TempStack temp_stack;
155  if (UNLIKELY(temp_stack_init(&temp_stack) != TempStackOk)) {
156  return InteropMemoryAllocFail;
157  }
158 
159  if (UNLIKELY(temp_stack_push(&temp_stack, t) != TempStackOk)) {
160  temp_stack_destroy(&temp_stack);
161  return InteropMemoryAllocFail;
162  }
163 
164  while (!temp_stack_is_empty(&temp_stack)) {
165  if (term_is_integer(t)) {
166  acc++;
167  t = temp_stack_pop(&temp_stack);
168 
169  } else if (term_is_nil(t)) {
170  t = temp_stack_pop(&temp_stack);
171 
172  } else if (term_is_nonempty_list(t)) {
173  if (UNLIKELY(temp_stack_push(&temp_stack, term_get_list_tail(t)) != TempStackOk)) {
174  temp_stack_destroy(&temp_stack);
175  return InteropMemoryAllocFail;
176  }
177  t = term_get_list_head(t);
178 
179  } else if (term_is_binary(t)) {
180  acc += term_binary_size(t);
181  t = temp_stack_pop(&temp_stack);
182 
183  } else {
184  temp_stack_destroy(&temp_stack);
185  return InteropBadArg;
186  }
187  }
188 
189  temp_stack_destroy(&temp_stack);
190 
191  *size = acc;
192  return InteropOk;
193 }

195 InteropFunctionResult interop_write_iolist(term t, char *p)
196 {
197  if (term_is_binary(t)) {
198  int len = term_binary_size(t);
199  memcpy(p, term_binary_data(t), len);
200  return InteropOk;
201  }
202 
203  struct TempStack temp_stack;
204  if (UNLIKELY(temp_stack_init(&temp_stack) != TempStackOk)) {
205  return InteropMemoryAllocFail;
206  }
207 
208  if (UNLIKELY(temp_stack_push(&temp_stack, t) != TempStackOk)) {
209  temp_stack_destroy(&temp_stack);
210  return InteropMemoryAllocFail;
211  }
212 
213  while (!temp_stack_is_empty(&temp_stack)) {
214  if (term_is_integer(t)) {
215  *p = term_to_int(t);
216  p++;
217  t = temp_stack_pop(&temp_stack);
218 
219  } else if (term_is_nil(t)) {
220  t = temp_stack_pop(&temp_stack);
221 
222  } else if (term_is_nonempty_list(t)) {
223  if (UNLIKELY(temp_stack_push(&temp_stack, term_get_list_tail(t)) != TempStackOk)) {
224  temp_stack_destroy(&temp_stack);
225  return InteropMemoryAllocFail;
226  }
227  t = term_get_list_head(t);
228 
229  } else if (term_is_binary(t)) {
230  int len = term_binary_size(t);
231  memcpy(p, term_binary_data(t), len);
232  p += len;
233  t = temp_stack_pop(&temp_stack);
234 
235  } else {
236  temp_stack_destroy(&temp_stack);
237  return InteropBadArg;
238  }
239  }
240 
241  temp_stack_destroy(&temp_stack);
242  return InteropOk;
243 }
