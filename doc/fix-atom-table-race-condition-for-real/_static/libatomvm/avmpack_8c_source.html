36 int avmpack_is_valid(const void *avmpack_binary, uint32_t size)
37 {

39  0x23, 0x21, 0x2f, 0x75,

41  0x69, 0x6e, 0x2f, 0x65,
42  0x6e, 0x76, 0x20, 0x41,
43  0x74, 0x6f, 0x6d, 0x56,
44  0x4d, 0x0a, 0x00, 0x00
45  };
46 
47  if (UNLIKELY(size < 24)) {
48  return 0;

54 int avmpack_find_section_by_flag(const void *avmpack_binary, uint32_t flags_mask, const void **ptr, uint32_t *size, const char **name)
55 {
56  int offset = AVMPACK_SIZE;
57  const uint32_t *flags;
58 
59  do {
60  const uint32_t *sizes = ((const uint32_t *) (avmpack_binary)) + offset / sizeof(uint32_t);
61  flags = ((const uint32_t *) (avmpack_binary)) + 1 + offset / sizeof(uint32_t);
62 
63  if ((ENDIAN_SWAP_32(*flags) & flags_mask) == flags_mask) {
64  const char *found_section_name = (const char *) (sizes + 3);
65  int section_name_len = pad(strlen(found_section_name) + 1);
66 
67  *ptr = sizes + 3 + section_name_len / sizeof(uint32_t);
68  *size = ENDIAN_SWAP_32(*sizes);
69  *name = (const char *) (sizes + 3);
70  return 1;
71  }
72 
73  offset += ENDIAN_SWAP_32(*sizes);
74 
75  } while (*flags);
76 
77  return 0;
78 }

105 void *avmpack_fold(void *accum, const void *avmpack_binary, avmpack_fold_fun fold_fun)
106 {
107  int offset = AVMPACK_SIZE;
108  uint32_t size = 0;
109 
110  do {
111  const uint32_t *size_ptr = ((const uint32_t *) (avmpack_binary)) + offset / sizeof(uint32_t);
112  size = ENDIAN_SWAP_32(*size_ptr);
113  if (size > 0) {
114  const uint32_t *flags_ptr = size_ptr + 1;
115  uint32_t flags = ENDIAN_SWAP_32(*flags_ptr);
116  const char *section_name = (const char *) (size_ptr + 3);
117  int section_name_len = pad(strlen(section_name) + 1);
118  accum = fold_fun(
119  accum,
120  size_ptr, size,
121  size_ptr + 3 + section_name_len / sizeof(uint32_t),
122  flags,
123  section_name);
124  offset += size;
125  }
126  } while (size > 0);
127 
128  return accum;
129 }
